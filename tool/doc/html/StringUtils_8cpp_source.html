121 size_t StringUtils::count(const string &in, const string &what)
122 {
123  size_t sizeof_what = what.size();
124  if(sizeof_what == 0)
125  return 0;
126  size_t count = 0;
127  size_t pos = in.find(what);
128  while(pos != string::npos)
129  {
130  ++count;
131  pos = in.find(what, pos + sizeof_what);
132  }
133  return count;
134 }

144 // -------------------------------------------------------------------------------------------
145 void StringUtils::replaceAllIn(string &in, const string &from, const string &to)
146 {
147  size_t sizeof_from = from.size();
148  if(sizeof_from == 0)
149  return;
150  size_t sizeof_to = to.size();
151  size_t pos = in.find(from);
152  while(pos != string::npos)
153  {
154  in.replace(pos, sizeof_from, to);
155  pos = in.find(from, pos + sizeof_to);
156  }
157 }

160 void StringUtils::tokenize(const string &in, vector<string> &tok, const string &delim)
161 {
162  size_t start = in.find_first_not_of(delim, 0);
163  size_t end = in.find_first_of(delim, start);
164  while(start != string::npos || end != string::npos)
165  {
166  tok.push_back(in.substr(start, end - start));
167  start = in.find_first_not_of(delim, end);
168  end = in.find_first_of(delim, start);
169  }
170 }
