45 CNF::CNF(const string &filename)
46 {
47  ifstream in_file(filename.c_str());
48  MASSERT(!in_file.fail(), "Could not open DIAMCS file '" + filename + "'.");
49  size_t nr_of_clauses = 0;
50 
51  in_file >> std::ws; // consume all whitespaces
52 
53  char c = 0;
54  while(in_file.good())
55  {
56  c = in_file.peek();
57  // skipping comments:
58  if(c == 'c')
59  while(in_file.good() && in_file.get() != '\n');
60  else if(c == 'p')
61  {
62  // parsing the header:
63  int max_var = 0;
64  string buf1, buf2;
65  in_file >> buf1 >> buf2 >> max_var >> nr_of_clauses;
66  MASSERT(!in_file.fail(), "Syntax error in header of DIMACS file.");
67  MASSERT(buf1 == "p" && buf2 == "cnf", "Syntax error in header of DIMACS file.");
68  MASSERT(max_var <= VarManager::instance().getMaxCNFVar(), "CNF contains unknown var.");
69  }
70  else
71  {
72  // parsing a clause:
73  vector<int> clause;
74  int literal = 0;
75  MASSERT(in_file.good(), "Unexpected end-of-file.");
76  in_file >> literal;
77  while(literal != 0)
78  {
79  clause.push_back(literal);
80  MASSERT(in_file.good(), "Unexpected end-of-file.");
81  in_file >> literal;
82  }
83  clauses_.push_back(clause);
84  }
85  in_file >> std::ws; // consume all whitespaces
86  }
87  in_file.close();
88  MASSERT(clauses_.size() == nr_of_clauses, "Mismatch in nr of clauses");
89 }

147 {
148  // search for clauses which are supersets of the new clause:

161  //L_DBG("CNF size reduction: " << init_size << " --> " << clauses_.size());

272 void CNF::negate()
273 {
274  list<vector<int> > original;
275  clauses_.swap(original);
276  vector<int> one_clause_false;
277  one_clause_false.reserve(original.size() + 1);
278  for(ClauseConstIter it = original.begin(); it != original.end(); ++it)
279  {
280  if(it->size() == 1)
281  one_clause_false.push_back(-(*it)[0]);
282  else
283  {
284  int clause_false_lit = VarManager::instance().createFreshTmpVar("tmpNegClauseTrue");
285  one_clause_false.push_back(clause_false_lit);
286  for(size_t lit_cnt = 0; lit_cnt < it->size(); ++lit_cnt)
287  add2LitClause(-clause_false_lit, -((*it)[lit_cnt]));
288  }
289  }
290  clauses_.push_back(one_clause_false);
291 }

366 void CNF::simplify()
367 {
368  // search for clauses which are supersets of other clauses:
369  size_t init_size = clauses_.size();
370  for(ClauseIter it1 = clauses_.begin(); it1 != clauses_.end();)
371  {
372  ClauseIter it2 = it1;
373  ++it2;
374  bool it1_changed = false;
375  for(; it2 != clauses_.end();)
376  {
377  if(Utils::isSubset(*it1, *it2))
378  it2 = clauses_.erase(it2);
379  else
380  ++it2;
381  if(Utils::isSubset(*it2, *it1))
382  {
383  it1 = clauses_.erase(it1);
384  it1_changed = true;
385  break;
386  }
387  }
388  if(!it1_changed)
389  ++it1;
390  }
391  if(init_size != clauses_.size())
392  {
393  L_DBG("CNF size reduction: " << init_size << " --> " << clauses_.size());
394  }
395 }

433 void CNF::setVarValue(int var, bool value)
434 {
435  for(ClauseIter it = clauses_.begin(); it != clauses_.end();)
436  {
437  bool clause_removed = false;
438  for(size_t lit_cnt = 0; lit_cnt < it->size(); )
439  {
440  int lit = (*it)[lit_cnt];
441  if((lit == var && value == false) || (-lit == var && value == true))
442  {
443  // the literal is replaced by false, i.e., removed from the clause
444  (*it)[lit_cnt] = it->back();
445  it->pop_back();
446  if(it->empty()) // if a clause gets empty, then the entire CNF is false:
447  {
448  clauses_.clear();
449  clauses_.push_back(vector<int>());
450  return;
451  }
452  }
453  else if ((lit == var && value == true) || (-lit == var && value == false))
454  {
455  // the literal is replaced by true, i.e., the clause is removed
456  it = clauses_.erase(it);
457  clause_removed = true;
458  break;
459  }
460  else
461  lit_cnt++;
462  }
463  if(!clause_removed)
464  ++it;
465  }
466 }

468 // -------------------------------------------------------------------------------------------
469 void CNF::doPureAndUnit(const vector<int> &keep)
470 {
471  set<int> unit;
472  bool unit_changed = true;
473  while(unit_changed)
474  {
475  unit_changed = false;
476  for(ClauseIter it = clauses_.begin(); it != clauses_.end();)
477  {
478  bool erase = false;
479  if(it->size() == 1)
480  {
481  unit.insert(it->front());
482  erase = true;
483  unit_changed = true;
484  }
485  else if(!unit.empty())
486  {
487  for(size_t c = 0; c < it->size(); )
488  {
489  int lit = (*it)[c];
490  if(unit.count(lit)) // this clause is true, remove it:
491  {
492  erase = true;
493  break;
494  }
495  else if(unit.count(-lit)) // this literal is false, remove it:
496  {
497  (*it)[c] = it->back();
498  it->pop_back();
499  if(it->size() == 1)
500  {
501  unit.insert(it->front());
502  erase = true;
503  unit_changed = true;
504  break;
505  }
506  }
507  else
508  ++c;
509  }
510  }
511  if(erase)
512  it = clauses_.erase(it);
513  else
514  ++it;
515  }
516  }
517 
518  for(size_t cnt = 0; cnt < keep.size(); ++cnt)
519  {
520  if(unit.count(keep[cnt]) != 0)
521  add1LitClause(keep[cnt]);
522  else if(unit.count(-keep[cnt]) != 0)
523  add1LitClause(-keep[cnt]);
524  }
525 }
