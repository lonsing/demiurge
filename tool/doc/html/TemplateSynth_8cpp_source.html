57 bool TemplateSynth::run()
58 {
59  //L_DBG(VarManager::instance().toString());
60  //L_DBG("TRANS:");
61  //L_DBG(AIG2CNF::instance().getTrans().toString());
62  L_INF("Starting to compute a winning region ...");
63  bool realizable = computeWinningRegion();
64  if(!realizable)
65  {
66  L_RES("The specification is unrealizable.");
67  return false;
68  }
69  L_RES("The specification is realizable.");
70  Utils::debugCheckWinReg(winning_region_);
71  if(Options::instance().doRealizabilityOnly())
72  return true;
73 
74  L_INF("Starting to extract a circuit ...");
75  QBFCertImplExtractor extractor;
76  extractor.extractCircuit(winning_region_);
77  L_INF("Synthesis done.");
78  return true;
79 }

82 bool TemplateSynth::computeWinningRegion()
83 {
84  size_t nr_of_clauses = 1;
85  size_t nr_of_bits = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE).size() - 1;
86  // (-1 because we do not use the error-bit in our state-space)
87  size_t max_nr_clauses = numeric_limits<size_t>::max();
88  if(nr_of_bits < numeric_limits<size_t>::digits - 1)
89  max_nr_clauses = (1UL << nr_of_bits);
90  while(true)
91  {
92  VarManager::push();
93  L_INF("Trying CNF template with " << nr_of_clauses << " clauses ...");
94  bool found = findWinRegCNFTempl(nr_of_clauses);
95  VarManager::pop(); // to eliminate all the temporary variables again.
96  if(found)
97  {
98  L_LOG("Found winning region with " << nr_of_clauses << " clauses.");
99  return true;
100  }
101  L_LOG("No winning region with " << nr_of_clauses << " clauses.");
102  if(nr_of_clauses >= max_nr_clauses)
103  return false;
104  if(nr_of_clauses == 1)
105  nr_of_clauses = 4;
106  else if(nr_of_clauses == 4)
107  nr_of_clauses = 16;
108  else
109  nr_of_clauses <<= 1;
110  }
111  return false;
112 }

115 bool TemplateSynth::findWinRegCNFTempl(size_t nr_of_clauses)
116 {
117  // Note: a lot of loops and code could be merged, but readability
118  // beats performance in this case.
119 
120  AIG2CNF& A2C = AIG2CNF::instance();
121  VarManager &VM = VarManager::instance();
122  vector<int> ps_vars = VM.getVarsOfType(VarInfo::PRES_STATE);
123  vector<int> ns_vars = VM.getVarsOfType(VarInfo::NEXT_STATE);
124 
125  // ps_vars[0] and ns_vars[0] are the error bits, which must be FALSE always.
126  // Hence, we do not include these signals in our template
127  ps_vars[0] = ps_vars[ps_vars.size() - 1];
128  ps_vars.pop_back();
129  ns_vars[0] = ns_vars[ns_vars.size() - 1];
130  ns_vars.pop_back();
131  size_t nr_of_vars = ps_vars.size();
132 
133  // Step 1: build the parameterized CNF:
134  int w1 = VM.createFreshTmpVar("w1");
135  CNF win_eq_w1;
136  int w2 = VM.createFreshTmpVar("w2");
137  CNF next_win_eq_w2;
138 
139  // Step 1a: create template parameters:
140  vector<int> clause_active_vars;
141  vector<vector<int> > contains_vars;
142  vector<vector<int> > negated_vars;
143  clause_active_vars.reserve(nr_of_clauses);
144  contains_vars.reserve(nr_of_clauses);
145  negated_vars.reserve(nr_of_clauses);
146  for(size_t c_cnt = 0; c_cnt < nr_of_clauses; ++c_cnt)
147  {
148  ostringstream act_name;
149  act_name << "act[" << c_cnt << "]";
150  clause_active_vars.push_back(VM.createFreshTemplParam(act_name.str()));
151  contains_vars.push_back(vector<int>());
152  negated_vars.push_back(vector<int>());
153  for(size_t v_cnt = 0; v_cnt < nr_of_vars; ++v_cnt)
154  {
155  ostringstream var_idx;
156  var_idx << "[" << c_cnt << "][" << v_cnt << "]";
157  contains_vars[c_cnt].push_back(VM.createFreshTemplParam("cont" + var_idx.str()));
158  negated_vars[c_cnt].push_back(VM.createFreshTemplParam("cont" + var_idx.str()));
159  }
160  }
161 
162  // Step 1b: constructing the constraints:
163  // We do this for next_win_eq_w2 and win_eq_w1 in parallel
164  vector<int> all_clause_lits1;
165  vector<int> all_clause_lits2;
166  all_clause_lits1.reserve(nr_of_clauses + 1);
167  all_clause_lits2.reserve(nr_of_clauses + 1);
168  // We add one fixed clause, namely that we are inside the safe states:
169  all_clause_lits1.push_back(-VM.getPresErrorStateVar());
170  all_clause_lits2.push_back(-VM.getNextErrorStateVar());
171  for(size_t c = 0; c < nr_of_clauses; ++c)
172  {
173  int clause_lit1 = VM.createFreshTmpVar(); // TRUE iff clause c is TRUE
174  int clause_lit2 = VM.createFreshTmpVar(); // TRUE iff clause c is TRUE
175  all_clause_lits1.push_back(clause_lit1);
176  all_clause_lits2.push_back(clause_lit2);
177  vector<int> literals_in_clause1;
178  vector<int> literals_in_clause2;
179  literals_in_clause1.reserve(nr_of_vars + 2);
180  literals_in_clause2.reserve(nr_of_vars + 2);
181  // if clause_active_var[c]=FALSE, then the clause should be TRUE
182  // so we add -clause_active_vars[c] as if it was a literal of this clause
183  literals_in_clause1.push_back(-clause_active_vars[c]);
184  literals_in_clause2.push_back(-clause_active_vars[c]);
185  for(size_t v = 0; v < nr_of_vars; ++v)
186  {
187  int activated_var1 = VM.createFreshTmpVar();
188  int activated_var2 = VM.createFreshTmpVar();
189  literals_in_clause1.push_back(activated_var1);
190  literals_in_clause2.push_back(activated_var2);
191  // Part A: activated_var should be equal to
192  // - FALSE if (contains_var[c][v]==FALSE)
193  // - not(var[v]) if (contains_var[c][v]==TRUE and negated_var[c][v]==TRUE)
194  // - var[v] if (contains_var[c][v]==TRUE and negated_var[c][v]==FALSE)
195  // Part A1: not(contains_vars[c][v]) implies not(activated_var)
196  win_eq_w1.add2LitClause(contains_vars[c][v], -activated_var1);
197  next_win_eq_w2.add2LitClause(contains_vars[c][v], -activated_var2);
198  // Part A2: (contains_var[c][v] and negated_var[c][v]) implies (activated_var <=> not(var[v]))
199  win_eq_w1.add4LitClause(-contains_vars[c][v], -negated_vars[c][v], ps_vars[v], activated_var1);
200  win_eq_w1.add4LitClause(-contains_vars[c][v], -negated_vars[c][v], -ps_vars[v], -activated_var1);
201  next_win_eq_w2.add4LitClause(-contains_vars[c][v], -negated_vars[c][v], ns_vars[v], activated_var2);
202  next_win_eq_w2.add4LitClause(-contains_vars[c][v], -negated_vars[c][v], -ns_vars[v], -activated_var2);
203  // Part A3: (contains_var[c][v] and not(negated_var[c][v])) implies (activated_var <=> var[v])
204  win_eq_w1.add4LitClause(-contains_vars[c][v], negated_vars[c][v], -ps_vars[v], activated_var1);
205  win_eq_w1.add4LitClause(-contains_vars[c][v], negated_vars[c][v], ps_vars[v], -activated_var1);
206  next_win_eq_w2.add4LitClause(-contains_vars[c][v], negated_vars[c][v], -ns_vars[v], activated_var2);
207  next_win_eq_w2.add4LitClause(-contains_vars[c][v], negated_vars[c][v], ns_vars[v], -activated_var2);
208  }
209  // Part B: clause_lit should be equal to OR(literals_in_clause):
210  // Part B1: if one literal in literals_in_clause is TRUE then clause_lit should be TRUE
211  for(size_t cnt = 0; cnt < literals_in_clause1.size(); ++cnt)
212  {
213  win_eq_w1.add2LitClause(-literals_in_clause1[cnt], clause_lit1);
214  next_win_eq_w2.add2LitClause(-literals_in_clause2[cnt], clause_lit2);
215  }
216  // Part B2: if all literals in literals_in_clause are FALSE, then the clause_lit should be FALSE
217  // That is, we add the clause [l1, l2, l3, l4, -clause_lit]
218  literals_in_clause1.push_back(-clause_lit1);
219  win_eq_w1.addClause(literals_in_clause1);
220  literals_in_clause2.push_back(-clause_lit2);
221  next_win_eq_w2.addClause(literals_in_clause2);
222  }
223  // Part C: w should be equal to AND(all_clause_lits)
224  // Part C1: if one literal in all_clause_lits is FALSE then w should be FALSE:
225  for(size_t cnt = 0; cnt < all_clause_lits1.size(); ++cnt)
226  {
227  win_eq_w1.add2LitClause(all_clause_lits1[cnt], -w1);
228  next_win_eq_w2.add2LitClause(all_clause_lits2[cnt], -w2);
229  }
230  // Part C2: if all literals in all_clause_lits are TRUE, the w should be TRUE:
231  // That is, we create a clause [-l1, -l2, -l3, -l4, w]
232  for(size_t cnt = 0; cnt < all_clause_lits1.size(); ++cnt)
233  {
234  all_clause_lits1[cnt] = -all_clause_lits1[cnt];
235  all_clause_lits2[cnt] = -all_clause_lits2[cnt];
236  }
237  all_clause_lits1.push_back(w1);
238  all_clause_lits2.push_back(w2);
239  win_eq_w1.addClause(all_clause_lits1);
240  next_win_eq_w2.addClause(all_clause_lits2);
241 
242 
243  //Step 2: Solve:
244  CNF query(A2C.getTransEqT());
245  query.addCNF(win_eq_w1);
246  query.addCNF(next_win_eq_w2);
247  vector<int> init_implies_win;
248  const vector<int> &all_ps_vars = VM.getVarsOfType(VarInfo::PRES_STATE);
249  init_implies_win.reserve(all_ps_vars.size() + 1);
250  for(size_t cnt = 0; cnt < all_ps_vars.size(); ++cnt)
251  init_implies_win.push_back(all_ps_vars[cnt]);
252  init_implies_win.push_back(w1);
253  query.addClause(init_implies_win);
254  // from win, the system can enforce to stay in win
255  query.add2LitClause(-w1, A2C.getT());
256  query.add2LitClause(-w1, w2);
257 
258  // build the quantifier prefix:
259  vector<pair<VarInfo::VarKind, QBFSolver::Quant> > quant;
260  quant.push_back(make_pair(VarInfo::TEMPL_PARAMS, QBFSolver::E));
261  quant.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::A));
262  quant.push_back(make_pair(VarInfo::INPUT, QBFSolver::A));
263  quant.push_back(make_pair(VarInfo::CTRL, QBFSolver::E));
264  quant.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
265  quant.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
266 
267  vector<int> model;
268  bool sat = qbf_solver_->isSatModel(quant, query, model);
269  if(!sat)
270  return false;
271 
272  // Step 3: Build a CNF for the winning region:
273  winning_region_.add1LitClause(-VM.getPresErrorStateVar());
274  vector<bool> can_be_0(VM.getMaxCNFVar() + 1, true);
275  vector<bool> can_be_1(VM.getMaxCNFVar() + 1, true);
276  for(size_t v_cnt = 0; v_cnt < model.size(); ++v_cnt)
277  {
278  if(model[v_cnt] < 0)
279  can_be_1[-model[v_cnt]] = false;
280  else
281  can_be_0[model[v_cnt]] = false;
282  }
283  for(size_t c_cnt = 0; c_cnt < nr_of_clauses; ++c_cnt)
284  {
285  if(can_be_0[clause_active_vars[c_cnt]])
286  continue;
287  vector<int> clause;
288  clause.reserve(nr_of_vars);
289  for(size_t v_cnt = 0; v_cnt < nr_of_vars; ++v_cnt)
290  {
291  if(can_be_0[contains_vars[c_cnt][v_cnt]])
292  continue;
293  if(can_be_0[negated_vars[c_cnt][v_cnt]])
294  clause.push_back(ps_vars[v_cnt]);
295  else
296  clause.push_back(-ps_vars[v_cnt]);
297  }
298  winning_region_.addClause(clause);
299  }
300  return true;
301 }
