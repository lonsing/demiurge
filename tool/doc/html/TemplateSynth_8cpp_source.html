102 bool TemplateSynth::computeWinningRegionCNF()
103 {
104  size_t nr_of_clauses = 1;
105  size_t nr_of_bits = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE).size() - 1;
106  // (-1 because we do not use the error-bit in our state-space)
107  size_t max_nr_clauses = numeric_limits<size_t>::max();
108  if(nr_of_bits < numeric_limits<size_t>::digits - 1)
109  max_nr_clauses = (1UL << nr_of_bits);
110  while(true)
111  {
112  VarManager::resetToLastPush();
113  L_INF("Trying CNF template with " << nr_of_clauses << " clauses ...");
114  bool found = findWinRegCNFTempl(nr_of_clauses);
115  if(found)
116  {
117  L_LOG("Found winning region with " << nr_of_clauses << " clauses.");
118  return true;
119  }
120  L_LOG("No winning region with " << nr_of_clauses << " clauses.");
121  if(nr_of_clauses >= max_nr_clauses)
122  return false;
123  if(nr_of_clauses < 4)
124  nr_of_clauses++;
125  else
126  nr_of_clauses <<= 1;
127  }
128  return false;
129 }

133 {
134  size_t nr_of_gates = 1;
135  size_t nr_of_bits = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE).size() - 1;

137  size_t max_nr_gates = numeric_limits<size_t>::max();
138  if(nr_of_bits < numeric_limits<size_t>::digits - 1)
139  max_nr_gates = (1UL << nr_of_bits);
140  while(true)

144  bool found = findWinRegANDNetwork(nr_of_gates);
145  if(found)
146  {

152  return false;
153  if(nr_of_gates < 4)
154  nr_of_gates++;
155  else
156  nr_of_gates <<= 1;
157  }
158  return false;

161 // -------------------------------------------------------------------------------------------
162 bool TemplateSynth::findWinRegCNFTempl(size_t nr_of_clauses)
163 {
164  // Note: a lot of loops and code could be merged, but readability
165  // beats performance in this case.
166 
167  VarManager &VM = VarManager::instance();
168  vector<int> ps_vars = VM.getVarsOfType(VarInfo::PRES_STATE);
169  vector<int> ns_vars = VM.getVarsOfType(VarInfo::NEXT_STATE);
170 
171  // ps_vars[0] and ns_vars[0] are the error bits, which must be FALSE always.
172  // Hence, we do not include these signals in our template
173  ps_vars[0] = ps_vars[ps_vars.size() - 1];
174  ps_vars.pop_back();
175  ns_vars[0] = ns_vars[ns_vars.size() - 1];
176  ns_vars.pop_back();
177  size_t nr_of_vars = ps_vars.size();
178 
179  // Step 1: build the parameterized CNF:
180  CNF win_constr;
181  int w1 = VM.createFreshTmpVar("w1");
182  int w2 = VM.createFreshTmpVar("w2");
183 
184  // Step 1a: create template parameters:
185  vector<int> clause_active_vars;
186  vector<vector<int> > contains_vars;
187  vector<vector<int> > negated_vars;
188  clause_active_vars.reserve(nr_of_clauses);
189  contains_vars.reserve(nr_of_clauses);
190  negated_vars.reserve(nr_of_clauses);
191  for(size_t c_cnt = 0; c_cnt < nr_of_clauses; ++c_cnt)
192  {
193  ostringstream act_name;
194  act_name << "act[" << c_cnt << "]";
195  clause_active_vars.push_back(VM.createFreshTemplParam(act_name.str()));
196  contains_vars.push_back(vector<int>());
197  negated_vars.push_back(vector<int>());
198  for(size_t v_cnt = 0; v_cnt < nr_of_vars; ++v_cnt)
199  {
200  ostringstream var_idx;
201  var_idx << "[" << c_cnt << "][" << v_cnt << "]";
202  contains_vars[c_cnt].push_back(VM.createFreshTemplParam("cont" + var_idx.str()));
203  negated_vars[c_cnt].push_back(VM.createFreshTemplParam("cont" + var_idx.str()));
204  }
205  }
206 
207  // Step 1b: constructing the constraints:
208  // We do this for next_win_eq_w2 and win_eq_w1 in parallel
209  vector<int> all_clause_lits1;
210  vector<int> all_clause_lits2;
211  all_clause_lits1.reserve(nr_of_clauses + 1);
212  all_clause_lits2.reserve(nr_of_clauses + 1);
213  // We add one fixed clause, namely that we are inside the safe states:
214  all_clause_lits1.push_back(-VM.getPresErrorStateVar());
215  all_clause_lits2.push_back(-VM.getNextErrorStateVar());
216  for(size_t c = 0; c < nr_of_clauses; ++c)
217  {
218  int clause_lit1 = VM.createFreshTmpVar(); // TRUE iff clause c is TRUE
219  int clause_lit2 = VM.createFreshTmpVar(); // TRUE iff clause c is TRUE
220  all_clause_lits1.push_back(clause_lit1);
221  all_clause_lits2.push_back(clause_lit2);
222  vector<int> literals_in_clause1;
223  vector<int> literals_in_clause2;
224  literals_in_clause1.reserve(nr_of_vars + 2);
225  literals_in_clause2.reserve(nr_of_vars + 2);
226  // if clause_active_var[c]=FALSE, then the clause should be TRUE
227  // so we add -clause_active_vars[c] as if it was a literal of this clause
228  literals_in_clause1.push_back(-clause_active_vars[c]);
229  literals_in_clause2.push_back(-clause_active_vars[c]);
230  for(size_t v = 0; v < nr_of_vars; ++v)
231  {
232  int activated_var1 = VM.createFreshTmpVar();
233  int activated_var2 = VM.createFreshTmpVar();
234  literals_in_clause1.push_back(activated_var1);
235  literals_in_clause2.push_back(activated_var2);
236  // Part A: activated_var should be equal to
237  // - FALSE if (contains_var[c][v]==FALSE)
238  // - not(var[v]) if (contains_var[c][v]==TRUE and negated_var[c][v]==TRUE)
239  // - var[v] if (contains_var[c][v]==TRUE and negated_var[c][v]==FALSE)
240  // Part A1: not(contains_vars[c][v]) implies not(activated_var)
241  win_constr.add2LitClause(contains_vars[c][v], -activated_var1);
242  win_constr.add2LitClause(contains_vars[c][v], -activated_var2);
243  // Part A2: (contains_var[c][v] and negated_var[c][v]) implies (activated_var <=> not(var[v]))
244  win_constr.add4LitClause(-contains_vars[c][v], -negated_vars[c][v], ps_vars[v], activated_var1);
245  win_constr.add4LitClause(-contains_vars[c][v], -negated_vars[c][v], -ps_vars[v], -activated_var1);
246  win_constr.add4LitClause(-contains_vars[c][v], -negated_vars[c][v], ns_vars[v], activated_var2);
247  win_constr.add4LitClause(-contains_vars[c][v], -negated_vars[c][v], -ns_vars[v], -activated_var2);
248  // Part A3: (contains_var[c][v] and not(negated_var[c][v])) implies (activated_var <=> var[v])
249  win_constr.add4LitClause(-contains_vars[c][v], negated_vars[c][v], -ps_vars[v], activated_var1);
250  win_constr.add4LitClause(-contains_vars[c][v], negated_vars[c][v], ps_vars[v], -activated_var1);
251  win_constr.add4LitClause(-contains_vars[c][v], negated_vars[c][v], -ns_vars[v], activated_var2);
252  win_constr.add4LitClause(-contains_vars[c][v], negated_vars[c][v], ns_vars[v], -activated_var2);
253  }
254  // Part B: clause_lit should be equal to OR(literals_in_clause):
255  // Part B1: if one literal in literals_in_clause is TRUE then clause_lit should be TRUE
256  for(size_t cnt = 0; cnt < literals_in_clause1.size(); ++cnt)
257  {
258  win_constr.add2LitClause(-literals_in_clause1[cnt], clause_lit1);
259  win_constr.add2LitClause(-literals_in_clause2[cnt], clause_lit2);
260  }
261  // Part B2: if all literals in literals_in_clause are FALSE, then the clause_lit should be FALSE
262  // That is, we add the clause [l1, l2, l3, l4, -clause_lit]
263  literals_in_clause1.push_back(-clause_lit1);
264  win_constr.addClause(literals_in_clause1);
265  literals_in_clause2.push_back(-clause_lit2);
266  win_constr.addClause(literals_in_clause2);
267  }
268  // Part C: w should be equal to AND(all_clause_lits)
269  // Part C1: if one literal in all_clause_lits is FALSE then w should be FALSE:
270  for(size_t cnt = 0; cnt < all_clause_lits1.size(); ++cnt)
271  {
272  win_constr.add2LitClause(all_clause_lits1[cnt], -w1);
273  win_constr.add2LitClause(all_clause_lits2[cnt], -w2);
274  }
275  // Part C2: if all literals in all_clause_lits are TRUE, the w should be TRUE:
276  // That is, we create a clause [-l1, -l2, -l3, -l4, w]
277  for(size_t cnt = 0; cnt < all_clause_lits1.size(); ++cnt)
278  {
279  all_clause_lits1[cnt] = -all_clause_lits1[cnt];
280  all_clause_lits2[cnt] = -all_clause_lits2[cnt];
281  }
282  all_clause_lits1.push_back(w1);
283  all_clause_lits2.push_back(w2);
284  win_constr.addClause(all_clause_lits1);
285  win_constr.addClause(all_clause_lits2);
286 
287 
288  // Step 2: solve
289  vector<int> model;
290  bool sat = false;
291  if(Options::instance().getBackEndMode() == 0)
292  sat = syntQBF(win_constr, w1, w2, model);
293  else
294  sat = syntSAT(win_constr, w1, w2, model);
295  if(!sat)
296  return false;
297 
298 
299  // Step 3: Build a CNF for the winning region:
300  winning_region_.add1LitClause(-VM.getPresErrorStateVar());
301  vector<bool> can_be_0(VM.getMaxCNFVar() + 1, true);
302  vector<bool> can_be_1(VM.getMaxCNFVar() + 1, true);
303  VM.resetToLastPush();
304  for(size_t v_cnt = 0; v_cnt < model.size(); ++v_cnt)
305  {
306  if(model[v_cnt] < 0)
307  can_be_1[-model[v_cnt]] = false;
308  else
309  can_be_0[model[v_cnt]] = false;
310  }
311  for(size_t c_cnt = 0; c_cnt < nr_of_clauses; ++c_cnt)
312  {
313  if(can_be_0[clause_active_vars[c_cnt]])
314  continue;
315  vector<int> clause;
316  clause.reserve(nr_of_vars);
317  for(size_t v_cnt = 0; v_cnt < nr_of_vars; ++v_cnt)
318  {
319  if(can_be_0[contains_vars[c_cnt][v_cnt]])
320  continue;
321  if(can_be_0[negated_vars[c_cnt][v_cnt]])
322  clause.push_back(ps_vars[v_cnt]);
323  else
324  clause.push_back(-ps_vars[v_cnt]);
325  }
326  winning_region_.addClause(clause);
327  }
328  neg_winning_region_ = winning_region_;
329  neg_winning_region_.negate();
330  return true;
331 }

334 // -------------------------------------------------------------------------------------------
335 bool TemplateSynth::findWinRegANDNetwork(size_t nr_of_gates)
336 {
337  // Note: a lot of loops and code could be merged, but readability
338  // beats performance in this case.
339 
340  VarManager &VM = VarManager::instance();
341  vector<int> ps_vars = VM.getVarsOfType(VarInfo::PRES_STATE);
342  vector<int> ns_vars = VM.getVarsOfType(VarInfo::NEXT_STATE);
343 
344  // ps_vars[0] and ns_vars[0] are the error bits, which must be FALSE always.
345  // Hence, we do not include these signals in our template
346  ps_vars[0] = ps_vars[ps_vars.size() - 1];
347  ps_vars.pop_back();
348  ns_vars[0] = ns_vars[ns_vars.size() - 1];
349  ns_vars.pop_back();
350  size_t nr_of_vars = ps_vars.size();
351 
352  // Step 1: build the parameterized CNF:
353  int w1 = VM.createFreshTmpVar("w1");
354  CNF win_constr;
355  int w2 = VM.createFreshTmpVar("w2");
356 
357  // Step 1a: create template parameters:
358  vector<vector<int> > used;
359  vector<vector<int> > neg;
360  used.reserve(nr_of_gates);
361  neg.reserve(nr_of_gates);
362  for(size_t g_cnt = 0; g_cnt < nr_of_gates; ++g_cnt)
363  {
364  used.push_back(vector<int>());
365  neg.push_back(vector<int>());
366  for(size_t v_cnt = 0; v_cnt < nr_of_vars + g_cnt; ++v_cnt)
367  {
368  ostringstream var_idx;
369  var_idx << "[" << g_cnt << "][" << v_cnt << "]";
370  used[g_cnt].push_back(VM.createFreshTemplParam("cont" + var_idx.str()));
371  neg[g_cnt].push_back(VM.createFreshTemplParam("neg" + var_idx.str()));
372  }
373  }
374 
375  // Step 1b: construct gate outputs and list of gate inputs:
376  vector<int> gop; // gop[i] = output of gate i (present state)
377  gop.reserve(nr_of_gates);
378  vector<int> gon; // gon[i] = output of gate i (next state)
379  gon.reserve(nr_of_gates);
380  vector<vector<int> > gip; // gip[i] = inputs of gate i (present state)
381  gip.reserve(nr_of_gates);
382  vector<vector<int> > gin; // gin[i] = inputs of gate i (next state)
383  gin.reserve(nr_of_gates);
384  for(size_t g_cnt = 0; g_cnt < nr_of_gates; ++g_cnt)
385  {
386  gop.push_back(VM.createFreshTmpVar());
387  gon.push_back(VM.createFreshTmpVar());
388  gip.push_back(ps_vars);
389  gin.push_back(ns_vars);
390  // gate i can also have the output of gate j as input if j < i:
391  for(size_t g_cnt2 = 0; g_cnt2 < g_cnt; ++g_cnt2)
392  {
393  gip[g_cnt].push_back(gop[g_cnt2]);
394  gin[g_cnt].push_back(gon[g_cnt2]);
395  }
396  }
397 
398  // Step 1c: constructing the constraints:
399  // We do this for w1 and w2 in parallel
400  for(size_t g = 0; g < nr_of_gates; ++g)
401  {
402  vector<int> act_in_varsP;
403  vector<int> act_in_varsN;
404  act_in_varsP.reserve(gip[g].size() + 1);
405  act_in_varsN.reserve(gip[g].size() + 1);
406 
407  // construct the constraints defining the activated inputs of the gate:
408  for(size_t v = 0; v < gip[g].size(); ++v)
409  {
410  int activated_varP = VM.createFreshTmpVar();
411  act_in_varsP.push_back(activated_varP);
412  int activated_varN = VM.createFreshTmpVar();
413  act_in_varsN.push_back(activated_varN);
414  // Part A: activated_var should be equal to
415  // - TRUE if (used[g][v]==FALSE)
416  // - not(var[v]) if (used[g][v]==TRUE and neg[g][v]==TRUE)
417  // - var[v] if (used[g][v]==TRUE and neg[g][v]==FALSE)
418  // Part A1: not(used[g][v]) implies activated_var
419  win_constr.add2LitClause(used[g][v], activated_varP);
420  win_constr.add2LitClause(used[g][v], activated_varN);
421  // Part A2: (used[g][v] and neg[g][v]) implies (activated_var <=> not(var[v]))
422  win_constr.add4LitClause(-used[g][v], -neg[g][v], gip[g][v], activated_varP);
423  win_constr.add4LitClause(-used[g][v], -neg[g][v], -gip[g][v], -activated_varP);
424  win_constr.add4LitClause(-used[g][v], -neg[g][v], gin[g][v], activated_varN);
425  win_constr.add4LitClause(-used[g][v], -neg[g][v], -gin[g][v], -activated_varN);
426  // Part A3: (used[g][v] and not(neg[g][v])) implies (activated_var <=> var[v])
427  win_constr.add4LitClause(-used[g][v], neg[g][v], -gip[g][v], activated_varP);
428  win_constr.add4LitClause(-used[g][v], neg[g][v], gip[g][v], -activated_varP);
429  win_constr.add4LitClause(-used[g][v], neg[g][v], -gin[g][v], activated_varN);
430  win_constr.add4LitClause(-used[g][v], neg[g][v], gin[g][v], -activated_varN);
431  }
432 
433  // Part B: gop[g] should be equal to AND(act_in_vars):
434  // Part B1: if one literal in act_in_vars is FALSE then clause_lit should be FALSE
435  for(size_t cnt = 0; cnt < act_in_varsP.size(); ++cnt)
436  {
437  win_constr.add2LitClause(act_in_varsP[cnt], -gop[g]);
438  win_constr.add2LitClause(act_in_varsN[cnt], -gon[g]);
439  }
440  // Part B2: if all literals in act_in_vars are TRUE, then gop[g] should be TRUE
441  // That is, we add the clause [-l1, -l2, -l3, -l4, ..., gop[g]]
442  Utils::negateLiterals(act_in_varsP);
443  act_in_varsP.push_back(gop[g]);
444  win_constr.addClause(act_in_varsP);
445  Utils::negateLiterals(act_in_varsN);
446  act_in_varsN.push_back(gon[g]);
447  win_constr.addClause(act_in_varsN);
448  }
449 
450  // Part C: the final output is the output of the last AND gate, possibly negated, and then
451  // conjuncted with the safe states:
452  int out_neg = VM.createFreshTemplParam();
453  int ps_safe = -VM.getPresErrorStateVar();
454  int ns_safe = -VM.getNextErrorStateVar();
455  win_constr.add2LitClause(ps_safe, -w1);
456  win_constr.add4LitClause(-ps_safe, out_neg, gop[nr_of_gates-1], -w1);
457  win_constr.add4LitClause(-ps_safe, out_neg, -gop[nr_of_gates-1], w1);
458  win_constr.add4LitClause(-ps_safe, -out_neg, gop[nr_of_gates-1], w1);
459  win_constr.add4LitClause(-ps_safe, -out_neg, -gop[nr_of_gates-1], -w1);
460  win_constr.add2LitClause(ns_safe, -w2);
461  win_constr.add4LitClause(-ns_safe, out_neg, gon[nr_of_gates-1], -w2);
462  win_constr.add4LitClause(-ns_safe, out_neg, -gon[nr_of_gates-1], w2);
463  win_constr.add4LitClause(-ns_safe, -out_neg, gon[nr_of_gates-1], w2);
464  win_constr.add4LitClause(-ns_safe, -out_neg, -gon[nr_of_gates-1], -w2);
465 
466  // Step 2: solve
467  vector<int> model;
468 
469  bool sat = false;
470  if(Options::instance().getBackEndMode() == 2)
471  sat = syntQBF(win_constr, w1, w2, model);
472  else
473  sat = syntSAT(win_constr, w1, w2, model);
474  if(!sat)
475  return false;
476 
477  // Step 3: parse the solution:
478  vector<int> params = VM.getVarsOfType(VarInfo::TEMPL_PARAMS);
479  int max_param_idx = 0;
480  for(size_t cnt = 0; cnt < params.size(); ++cnt)
481  if(params[cnt] > max_param_idx)
482  max_param_idx = params[cnt];
483 
484  // 3a: reset the manager so that we get rid of all parameters and auxiliary vars that are
485  // not needed any more.
486  // We re-do the aux-vars that are the output of the and gates in a persistent way:
487  VM.resetToLastPush();
488  gop.clear(); // gop[i] = output of gate i (present state)
489  gop.reserve(nr_of_gates);
490  gip.clear(); // gip[i] = inputs of gate i (present state)
491  gip.reserve(nr_of_gates);
492  vector<int> gop2; // copy for the negation of the winning region
493  gop2.reserve(nr_of_gates);
494  vector<vector<int> > gip2; // copy for the negation of the winning region
495  gip2.reserve(nr_of_gates);
496  for(size_t g_cnt = 0; g_cnt < nr_of_gates; ++g_cnt)
497  {
498  gop.push_back(VM.createFreshTmpVar());
499  gop2.push_back(VM.createFreshTmpVar());
500  gip.push_back(ps_vars);
501  gip2.push_back(ps_vars);
502  // gate i can also have the output of gate j as input if j < i:
503  for(size_t g_cnt2 = 0; g_cnt2 < g_cnt; ++g_cnt2)
504  {
505  gip[g_cnt].push_back(gop[g_cnt2]);
506  gip2[g_cnt].push_back(gop2[g_cnt2]);
507  }
508  }
509 
510 
511  // 3b: parse the model:
512  vector<bool> can_be_0(max_param_idx + 1, true);
513  vector<bool> can_be_1(max_param_idx + 1, true);
514  for(size_t v_cnt = 0; v_cnt < model.size(); ++v_cnt)
515  {
516  if(model[v_cnt] < 0)
517  can_be_1[-model[v_cnt]] = false;
518  else
519  can_be_0[model[v_cnt]] = false;
520  }
521 
522  // 3c: construct the winning region formula:
523  for(size_t g = 0; g < nr_of_gates; ++g)
524  {
525  vector<int> and_over; // for winning_region_
526  vector<int> and_over2; // for neg_winning_region_
527  and_over.reserve(gip[g].size() + 1);
528  and_over2.reserve(gip[g].size() + 1);
529  for(size_t v = 0; v < gip[g].size(); ++v)
530  {
531  if(can_be_0[used[g][v]])
532  continue;
533 
534  if(can_be_0[neg[g][v]])
535  {
536  // var is unnegated:
537  winning_region_.add2LitClause(gip[g][v], -gop[g]);
538  neg_winning_region_.add2LitClause(gip2[g][v], -gop2[g]);
539  and_over.push_back(gip[g][v]);
540  and_over2.push_back(gip2[g][v]);
541  }
542  else
543  {
544  // var is negated:
545  winning_region_.add2LitClause(-gip[g][v], -gop[g]);
546  neg_winning_region_.add2LitClause(-gip2[g][v], -gop2[g]);
547  and_over.push_back(-gip[g][v]);
548  and_over2.push_back(gip2[g][v]);
549  }
550  }
551  // if all literals in and_over are TRUE, then gop[g] should be TRUE
552  // That is, we add the clause [-l1, -l2, -l3, -l4, ..., gop[g]]
553  Utils::negateLiterals(and_over);
554  and_over.push_back(gop[g]);
555  winning_region_.addClause(and_over);
556  Utils::negateLiterals(and_over2);
557  and_over2.push_back(gop2[g]);
558  neg_winning_region_.addClause(and_over2);
559  }
560  winning_region_.add1LitClause(-VM.getPresErrorStateVar());
561  if(can_be_0[out_neg])
562  {
563  winning_region_.add1LitClause(gop[nr_of_gates - 1]);
564  neg_winning_region_.add2LitClause(VM.getPresErrorStateVar(), -gop2[nr_of_gates - 1]);
565  }
566  else
567  {
568  winning_region_.add1LitClause(-gop[nr_of_gates - 1]);
569  neg_winning_region_.add2LitClause(VM.getPresErrorStateVar(), gop2[nr_of_gates - 1]);
570  }
571  return true;
572 }
