114 void LearnStatisticsQBF::logStatistics() const
115 {
116  L_LOG("Nr. of bad state computations for winning region: " << nr_of_cube_computations_);
117  L_LOG(" Bad state computations took:");
118  L_LOG(" " << sum_compute_cpu_time_ << " sec CPU time in total.");
119  L_LOG(" " << sum_compute_real_time_ << " sec real time in total.");
120  double avg_compute_ctime = 0.0;
121  double avg_compute_rtime = 0.0;
122  if(nr_of_cube_computations_ != 0)
123  {
124  avg_compute_ctime = sum_compute_cpu_time_ / nr_of_cube_computations_;
125  avg_compute_rtime = static_cast<double>(sum_compute_real_time_)/nr_of_cube_computations_;
126  }
127  L_LOG(" " << avg_compute_ctime << " sec CPU time on average.");
128  L_LOG(" " << avg_compute_rtime << " sec real time on average.");
129 
130  L_LOG("Nr. of cube minimizations for winning region: " << nr_of_cube_min_);
131  L_LOG(" Cube minimizations took:");
132  L_LOG(" " << sum_min_cpu_time_ << " sec CPU time in total.");
133  L_LOG(" " << sum_min_real_time_ << " sec real time in total.");
134  double avg_min_ctime = 0.0;
135  double avg_min_rtime = 0.0;
136  if(nr_of_cube_min_ != 0)
137  {
138  avg_min_ctime = sum_min_cpu_time_ / nr_of_cube_min_;
139  avg_min_rtime = static_cast<double>(sum_min_real_time_) / nr_of_cube_min_;
140  }
141  L_LOG(" " << avg_min_ctime << " sec CPU time on average.");
142  L_LOG(" " << avg_min_rtime << " sec real time on average.");
143 
144  L_LOG("Total cube size reduction: " << sum_cube_size_from_ << " --> " << sum_cube_size_to_);
145  double avg_size_from = 0.0;
146  double avg_size_to = 0.0;
147  if(nr_of_cube_min_ != 0)
148  {
149  avg_size_from = static_cast<double>(sum_cube_size_from_) / nr_of_cube_min_;
150  avg_size_to = static_cast<double>(sum_cube_size_to_) / nr_of_cube_min_;
151  }
152  L_LOG("Average cube size reduction: " << avg_size_from << " --> " << avg_size_to);
153  L_LOG("Winning region computation time: " << win_reg_cpu_time_ << " sec CPU time.");
154  L_LOG("Winning region computation time: " << win_reg_real_time_ << " sec real time.");
155 }
