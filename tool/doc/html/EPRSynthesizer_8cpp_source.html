60 bool EPRSynthesizer::run()
61 {
62  tptp_query_.clear();
63  const char *error = NULL;
64  aiger *aig = aiger_init();
65  const string &file = Options::instance().getAigInFileName();
66  error = aiger_open_and_read_from_file (aig, file.c_str());
67  MASSERT(error == NULL, "Could not open AIGER file " << file << " (" << error << ").");
68  MASSERT(aig->num_outputs == 1, "Strange number of outputs in AIGER file.");
69 
70  // 'names' maps AIGER literals to corresponding EPR formulas.
71  vector<string> names(2*(aig->maxvar+1), "");
72  names[0] = "p(0)";
73  names[1] = "p(1)";
74  // For every AIGER literal, deps stores the inputs and state bits on which it depends.
75  map<unsigned, set<string> > deps;
76  set<string> emptyset;
77  deps[0] = emptyset;
78  deps[1] = emptyset;
79  // The prefix 'X_' means current state, 'Y_' means next state, 'I_' means
80  // uncontrollable input:
81  string all_inputs_and_states = "X_err";
82  set<string> all_inputs_and_states_set;
83  all_inputs_and_states_set.insert("X_err");
84  string all_states = "X_err";
85  string all_next_states = "Y_err";
86 
87  // Analyze the uncontrollable inputs and fill the data structures (names, deps, ...)
88  for(unsigned cnt = 0; cnt < aig->num_inputs; ++cnt)
89  {
90  string aig_name;
91  if(aig->inputs[cnt].name != NULL)
92  aig_name = aig->inputs[cnt].name;
93  string aig_name_lower = StringUtils::toLowerCase(aig_name);
94  if(aig_name_lower.find("controllable_") != 0)
95  {
96  unsigned lit = aig->inputs[cnt].lit;
97  ostringstream oss;
98  oss << lit;
99  names[lit] = "p(I_" + oss.str() + ")";
100  names[lit + 1] = "~p(I_" + oss.str() + ")";
101  all_inputs_and_states += ",I_" + oss.str();
102  all_inputs_and_states_set.insert("I_" + oss.str());
103  set<string> dep_set;
104  dep_set.insert("I_" + oss.str());
105  deps[lit] = dep_set;
106  deps[lit + 1] = dep_set;
107  }
108  }
109  // Analyze the state variables and fill the data structures (names, deps, ...)
110  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
111  {
112  // present state:
113  unsigned lit = aig->latches[cnt].lit;
114  ostringstream oss;
115  oss << lit;
116  names[lit] = "p(X_" + oss.str() + ")";
117  names[lit + 1] = "~p(X_" + oss.str() + ")";
118  all_inputs_and_states += ",X_" + oss.str();
119  all_inputs_and_states_set.insert("X_" + oss.str());
120  all_states += ",X_" + oss.str();
121  all_next_states += ",Y_" + oss.str();
122  set<string> dep_set;
123  dep_set.insert("X_" + oss.str());
124  deps[lit] = dep_set;
125  deps[lit + 1] = dep_set;
126  }
127 
128  // Analyze the uncontrollable inputs and fill the data structures (names, deps, ...)
129  for(unsigned cnt = 0; cnt < aig->num_inputs; ++cnt)
130  {
131  string aig_name;
132  if(aig->inputs[cnt].name != NULL)
133  aig_name = aig->inputs[cnt].name;
134  string aig_name_lower = StringUtils::toLowerCase(aig_name);
135  if(aig_name_lower.find("controllable_") == 0)
136  {
137  unsigned lit = aig->inputs[cnt].lit;
138  ostringstream oss;
139  oss << "csk_" << lit << "(" << all_inputs_and_states << ")";
140  names[lit] = oss.str();
141  names[lit + 1] = "~" + oss.str();
142  deps[lit] = all_inputs_and_states_set;
143  deps[lit + 1] = all_inputs_and_states_set;
144  }
145  }
146 
147  // let's find out on which inputs and state bits the outputs of the AND gates
148  // depend:
149  bool something_found = true;
150  while(something_found)
151  {
152  something_found = false;
153  for (unsigned cnt = 0; cnt < aig->num_ands; cnt++)
154  {
155  unsigned lit = aig->ands[cnt].lhs;
156  unsigned rhs0 = aig->ands[cnt].rhs0;
157  unsigned rhs1 = aig->ands[cnt].rhs1;
158  if(deps.count(lit) == 0 && deps.count(rhs0) != 0 && deps.count(rhs1) != 0)
159  {
160  something_found = true;
161  set<string> lit_deps(deps[rhs0]);
162  lit_deps.insert(deps[rhs1].begin(), deps[rhs1].end());
163  deps[lit] = lit_deps;
164  deps[lit+1] = lit_deps;
165  ostringstream oss;
166  oss << "tmpsk_" << lit;
167  if(!lit_deps.empty())
168  oss << "(";
169  for(set<string>::const_iterator it = lit_deps.begin(); it != lit_deps.end(); ++it)
170  {
171  if(it == lit_deps.begin())
172  oss << *it;
173  else
174  oss << "," << *it;
175  }
176  if(!lit_deps.empty())
177  oss << ")";
178  names[lit] = oss.str();
179  names[lit+1] = "~" + oss.str();
180  }
181  }
182  }
183 
184 #ifndef NDEBUG
185  for (unsigned cnt = 0; cnt < aig->num_ands; cnt++)
186  {
187  if(names[aig->ands[cnt].lhs] == "")
188  {
189  L_ERR("and gate " << aig->ands[cnt].lhs << " has no name.");
190  MASSERT(false, "Some AND gate has no name.");
191  }
192  else if(names[aig->ands[cnt].lhs + 1] == "")
193  {
194  L_ERR("neg and gate " << aig->ands[cnt].lhs << " has no name.");
195  MASSERT(false, "Some AND gate has no name.");
196  }
197  }
198 #endif
199 
200  string pres_win = "win(" + all_states + ")";
201  if(all_states.size() == 0)
202  pres_win = "win";
203  string next_win = "win(" + all_next_states + ")";
204  if(all_next_states.size() == 0)
205  next_win = "win";
206 
207  // 1. initial implies winning I(x) -> P(x):
208  string initial_implies_winning = "p(X_err)";
209  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
210  initial_implies_winning += " | " + names[aig->latches[cnt].lit];
211  initial_implies_winning += " | " + pres_win;
212  addClause(initial_implies_winning, "The initial state is winning: I(X) -> W(X)");
213 
214  // 2. winning implies safe W(x) -> P(x):
215  string c = "All winning states are safe: W(X) -> P(X)";
216  addClause("~" + pres_win + " | ~p(X_err)", c);
217 
218  // 3. encode the transition relation t <-> T(x,i,c,x')
219  // 3.1 AND gates:
220  for (unsigned cnt = 0; cnt < aig->num_ands; cnt++)
221  {
222  unsigned lhs = aig->ands[cnt].lhs;
223  unsigned rhs0 = aig->ands[cnt].rhs0;
224  unsigned rhs1 = aig->ands[cnt].rhs1;
225  // (lhs --> rhs1) AND (lhs --> rhs0)
226  ostringstream gate_name;
227  gate_name << "AND Gate (AIG" << lhs << " = AIG" << rhs0 << " & AIG" << rhs1 << "), ";
228  c = gate_name.str() + "rule 1: l->r0";
229  addClause(names[neg(lhs)] + " | " + names[rhs0], c);
230  c = gate_name.str() + "rule 2: l->r1";
231  addClause(names[neg(lhs)] + " | " + names[rhs1], c);
232  // (!lhs --> (!rhs1 OR !rhs0)
233  c = gate_name.str() + "rule 3: !l -> (!r0 | !r1)";
234  addClause(names[lhs] + " | " + names[neg(rhs0)] + " | " + names[neg(rhs1)], c);
235  }
236 
237  // 3.2 next state variables are set correctly:
238  string t = "tsk(" + all_inputs_and_states + "," + all_next_states +")";
239  // all_equal_clause contains: err_eq_name AND eq_name1 AND eq_name2 AND ... --> t
240  string all_equal_clause = t;
241 
242  // 3.2.1 fake error latch:
243  unsigned err_aig = aig->outputs[0].lit;
244  string err_eq_name = "teqsk_err(Y_err";
245  for(set<string>::const_iterator it = deps[err_aig].begin(); it != deps[err_aig].end(); ++it)
246  err_eq_name += "," + *it;
247  err_eq_name += ")";
248  c = "teqsk_err <-> (Y_err <-> err_out), case 1: (!Y_err & !error) -> teqsk_err";
249  addClause("p(Y_err) | " + names[err_aig] + " | " + err_eq_name, c);
250  c = "teqsk_err <-> (Y_err <-> err_out), case 2: (!Y_err & error) -> !teqsk_err";
251  addClause("p(Y_err) | " + names[neg(err_aig)] + " | ~" + err_eq_name, c);
252  c = "teqsk_err <-> (Y_err <-> err_out), case 3: (Y_err & !error) -> !teqsk_err";
253  addClause("~p(Y_err) | " + names[err_aig] + " | ~" + err_eq_name, c);
254  c = "teqsk_err <-> (Y_err <-> err_out), case 4: (Y_err & error) -> teqsk_err";
255  addClause("~p(Y_err) | " + names[neg(err_aig)] + " | " + err_eq_name, c);
256  // err_eq_name is false --> t is false
257  c = "!teqsk_err -> !t";
258  addClause(err_eq_name + " | ~" + t, c);
259  all_equal_clause += " | ~" + err_eq_name;
260 
261  // 3.2.2 real latches:
262  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
263  {
264  unsigned n_aig = aig->latches[cnt].next;
265  unsigned p_aig = aig->latches[cnt].lit;
266  ostringstream ps;
267  ps << p_aig;
268  string eq_name = "teqsk_" + ps.str() + "(Y_" + ps.str();
269  for(set<string>::const_iterator it = deps[n_aig].begin(); it != deps[n_aig].end(); ++it)
270  eq_name += "," + *it;
271  eq_name += ")";
272 
273  // eq_name is true iff n_aig is equal to p_aig
274  string c0 = "teqsk_" + ps.str() + " <-> (Y_" + ps.str() + " <-> " + names[n_aig] + ")";
275  c = c0 + ", case 1: (!Y_" + ps.str() + " & !" + names[n_aig] + ") -> teqsk_" + ps.str();
276  addClause("p(Y_" + ps.str() + ") | " + names[n_aig] + " | " + eq_name, c);
277  c = c0 + ", case 2: (!Y_" + ps.str() + " & " + names[n_aig] + ") -> !teqsk_" + ps.str();
278  addClause("p(Y_" + ps.str() + ") | " + names[neg(n_aig)] + " | ~" + eq_name, c);
279  c = c0 + ", case 2: (Y_" + ps.str() + " & !" + names[n_aig] + ") -> !teqsk_" + ps.str();
280  addClause("~p(Y_" + ps.str() + ") | " + names[n_aig] + " | ~" + eq_name, c);
281  c = c0 + ", case 2: (Y_" + ps.str() + " & " + names[n_aig] + ") -> teqsk_" + ps.str();
282  addClause("~p(Y_" + ps.str() + ") | " + names[neg(n_aig)] + " | " + eq_name, c);
283  // eq_name is false --> t is false
284  c = "!teqsk_" + ps.str() + " -> !t";
285  addClause(eq_name + " | ~" + t, c);
286  all_equal_clause += " | ~" + eq_name;
287  }
288  c = "(all teqsk_* are true) -> (t is true)";
289  addClause(all_equal_clause, c);
290 
291  // 4. W(x) && T(x,i,c,x') --> W(x')
292  c = "W(X) && T(X,I,c(X,I),Y) --> W(Y)";
293  addClause("~" + pres_win + " | ~" + t + " | " + next_win, c);
294 
295  tptp_query_ << "cnf(rule_true,axiom, p(1))." << endl;
296  tptp_query_ << "cnf(rule_false,axiom, ~p(0))." << endl;
297 
298  // Now we are done creatint the EPR formula. Let's start the solver:
299 
300  FileUtils::writeFile(in_file_name_, tptp_query_.str());
301  string solver_command = path_to_iprover_;
302  solver_command += " --out_options none";
303  solver_command += " --fof false"; // assumes cnf
304  // a schedule which works well for verification problems encoded in the EPR fragment:
305  solver_command += " --schedule verification_epr";
306  solver_command += " --sat_epr_types true"; // using EPR types in sat mode
307  // solver_command += " --sat_mode true";
308  // solver_command += " --sat_finite_models true";
309  // solver_command += " --sat_out_model small"; // <small|pos|neg|implied|debug|intel|none>
310  solver_command += " " + in_file_name_ + " > " + out_file_name_;
311  int ret = system(solver_command.c_str());
312  ret = WEXITSTATUS(ret);
313  MASSERT(ret == 0, "Strange exit code from iprover.");
314 
315  string result_file_content;
316  FileUtils::readFile(out_file_name_, result_file_content);
317  remove(in_file_name_.c_str());
318  remove(out_file_name_.c_str());
319  if(result_file_content.find("SZS status Unsatisfiable") != string::npos)
320  {
321  L_RES("The specification is unrealizable.");
322  return false;
323  }
324  if(result_file_content.find("SZS status Satisfiable") != string::npos)
325  {
326  L_RES("The specification is realizable.");
327  if(!Options::instance().doRealizabilityOnly())
328  {
329  MASSERT(false, "TODO: circuit extraction not yet implemented.");
330  }
331  return true;
332  }
333  L_ERR("Strange response from iprover.");
334  return false;
335 }

337 void EPRSynthesizer::addClause(const string &clause, string comment)
338 {
339  tptp_query_ << "% --------------------------------------------------------" << endl;
340  if(comment != "")
341  tptp_query_ << "% " << comment << endl;
342  tptp_query_ << "cnf(cl_" << next_clause_nr_ << ", plain, (";
343  tptp_query_ << clause;
344  tptp_query_ << "))." << endl << endl;
345  next_clause_nr_++;
346 }
