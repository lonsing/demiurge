40 LearnSynthQBF::LearnSynthQBF() :
41  BackEnd(),
42  qbf_solver_(Options::instance().getQBFSolver()),
43  solver_i_(Options::instance().getSATSolver()),
44  solver_ctrl_(Options::instance().getSATSolver()),
45  incremental_vars_to_keep_(VarManager::instance().getAllNonTempVars()),
46  solver_i_precise_(true)
47 {
48  // build the quantifier prefix for checking for counterexamples:
49  // check_quant_ = exists x,i: forall c: exists x',tmp:
50  check_quant_.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
51  check_quant_.push_back(make_pair(VarInfo::INPUT, QBFSolver::E));
52  check_quant_.push_back(make_pair(VarInfo::CTRL, QBFSolver::A));
53  check_quant_.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
54  check_quant_.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
55 
56  // build the quantifier prefix for generalizing cubes:
57  // gen_quant_ = exists x: forall i: exists c,x',tmp:
58  gen_quant_.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
59  gen_quant_.push_back(make_pair(VarInfo::INPUT, QBFSolver::A));
60  gen_quant_.push_back(make_pair(VarInfo::CTRL, QBFSolver::E));
61  gen_quant_.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
62  gen_quant_.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
63 
64  // if we use computeWinningRegionOneSAT() then we need two SAT-solvers for computing
65  // counterexamples:
66  solver_i_->startIncrementalSession(incremental_vars_to_keep_, false);
67  solver_i_->incAddCNF(AIG2CNF::instance().getNextUnsafeStates());
68  solver_i_->incAddCNF(AIG2CNF::instance().getTrans());
69  solver_i_->incAddCNF(AIG2CNF::instance().getSafeStates());
70  solver_i_precise_ = true;
71  solver_ctrl_->startIncrementalSession(incremental_vars_to_keep_, false);
72  solver_ctrl_->incAddCNF(AIG2CNF::instance().getNextSafeStates());
73  solver_ctrl_->incAddCNF(AIG2CNF::instance().getTrans());
74  solver_ctrl_->incAddCNF(AIG2CNF::instance().getSafeStates());
75 }

117  statistics_.notifyRelDetEnd();

344 bool LearnSynthQBF::computeCounterexampleSAT(vector<int> &ce)
345 {
346  statistics_.notifyBeforeComputeCube();
347  ce.clear();
348  vector<int> none;
349  const vector<int> &s = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE);
350  const vector<int> &c = VarManager::instance().getVarsOfType(VarInfo::CTRL);
351  const vector<int> &i = VarManager::instance().getVarsOfType(VarInfo::INPUT);
352  vector<int> si;
353  si.reserve(s.size() + i.size());
354  si.insert(si.end(), s.begin(), s.end());
355  si.insert(si.end(), i.begin(), i.end());
356  size_t it_cnt = 0;
357  vector<int> model_or_core;
358  while(true)
359  {
360  ++it_cnt;
361  // Here, we try to find some transition that leaves the winning region.
362  bool sat = solver_i_->incIsSatModelOrCore(vector<int>(), si, model_or_core);
363  if(!sat)
364  {
365  if(solver_i_precise_)
366  {
367  // None found and solver_i_ is precise: this means that we are done with the
368  // computation of the winning region (no more counterexamples exist):
369  statistics_.notifyAfterComputeCube();
370  return false;
371  }
372  // We could not find a possibility to leave the winning region, but solver_i_
373  // is not precise (it uses an out-dated next-state-copy of the winning region for
374  // performance reasons). In this case, we simply update the out-dated next-state copy
375  // of the winning region and try again:
376  L_DBG("Need to restart the solver_i_ (iteration " << it_cnt << ")");
377  Utils::compressStateCNF(winning_region_);
378  solver_i_->startIncrementalSession(incremental_vars_to_keep_, false);
379  CNF leave_win(winning_region_);
380  leave_win.swapPresentToNext();
381  leave_win.negate();
382  leave_win.addCNF(AIG2CNF::instance().getTrans());
383  leave_win.addCNF(winning_region_);
384  solver_i_->incAddCNF(leave_win);
385  solver_i_precise_ = true;
386  continue;
387  }
388 
389  vector<int> state_input = model_or_core;
390  vector<int> state = Utils::extract(state_input, VarInfo::PRES_STATE);
391  vector<int> input = Utils::extract(state_input, VarInfo::INPUT);
392 
393  // solver_i_ has found a way to leave the winning region in form of a state-input pair
394  // for which some control signal values make us leave the winning region. Now we check if
395  // there is a response (in form of control values) to this input such that we do not
396  // leave the winning region:
397  sat = solver_ctrl_->incIsSatModelOrCore(state, input, c, model_or_core);
398  if(!sat)
399  {
400  // No such response exists --> we have found a counterexample state
401  // L_DBG("Found bad state in iteration " << it_cnt);
402  // We take the unsatisfiable core because it represents a generalization of the
403  // counterexample state. We will do the generalization step with a QBF-solver later,
404  // but we get the core (almost) for free, and the more we generalize now, the less work
405  // we have to do later:
406  ce = model_or_core;
407  solver_i_precise_ = false;
408  statistics_.notifyAfterComputeCube();
409  return true;
410  }
411  else
412  {
413  // There exists a response. This means: the state input combination state_input is not
414  // useful for the antagonist in trying to break out of winning region. What we do now
415  // is: we generalize this state_input into a larger set of state-input combinations by
416  // dropping literals from the cube. This is done by computing an unsatisfiable core.
417  // Finally, we exclude the generalized state_input from solver_i_. This prevents that
418  // the same state-input pair will be tried in a future iteration.
419  const vector<int> &ctrl_cube = model_or_core;
420  sat = solver_i_->incIsSatModelOrCore(state_input, ctrl_cube, none, model_or_core);
421  MASSERT(sat == false, "Impossible.");
422  for(size_t cnt = 0; cnt < model_or_core.size(); ++cnt)
423  model_or_core[cnt] = -model_or_core[cnt];
424  solver_i_->incAddClause(model_or_core);
425  // L_DBG("U core: " << si.size() << " --> " << model_or_core.size());
426  }
427  }
428 }

449 bool LearnSynthQBF::computeAllBlockingClauses(vector<int> &ce,
450  vector<vector<int> > &blocking_clauses)
451 {
452  statistics_.notifyBeforeCubeMin();
453  list<set<int> > must_not_contain_queue;
454  list<vector<int> > gen_ces;
455  vector<int> orig_ce(ce);
456  vector<int> first_ce(orig_ce);
457  generalizeCounterexample(first_ce, false);
458  statistics_.notifyCubeMin(orig_ce.size(), first_ce.size());
459  if(Utils::containsInit(first_ce))
460  return false;
461  blocking_clauses.clear();
462  vector<int> blocking_clause(first_ce);
463  Utils::negateLiterals(blocking_clause);
464  blocking_clauses.push_back(blocking_clause);
465 
466  gen_ces.push_back(first_ce);
467  for(size_t cnt = 0; cnt < first_ce.size(); ++cnt)
468  {
469  set<int> singleton_set;
470  singleton_set.insert(first_ce[cnt]);
471  must_not_contain_queue.push_back(singleton_set);
472  }
473 
474  // we add our blocking_clause to generalize_clause_cnf_ to benefit from it immediately:
475  generalize_clause_cnf_.addClause(blocking_clause);
476  Utils::swapPresentToNext(blocking_clause);
477  generalize_clause_cnf_.addClause(blocking_clause);
478  while(!must_not_contain_queue.empty())
479  {
480  set<int> must_not_contain = must_not_contain_queue.front();
481  must_not_contain_queue.pop_front();
482 
483  // check if we already have a ce in gen_ces which satisfies our needs:
484  bool found = false;
485  for(list<vector<int> >::const_iterator it = gen_ces.begin(); it != gen_ces.end(); ++it)
486  {
487  if(Utils::intersectionEmpty(*it, must_not_contain))
488  {
489  found = true;
490  break;
491  }
492  }
493  if(found)
494  continue;
495 
496  // if not: we compute one
497  vector<int> new_ce(orig_ce);
498  set<int>::const_iterator it = must_not_contain.begin();
499  for( ; it != must_not_contain.end(); ++it)
500  Utils::remove(new_ce, *it);
501  size_t orig_size = new_ce.size();
502  bool is_ok = generalizeCounterexample(new_ce);
503  statistics_.notifyCubeMin(orig_size, new_ce.size());
504  if(is_ok)
505  {
506  if(Utils::containsInit(new_ce))
507  return false;
508  gen_ces.push_back(new_ce);
509  blocking_clause = new_ce;
510  Utils::negateLiterals(blocking_clause);
511  blocking_clauses.push_back(blocking_clause);
512  for(size_t cnt = 0; cnt < new_ce.size(); ++cnt)
513  {
514  set<int> new_must_not_contain(must_not_contain);
515  new_must_not_contain.insert(new_ce[cnt]);
516  must_not_contain_queue.push_back(new_must_not_contain);
517  }
518  // we add our blocking_clause to generalize_clause_cnf_ to benefit from it immediately:
519  generalize_clause_cnf_.addClause(blocking_clause);
520  Utils::swapPresentToNext(blocking_clause);
521  generalize_clause_cnf_.addClause(blocking_clause);
522  }
523  }
524 
525  statistics_.notifyAfterCubeMin();
526 
527  return true;
528 }

531 void LearnSynthQBF::reduceExistingClauses()
532 {
533  statistics_.notifyBeforeCubeMin();
534 
535  size_t or_cl = winning_region_.getNrOfClauses();
536  size_t or_lits = winning_region_.getNrOfLits();
537  const list<vector<int> > orig_clauses = winning_region_.getClauses();
538  for(CNF::ClauseConstIter it = orig_clauses.begin(); it != orig_clauses.end(); ++it)
539  {
540  vector<int> clause(*it);
541  for(size_t cnt = 0; cnt < it->size(); ++cnt)
542  {
543  vector<int> tmp(clause);
544  Utils::remove(tmp, (*it)[cnt]);
545  CNF tmp_cnf(generalize_clause_cnf_);
546  tmp_cnf.addNegClauseAsCube(tmp);
547  if(!qbf_solver_->isSat(gen_quant_, tmp_cnf))
548  clause = tmp;
549  }
550  if(it->size() > clause.size())
551  {

607 void LearnSynthQBF::restrictToStates(vector<int> &vec) const
608 {
609  for(size_t cnt = 0; cnt < vec.size(); ++cnt)
610  {
611  int var = (vec[cnt] < 0) ? -vec[cnt] : vec[cnt];
612  // We can always drop the error-bit of the state space (because we can be sure that all
613  // states with error=1 have already been removed from the winning region). This saves
614  // one iteration during the generalization.
615  if(var == VarManager::instance().getPresErrorStateVar() ||
616  VarManager::instance().getInfo(var).getKind() != VarInfo::PRES_STATE)
617  {
618  vec[cnt] = vec[vec.size() - 1];
619  vec.pop_back();
620  cnt--;
621  }
622  }

627 {

643  vector<int> next_gen_ce(gen_ce);
644  Utils::swapPresentToNext(next_gen_ce);
645  tmp_cnf.addNegCubeAsClause(next_gen_ce);
646  if(!qbf_solver_->isSat(gen_quant_, tmp_cnf))
647  gen_ce = tmp;
648  }
