63 aiger* DepQBFExt::qbfCert(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
64  const CNF& cnf)
65 {
66  dumpQBF(quantifier_prefix, cnf, in_file_name_);
67  string call_command = path_to_qbfcert_ + " " + in_file_name_ + " > /dev/null 2>&1";
68  int ret = system(call_command.c_str());
69  ret = WEXITSTATUS(ret);
70  MASSERT(ret == 0, "QBFCert terminated with strange exit code.");
71 
72  string qbfcert_out = in_file_name_ + ".aiger";
73  aiger *res = aiger_init();
74  const char *err = aiger_open_and_read_from_file (res, qbfcert_out.c_str());
75  MASSERT(err == NULL, "Could not open AIGER file " << qbfcert_out << " (" << err << ").");
76  cleanup();
77  remove(qbfcert_out.c_str());
78  return res;
79 }

106 bool DepQBFExt::parseModel(int ret, const vector<int> &get, vector<int> &model) const
107 {
108  if(ret != 10 && ret != 20)
109  throw DemiurgeException("Timeout or crash");
110 
111  if(use_bloqqer_)
112  {
113  if(ret == 20)
114  return false;
115  string answer;
116  bool success = FileUtils::readFile(out_file_name_, answer);
117  MASSERT(success, "Could not read result file.");
118  vector<string> lines;
119  StringUtils::splitLines(answer, lines, false);
120  for(size_t l_cnt = 0; l_cnt < lines.size(); ++l_cnt)
121  {
122  int lit = 0, val = 0;
123  istringstream iss(lines[l_cnt]);
124  iss >> lit;
125  if(!iss.good())
126  continue;
127  iss >> val;
128  if(!iss.good())
129  continue;
130  if(Utils::contains(get, lit))
131  model.push_back(val);
132  }
133  return true;
134  }
135  else
136  {
137  string answer;
138  bool success = FileUtils::readFile(out_file_name_, answer);
139  MASSERT(success, "Could not read result file.");
140  if(answer.find("s cnf 0") == 0)
141  return false;
142  MASSERT(answer.find("s cnf 1") == 0, "Strange response from Solver.");
143  size_t start = answer.find("V ", 0);
144  while(start != string::npos)
145  {
146  start = start + 2;
147  size_t end = answer.find(" 0", start);
148  MASSERT(end != string::npos, "Strange response from Solver.")
149  istringstream iss(answer.substr(start, end-start));
150  int value;
151  iss >> value;
152  model.push_back(value);
153  start = answer.find("V ", end);
154  }
155  return true;
156  }
157 }
