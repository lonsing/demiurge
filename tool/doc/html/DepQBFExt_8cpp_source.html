58 aiger* DepQBFExt::qbfCert(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
59  const CNF& cnf)
60 {
61  dumpQBF(quantifier_prefix, cnf, in_file_name_);
62  string call_command = path_to_qbfcert_ + " " + in_file_name_ + " > /dev/null 2>&1";
63  int ret = system(call_command.c_str());
64  ret = WEXITSTATUS(ret);
65  MASSERT(ret == 0, "QBFCert terminated with strange exit code.");
66 
67  string qbfcert_out = in_file_name_ + ".aiger";
68  aiger *res = aiger_init();
69  const char *err = aiger_open_and_read_from_file (res, qbfcert_out.c_str());
70  MASSERT(err == NULL, "Could not open AIGER file " << qbfcert_out << " (" << err << ").");
71  cleanup();
72  remove(qbfcert_out.c_str());
73  return res;
74 }
