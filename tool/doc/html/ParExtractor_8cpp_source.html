84 void ParExtractor::run(const CNF &win, const CNF &neg_win)
85 {
86  for(size_t cnt = 0; cnt < extractors_.size(); ++cnt)
87  delete extractors_[cnt];
88  extractors_.clear();
89  if(nr_of_threads_ >= 1)
90  extractors_.push_back(new ParExtractorWorker(*this, win, neg_win));
91  if(nr_of_threads_ >= 2)
92  extractors_.push_back(new ParExtractorQBFWorker(*this, win, neg_win));
93  if(nr_of_threads_ >= 3)
94  extractors_.push_back(new ParExtractorWorker(*this, win, neg_win));
95 
96  // Start the threads:
97  vector<thread> extractor_threads;
98  if(nr_of_threads_ >= 2)
99  {
100  pthread_t thread_handle;
101  int ret = pthread_create(&thread_handle, 0, &ParExtractorQBFWorker::startQBF, extractors_[1]);
102  MASSERT(ret == 0, "Could not start thread with pthread_create");
103  extractor_pthreads_.push_back(thread_handle);
104  }
105  //extractor_threads.push_back(thread(&ParExtractorWorker::runQBF, extractors_[1]));
106  if(nr_of_threads_ >= 3)
107  extractor_threads.push_back(thread(&ParExtractorWorker::runSAT, extractors_[2]));
108 
109  //The main thread executes the first explorer:
110  extractors_[0]->runSATDep();
111 
112  // Wait until the threads are finished:
113  for(size_t cnt = 0; cnt < extractor_threads.size(); ++cnt)
114  extractor_threads[cnt].join();
115  for(size_t cnt = 0; cnt < extractor_pthreads_.size(); ++cnt)
116  {
117  int ret = pthread_join((extractor_pthreads_[cnt]), 0);
118  MASSERT(ret == 0, "Could not join thread with pthread_join");
119  }
120 
121  // Find best circuit:
122  size_t best = extractors_.size();
123  for(size_t cnt = 0; cnt < extractors_.size(); ++cnt)
124  {
125  if(extractors_[cnt]->done_ && best == extractors_.size())
126  best = cnt;
127  else if(extractors_[cnt]->done_ && best < extractors_.size() &&
128  extractors_[cnt]->standalone_circuit_->num_ands <
129  extractors_[best]->standalone_circuit_->num_ands)
130  best = cnt;
131  }
132  MASSERT(best < extractors_.size(), "Something is wrong with the done_ flag.");
133  nr_of_new_ands_ = insertIntoSpec(extractors_[best]->standalone_circuit_);
134  extractors_[best]->statistics.notifyFinalSize(nr_of_new_ands_);
135 }

193 ParExtractorWorker::ParExtractorWorker(ParExtractor &coordinator,
194  const CNF &win_region,
195  const CNF &neg_win_region) :
196  done_(false),
197  waiting_(false),
198  did_second_run_(false),
199  standalone_circuit_(aiger_init()),
200  coordinator_(coordinator),
201  check_(Options::instance().getSATSolverExtr(false, false)),
202  gen_(Options::instance().getSATSolverExtr(false, true)),
203  win_region_(win_region),
204  neg_win_region_(neg_win_region),
205  next_free_cnf_lit_(VarManager::instance().getMaxCNFVar() + 1),
206  standalone_circuit2_(aiger_init())
207 {
208  // compute some lists that are used often:
209  const vector<int> &in = VarManager::instance().getVarsOfType(VarInfo::INPUT);
210  const vector<int> &pres = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE);
211  const vector<int> &ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL);
212  ip_.reserve(in.size() + pres.size());
213  ip_.insert(ip_.end(), in.begin(), in.end());
214  ip_.insert(ip_.end(), pres.begin(), pres.end());
215  ipc_.reserve(ip_.size() + ctrl.size());
216  ipc_.insert(ipc_.end(), ip_.begin(), ip_.end());
217  ipc_.insert(ipc_.end(), ctrl.begin(), ctrl.end());
218 
219  // next, we would like to compute a map from CNF variables to the corresponding AIGER
220  // literals in the standalone_circuit_. We want to use a vector instead of a map for
221  // performance reasons. Hence, we first of all need to find out the highest CNF var
222  // we may want to want to transform into the corresponding AIGER variable:
223  int max_cnf_var_of_interest = 0;
224  for(size_t cnt = 0; cnt < ipc_.size(); ++cnt)
225  if(ipc_[cnt] > max_cnf_var_of_interest)
226  max_cnf_var_of_interest = ipc_[cnt];
227  const map<int, set<VarInfo> > &trans_deps = AIG2CNF::instance().getTmpDepsTrans();
228  for(AIG2CNF::DepConstIter it = trans_deps.begin(); it != trans_deps.end(); ++it)
229  if(it->first > max_cnf_var_of_interest)
230  max_cnf_var_of_interest = it->first;
231 
232  // Now we fill the map from CNF variables to AIGER variables. While doing this, we also
233  // initialize the standalone_circuit_:
234  next_free_aig_lit_ = 2;
235  cnf_var_to_standalone_aig_var_.resize(max_cnf_var_of_interest + 1, 0);
236  for(size_t cnt = 0; cnt < ip_.size(); ++cnt)
237  {
238  if(ip_[cnt] != VarManager::instance().getPresErrorStateVar())
239  {
240  cnf_var_to_standalone_aig_var_[ip_[cnt]] = next_free_aig_lit_;
241  ostringstream name;
242  VarInfo info = VarManager::instance().getInfo(ip_[cnt]);
243  name << "Original AIG literal: " << info.getLitInAIG();
244  if(info.getName() != "")
245  name << " (" << info.getName() << ")";
246  aiger_add_input(standalone_circuit_, next_free_aig_lit_, name.str().c_str());
247  aiger_add_input(standalone_circuit2_, next_free_aig_lit_, name.str().c_str());
248  next_free_aig_lit_ += 2;
249  }
250  }
251 
252  for(size_t cnt = 0; cnt < ctrl.size(); ++cnt)
253  {
254  cnf_var_to_standalone_aig_var_[ctrl[cnt]] = next_free_aig_lit_;
255  ostringstream name;
256  VarInfo info = VarManager::instance().getInfo(ctrl[cnt]);
257  name << "Original AIG literal: " << info.getLitInAIG();
258  if(info.getName() != "")
259  name << " (" << info.getName() << ")";
260  aiger_add_output(standalone_circuit_, next_free_aig_lit_, name.str().c_str());
261  aiger_add_output(standalone_circuit2_, next_free_aig_lit_, name.str().c_str());
262  next_free_aig_lit_ += 2;
263  }
264 
265 
266  for(AIG2CNF::DepConstIter it = trans_deps.begin(); it != trans_deps.end(); ++it)
267  {
268  if(AIG2CNF::instance().isTrueInTrans() && it->first == 1) // CNF var 1 is constant TRUE
269  cnf_var_to_standalone_aig_var_[1] = 1;
270  else
271  {
272  cnf_var_to_standalone_aig_var_[it->first] = next_free_aig_lit_;
273  next_free_aig_lit_ += 2;
274  }
275  }
276  next_free_aig_lit2_ = next_free_aig_lit_;
277 }

294 void ParExtractorWorker::runSAT()
295 {
296  PointInTime start_time = Stopwatch::start();
297 
298  const vector<int> &ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL);
299  vector<int> none;
300 
301  vector<CNF> c_eq_impl;
302  c_eq_impl.reserve(ctrl.size());
303  vector<CNF> impl;
304  impl.reserve(ctrl.size());
305 
306  // we need to rename all variables except for the states and inputs:
307  int nr_of_vars = next_free_cnf_lit_;
308  vector<int> rename_map(nr_of_vars, 0);
309  for(int cnt = 1; cnt < nr_of_vars; ++cnt)
310  rename_map[cnt] = next_free_cnf_lit_++;
311  for(size_t cnt = 0; cnt < ip_.size(); ++cnt)
312  rename_map[ip_[cnt]] = ip_[cnt];
313 
314  CNF mustBe(neg_win_region_);
315  mustBe.swapPresentToNext();
316  mustBe.addCNF(win_region_);
317  mustBe.addCNF(AIG2CNF::instance().getTrans());
318  CNF goto_win(win_region_);
319  goto_win.swapPresentToNext();
320  goto_win.addCNF(AIG2CNF::instance().getTrans());
321  goto_win.renameVars(rename_map);
322  mustBe.addCNF(goto_win);
323 
324  // We need one activation variable per control signal. This activation variable
325  // activates all control-signal-dependent constraints:
326  vector<int> act_vars(ctrl.size(), 0);
327  // If eq_vars[cnt] is set then ctrl[i] is equal to its copy for all cnt<i<ctrl.size()
328  vector<int> eq_vars(ctrl.size(), 0);
329  for(size_t cnt = 0; cnt < ctrl.size(); ++cnt)
330  {
331  act_vars[cnt] = next_free_cnf_lit_++;
332  eq_vars[cnt] = next_free_cnf_lit_++;
333  }
334  for(size_t cnt = 0; cnt < ctrl.size() - 1; ++cnt)
335  {
336  // ctrl[cnt + 1] must be equal to its copy:
337  mustBe.add3LitClause(-eq_vars[cnt], ctrl[cnt+1], -rename_map[ctrl[cnt+1]]);
338  mustBe.add3LitClause(-eq_vars[cnt], -ctrl[cnt+1], rename_map[ctrl[cnt+1]]);
339  // eq_vars[cnt + 1] must be set:
340  mustBe.add2LitClause(-eq_vars[cnt], eq_vars[cnt+1]);
341  mustBe.add2LitClause(-act_vars[cnt], eq_vars[cnt]);
342  }
343  CNF mustBe0(mustBe);
344  CNF mustBe1(mustBe);
345  for(size_t cnt = 0; cnt < ctrl.size(); ++cnt)
346  {
347  mustBe0.add2LitClause(-act_vars[cnt], ctrl[cnt]);
348  mustBe0.add2LitClause(-act_vars[cnt], -rename_map[ctrl[cnt]]);
349  mustBe1.add2LitClause(-act_vars[cnt], -ctrl[cnt]);
350  mustBe1.add2LitClause(-act_vars[cnt], rename_map[ctrl[cnt]]);
351  }
352 
353  // now we initialize the solvers:
354  vector<int> vars_to_keep;
355  vars_to_keep.reserve(ipc_.size() + ctrl.size() + ctrl.size());
356  vars_to_keep.insert(vars_to_keep.end(), ipc_.begin(), ipc_.end());
357  for(size_t cnt = 0; cnt < ctrl.size(); ++cnt)
358  {
359  vars_to_keep.push_back(act_vars[cnt]);
360  vars_to_keep.push_back(rename_map[ctrl[cnt]]);
361  }
362  check_->startIncrementalSession(vars_to_keep, false);
363  check_->incAddCNF(mustBe0);
364  gen_->startIncrementalSession(vars_to_keep, false);
365  gen_->incAddCNF(mustBe1);
366 
367  vector<int> dep_vars(ipc_);
368  for(size_t ctrl_cnt = 0; ctrl_cnt < ctrl.size(); ++ctrl_cnt)
369  {
370  statistics.notifyBeforeCtrlSignal();
371  int current_ctrl = ctrl[ctrl_cnt];
372  Utils::remove(dep_vars, current_ctrl);
373  vector<int> fixed(1, act_vars[ctrl_cnt]);
374 
375  // now we can do the actual learning loop:
376  CNF solution;
377  while(true)
378  {
379  if(coordinator_.extr_command_ == EXTR_STOP)
380  return;
381 
382  vector<int> false_pos;
383  // compute a false-positives (ctrl-signal is 1 but must not be):
384  statistics.notifyBeforeClauseComp();
385  bool false_pos_exists = check_->incIsSatModelOrCore(none, fixed, dep_vars, false_pos);
386  statistics.notifyAfterClauseComp();
387  if(!false_pos_exists)
388  break;
389 
390  if(coordinator_.extr_command_ == EXTR_STOP)
391  return;
392 
393  // generalize the false-positives:
394  // (compute a larger set of situations (including the false-positive) for which
395  // setting the control signal to 0 is allowed:
396  vector<int> gen_false_pos;
397  statistics.notifyBeforeClauseMin();
398  bool sat = gen_->incIsSatModelOrCore(false_pos, fixed, none, gen_false_pos);
399  MASSERT(!sat, "Impossible.");
400 
401  statistics.notifyAfterClauseMin(false_pos.size(), gen_false_pos.size());
402  solution.addNegCubeAsClause(gen_false_pos);
403  gen_false_pos.push_back(act_vars[ctrl_cnt]);
404  check_->incAddNegCubeAsClause(gen_false_pos);
405  }
406 
407  if(coordinator_.extr_command_ == EXTR_STOP)
408  return;
409 
410  // dump solution in AIGER format:
411  addToStandAloneAiger(current_ctrl, solution);
412  // store it for future optimization or dumping:
413  impl.push_back(solution);
414 
415  if(coordinator_.extr_command_ == EXTR_STOP)
416  return;
417 
418  // re-substitution:
419  CNF orig_resub = makeEq(current_ctrl, solution);
420  c_eq_impl.push_back(orig_resub);
421  check_->incAddCNF(orig_resub);
422  gen_->incAddCNF(orig_resub);
423  solution.renameVars(rename_map);
424  CNF renamed_resub = makeEq(rename_map[current_ctrl], solution);
425  check_->incAddCNF(renamed_resub);
426  gen_->incAddCNF(renamed_resub);
427  statistics.notifyAfterCtrlSignal();
428  }
429 
430  if(coordinator_.extr_command_ == EXTR_STOP)
431  return;
432  statistics.notifyBeforeABC(standalone_circuit_->num_ands);
433  aiger *opt = CNFImplExtractor::optimizeWithABC(standalone_circuit_);
434  aiger_reset(standalone_circuit_);
435  standalone_circuit_ = opt;
436  statistics.notifyAfterABC(standalone_circuit_->num_ands);
437  done_ = true;
438 
439  // Now we are done in principle. Let's now decide if we have time for a second minimization
440  // round:
441  bool second_run = true;
442  if(standalone_circuit_->num_ands <= 1)
443  second_run = false;
444  size_t elapsed_in_first_round = Stopwatch::getRealTimeSec(start_time);
445  size_t remaining = Options::instance().getEstimatedTimeRemaining();
446  if(remaining == 0 || remaining < 10 * elapsed_in_first_round)
447  second_run = false;
448 
449  if(second_run)
450  {
451  // Now the second step: try to reduce the implementations further
452  gen_->doMinCores(true);
453  CNF mustBe(neg_win_region_);
454  mustBe.swapPresentToNext();
455  mustBe.addCNF(win_region_);
456  mustBe.addCNF(AIG2CNF::instance().getTrans());
457  mustBe.setVarValue(VarManager::instance().getPresErrorStateVar(), false);
458  for(size_t cnt = 0; cnt < ctrl.size() - 1; ++cnt)
459  {
460  // eq_vars[cnt] implies that ctrl[cnt + 1] is set to the old solution
461  const list<vector<int> > &clauses = c_eq_impl[cnt+1].getClauses();
462  for(CNF::ClauseConstIter it = clauses.begin(); it != clauses.end(); ++it)
463  {
464  vector<int> cl(*it);
465  cl.push_back(-eq_vars[cnt]);
466  mustBe.addClause(cl);
467  }
468  // eq_vars[cnt] implies that eq_vars[cnt+1] is set
469  mustBe.add2LitClause(-eq_vars[cnt], eq_vars[cnt+1]);
470  mustBe.add2LitClause(-act_vars[cnt], eq_vars[cnt]);
471  }
472  CNF mustBe0(mustBe);
473  CNF mustBe1(mustBe);
474  for(size_t cnt = 0; cnt < ctrl.size(); ++cnt)
475  {
476  mustBe0.add2LitClause(-act_vars[cnt], ctrl[cnt]);
477  mustBe1.add2LitClause(-act_vars[cnt], -ctrl[cnt]);
478  }
479 
480  // now we initialize the solvers:
481  vector<int> vars_to_keep;
482  vars_to_keep.reserve(ipc_.size() + act_vars.size());
483  vars_to_keep.insert(vars_to_keep.end(), ipc_.begin(), ipc_.end());
484  vars_to_keep.insert(vars_to_keep.end(), act_vars.begin(), act_vars.end());
485  check_->startIncrementalSession(vars_to_keep, false);
486  check_->incAddCNF(mustBe0);
487  gen_->startIncrementalSession(vars_to_keep, false);
488  gen_->incAddCNF(mustBe1);
489 
490  for(size_t ctrl_cnt = 0; ctrl_cnt < ctrl.size(); ++ctrl_cnt)
491  {
492  statistics.notifyBeforeCtrlSignal();
493  int current_ctrl = ctrl[ctrl_cnt];
494  vector<int> fixed(1, act_vars[ctrl_cnt]);
495 
496  CNF solution;
497  while(impl[ctrl_cnt].getNrOfClauses() > 0)
498  {
499  vector<int> neg_clause = impl[ctrl_cnt].removeSmallest();
500  Utils::negateLiterals(neg_clause);
501  vector<int> smaller_cube;
502  statistics.notifyBeforeClauseMin();
503 
504  if(coordinator_.extr_command_ == EXTR_STOP)
505  return;
506 
507  bool sat = gen_->incIsSatModelOrCore(neg_clause, fixed, none, smaller_cube);
508  statistics.notifyAfterClauseMin(neg_clause.size(), smaller_cube.size());
509  MASSERT(!sat, "Impossible");
510  solution.addNegCubeAsClause(smaller_cube);
511  smaller_cube.push_back(act_vars[ctrl_cnt]);
512  check_->incAddNegCubeAsClause(smaller_cube);
513  statistics.notifyBeforeClauseComp();
514 
515  if(coordinator_.extr_command_ == EXTR_STOP)
516  return;
517 
518  sat = check_->incIsSat(fixed);
519  statistics.notifyAfterClauseComp();
520  if(!sat)
521  break;
522  }
523 
524  if(coordinator_.extr_command_ == EXTR_STOP)
525  return;
526 
527  addToStandAloneAiger2(current_ctrl, solution);
528  c_eq_impl[ctrl_cnt] = makeEq(current_ctrl, solution);
529  check_->incAddCNF(c_eq_impl[ctrl_cnt]);
530  gen_->incAddCNF(c_eq_impl[ctrl_cnt]);
531  }
532 
533  if(coordinator_.extr_command_ == EXTR_STOP)
534  return;
535  statistics.notifyBeforeABC(standalone_circuit2_->num_ands);
536  aiger *opt = CNFImplExtractor::optimizeWithABC(standalone_circuit2_);
537  aiger_reset(standalone_circuit2_);
538  standalone_circuit2_ = opt;
539  statistics.notifyAfterABC(standalone_circuit2_->num_ands);
540 
541  // swap:
542  aiger *tmp = standalone_circuit_;
543  standalone_circuit_ = standalone_circuit2_;
544  standalone_circuit2_ = tmp;
545  did_second_run_ = true;
546  }
547 
548  size_t elapsed_until_done = Stopwatch::getRealTimeSec(start_time);
549  waitForOthersToFinish(elapsed_until_done);
550 }

553 void ParExtractorWorker::runSATDep()
554 {
555  PointInTime start_time = Stopwatch::start();
556 
557  const vector<int> &ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL);
558  typedef map<int, set<int> >::const_iterator MapConstIter;
559  CNF trans = AIG2CNF::instance().getTrans();
560  vector<int> none;
561 
562  CNF next_win(win_region_);
563  next_win.swapPresentToNext();
564  CNF leave_win(neg_win_region_);
565  leave_win.swapPresentToNext();
566  leave_win.addCNF(win_region_);
567 
568  vector<CNF> c_eq_impl;
569  c_eq_impl.reserve(ctrl.size());
570  vector<CNF> impl;
571  impl.reserve(ctrl.size());
572 
573  vector<int> dep_vars(ipc_);
574  map<int, set<int> > ctrl_dep_on;
575  for(size_t ctrl_cnt = 0; ctrl_cnt < ctrl.size(); ++ctrl_cnt)
576  {
577  statistics.notifyBeforeCtrlSignal();
578  int current_ctrl = ctrl[ctrl_cnt];
579  Utils::remove(dep_vars, current_ctrl);
580  const map<int, set<VarInfo> > &trans_deps = AIG2CNF::instance().getTmpDepsTrans();
581  vector<int> real_dep_vars;
582  real_dep_vars.reserve(dep_vars.size() + trans_deps.size() + ctrl_dep_on.size());
583  real_dep_vars.insert(real_dep_vars.end(), dep_vars.begin(), dep_vars.end());
584  // check out on which of the already computed ctrl signals we can depend:
585  for(map<int, set<int> >::iterator it = ctrl_dep_on.begin(); it != ctrl_dep_on.end(); ++it)
586  if(it->second.count(current_ctrl) == 0)
587  real_dep_vars.push_back(it->first);
588  // check out on which temporary variables of the transition relation we can depend:
589  for(AIG2CNF::DepConstIter it = trans_deps.begin(); it != trans_deps.end(); ++it)
590  {
591  bool can_dep_on_this_tmp = true;
592  for(set<VarInfo>::const_iterator i2 = it->second.begin(); i2 != it->second.end(); ++i2)
593  {
594  if(i2->getKind() == VarInfo::CTRL && !Utils::contains(real_dep_vars, i2->getLitInCNF()))
595  {
596  can_dep_on_this_tmp = false;
597  break;
598  }
599  }
600  if(can_dep_on_this_tmp)
601  real_dep_vars.push_back(it->first);
602  }
603 
604  // we need to rename all variables except for the dep_vars:
605  int nr_of_vars = next_free_cnf_lit_;
606  int local_next_free_var = nr_of_vars;
607  vector<int> rename_map(nr_of_vars, 0);
608  for(int cnt = 1; cnt < nr_of_vars; ++cnt)
609  rename_map[cnt] = local_next_free_var++;
610  for(size_t cnt = 0; cnt < real_dep_vars.size(); ++cnt)
611  rename_map[real_dep_vars[cnt]] = real_dep_vars[cnt];
612 
613  CNF canBe1(trans);
614  canBe1.setVarValue(current_ctrl, true);
615  canBe1.addCNF(next_win);
616  canBe1.renameVars(rename_map);
617  CNF mustBe1(trans);
618  mustBe1.setVarValue(current_ctrl, false);
619  mustBe1.addCNF(leave_win);
620  mustBe1.addCNF(canBe1);
621  // just to be sure that the solution does not depend on our artificial state variable
622  // (this could otherwise happen if we do not use minimal cores):
623  mustBe1.setVarValue(VarManager::instance().getPresErrorStateVar(), false);
624 
625  CNF canBe0(trans);
626  canBe0.setVarValue(current_ctrl, false);
627  canBe0.addCNF(next_win);
628  canBe0.renameVars(rename_map);
629  CNF mustBe0(trans);
630  mustBe0.setVarValue(current_ctrl, true);
631  mustBe0.addCNF(leave_win);
632  mustBe0.addCNF(canBe0);
633  // just to be sure that the solution does not depend on our artificial state variable
634  // (this could otherwise happen if we do not use minimal cores):
635  mustBe0.setVarValue(VarManager::instance().getPresErrorStateVar(), false);
636 
637  // now we can do the actual learning loop:
638  check_->startIncrementalSession(real_dep_vars, false);
639  check_->incAddCNF(mustBe0);
640  gen_->startIncrementalSession(real_dep_vars, false);
641  gen_->incAddCNF(mustBe1);
642  CNF solution;
643  while(true)
644  {
645 
646  if(coordinator_.extr_command_ == EXTR_STOP)
647  return;
648 
649  vector<int> false_pos;
650  // compute a false-positives (ctrl-signal is 1 but must not be):
651  statistics.notifyBeforeClauseComp();
652  bool false_pos_exists = check_->incIsSatModelOrCore(none, real_dep_vars, false_pos);
653 
654  statistics.notifyAfterClauseComp();
655  if(!false_pos_exists)
656  break;
657 
658  if(coordinator_.extr_command_ == EXTR_STOP)
659  return;
660 
661  // generalize the false-positives:
662  // (compute a larger set of situations (including the false-positive) for which
663  // setting the control signal to 0 is allowed:
664  vector<int> gen_false_pos;
665  statistics.notifyBeforeClauseMin();
666  Utils::randomize(false_pos);
667  bool sat = gen_->incIsSatModelOrCore(false_pos, none, gen_false_pos);
668  MASSERT(!sat, "Impossible.");
669  statistics.notifyAfterClauseMin(false_pos.size(), gen_false_pos.size());
670  solution.addNegCubeAsClause(gen_false_pos);
671  check_->incAddNegCubeAsClause(gen_false_pos);
672  }
673 
674  if(coordinator_.extr_command_ == EXTR_STOP)
675  return;
676 
677  // dump solution in AIGER format:
678  addToStandAloneAiger(current_ctrl, solution);
679  // store solution for future optimization:
680  impl.push_back(solution);
681  // re-substitution:
682  CNF resub = makeEq(current_ctrl, solution);
683  trans.addCNF(resub);
684  c_eq_impl.push_back(resub);
685 
686  // update the dependencies:
687  // 1: find out on which of the remaining ctrl-signals current_ctrl depends on
688  // 1a: other control signals:
689  set<int> ctrl_dep;
690  for(size_t cnt = 0; cnt < ctrl.size(); ++cnt)
691  if(solution.contains(ctrl[cnt]))
692  ctrl_dep.insert(ctrl[cnt]);
693  // 1b: temporary signals which, in turn, may depend on ctrl-signals:
694  for(AIG2CNF::DepConstIter it = trans_deps.begin(); it != trans_deps.end(); ++it)
695  if(solution.contains(it->first))
696  for(set<VarInfo>::const_iterator i2 = it->second.begin(); i2 != it->second.end(); ++i2)
697  if(i2->getKind() == VarInfo::CTRL)
698  ctrl_dep.insert(i2->getLitInCNF());
699  // 1c: some ctrl-signals may already be defined; we need to take their deps:
700  for(MapConstIter it = ctrl_dep_on.begin(); it != ctrl_dep_on.end(); ++it)
701  {
702  if(ctrl_dep.count(it->first))
703  {
704  ctrl_dep.erase(it->first);
705  ctrl_dep.insert(it->second.begin(), it->second.end());
706  }
707  }
708  // 2: other defined ctrl-signals may depend on current_ctrl; we need to update their deps:
709  for(map<int, set<int> >::iterator it = ctrl_dep_on.begin(); it != ctrl_dep_on.end(); ++it)
710  {
711  if(it->second.count(current_ctrl))
712  {
713  it->second.erase(current_ctrl);
714  it->second.insert(ctrl_dep.begin(), ctrl_dep.end());
715  }
716  }
717  ctrl_dep_on[current_ctrl] = ctrl_dep;
718  // 3: check for cyclic dependencies (just for debugging purposes)
719  for(MapConstIter it = ctrl_dep_on.begin(); it != ctrl_dep_on.end(); ++it)
720  {
721  DASSERT(it->second.count(it->first) == 0, "Cyclic dependency");
722  }
723  statistics.notifyAfterCtrlSignal();
724  }
725 
726  if(coordinator_.extr_command_ == EXTR_STOP)
727  return;
728  // finally, we have to copy AND gates from the transition relation:
729  insertMissingAndFromTrans(standalone_circuit_);
730  if(coordinator_.extr_command_ == EXTR_STOP)
731  return;
732 
733  // and do the optimization with ABC
734  statistics.notifyBeforeABC(standalone_circuit_->num_ands);
735  aiger *opt = CNFImplExtractor::optimizeWithABC(standalone_circuit_);
736  aiger_reset(standalone_circuit_);
737  standalone_circuit_ = opt;
738  statistics.notifyAfterABC(standalone_circuit_->num_ands);
739  done_ = true;
740 
741  // Now we are done in principle. Let's now decide if we have time for a second minimization
742  // round:
743  bool second_run = true;
744  if(standalone_circuit_->num_ands <= 1)
745  second_run = false;
746  size_t elapsed_in_first_round = Stopwatch::getRealTimeSec(start_time);
747  size_t remaining = Options::instance().getEstimatedTimeRemaining();
748  if(remaining == 0 || remaining < 10 * elapsed_in_first_round)
749  second_run = false;
750 
751  if(second_run)
752  {
753  // Now the second step: try to reduce the implementations further
754  gen_->doMinCores(true);
755  CNF neg_rel(leave_win);
756  neg_rel.addCNF(AIG2CNF::instance().getTrans());
757  neg_rel.setVarValue(VarManager::instance().getPresErrorStateVar(), false);
758 
759  for(size_t ctrl_cnt = 0; ctrl_cnt < ctrl.size(); ++ctrl_cnt)
760  {
761  statistics.notifyBeforeCtrlSignal();
762  int current_ctrl = ctrl[ctrl_cnt];
763  CNF neg_rel_fixed(neg_rel);
764  for(size_t ctrl_cnt2 = 0; ctrl_cnt2 < ctrl.size(); ++ctrl_cnt2)
765  if(ctrl_cnt2 != ctrl_cnt)
766  neg_rel_fixed.addCNF(c_eq_impl[ctrl_cnt2]);
767  dep_vars = impl[ctrl_cnt].getVars();
768  check_->startIncrementalSession(dep_vars, false);
769  check_->incAddCNF(neg_rel_fixed);
770  check_->incAddUnitClause(current_ctrl);
771  gen_->startIncrementalSession(dep_vars, false);
772  gen_->incAddCNF(neg_rel_fixed);
773  gen_->incAddUnitClause(-current_ctrl);
774 
775  CNF solution;
776  while(impl[ctrl_cnt].getNrOfClauses() > 0)
777  {
778  vector<int> neg_clause = impl[ctrl_cnt].removeSmallest();
779  Utils::negateLiterals(neg_clause);
780  vector<int> smaller_cube;
781 
782  if(coordinator_.extr_command_ == EXTR_STOP)
783  return;
784 
785  statistics.notifyBeforeClauseMin();
786  bool sat = gen_->incIsSatModelOrCore(neg_clause, none, smaller_cube);
787  statistics.notifyAfterClauseMin(neg_clause.size(), smaller_cube.size());
788  MASSERT(!sat, "Impossible");
789  solution.addNegCubeAsClause(smaller_cube);
790 
791  if(coordinator_.extr_command_ == EXTR_STOP)
792  return;
793 
794  check_->incAddNegCubeAsClause(smaller_cube);
795  statistics.notifyBeforeClauseComp();
796  sat = check_->incIsSat();
797  statistics.notifyAfterClauseComp();
798  if(!sat)
799  break;
800  }
801  if(coordinator_.extr_command_ == EXTR_STOP)
802  return;
803  impl[ctrl_cnt] = solution;
804  addToStandAloneAiger2(current_ctrl, solution);
805  c_eq_impl[ctrl_cnt] = makeEq(current_ctrl, solution);
806  statistics.notifyAfterCtrlSignal();
807  }
808 
809  if(coordinator_.extr_command_ == EXTR_STOP)
810  return;
811  // finally, we have to copy AND gates from the transition relation:
812  insertMissingAndFromTrans(standalone_circuit2_);
813 
814  if(coordinator_.extr_command_ == EXTR_STOP)
815  return;
816  statistics.notifyBeforeABC(standalone_circuit2_->num_ands);
817  aiger *opt = CNFImplExtractor::optimizeWithABC(standalone_circuit2_);
818  aiger_reset(standalone_circuit2_);
819  standalone_circuit2_ = opt;
820  statistics.notifyAfterABC(standalone_circuit2_->num_ands);
821 
822  // swap:
823  aiger *tmp = standalone_circuit_;
824  standalone_circuit_ = standalone_circuit2_;
825  standalone_circuit2_ = tmp;
826  did_second_run_ = true;
827 
828  }
829  size_t elapsed_until_done = Stopwatch::getRealTimeSec(start_time);
830  waitForOthersToFinish(elapsed_until_done);
831 }

870 // -------------------------------------------------------------------------------------------
871 CNF ParExtractorWorker::makeEq(int var, CNF impl, vector<int> *new_temps)
872 {
873  CNF res;
874  // re-substitution:
875  // var -> impl
876  const list<vector<int> > &impl_clauses = impl.getClauses();
877  for(CNF::ClauseConstIter it = impl_clauses.begin(); it != impl_clauses.end(); ++it)
878  {
879  vector<int> clause(*it);
880  clause.push_back(-var);
881  res.addClause(clause);
882  }
883  // impl -> var
884  // begin negation of impl
885  const list<vector<int> > &original = impl.getClauses();
886  list<vector<int> > neg;
887  vector<int> one_clause_false;
888  one_clause_false.reserve(original.size() + 1);
889  for(CNF::ClauseConstIter it = original.begin(); it != original.end(); ++it)
890  {
891  if(it->size() == 1)
892  one_clause_false.push_back(-(*it)[0]);
893  else
894  {
895  int clause_false_lit = next_free_cnf_lit_++;
896  if(new_temps != NULL)
897  new_temps->push_back(clause_false_lit);
898  one_clause_false.push_back(clause_false_lit);
899  for(size_t lit_cnt = 0; lit_cnt < it->size(); ++lit_cnt)
900  {
901  vector<int> twolitclause(2,0);
902  twolitclause[0] = -clause_false_lit;
903  twolitclause[1] = -((*it)[lit_cnt]);
904  neg.push_back(twolitclause);
905  }
906  }
907  }
908  // here comes the implication:
909  one_clause_false.push_back(var);
910  neg.push_back(one_clause_false);
911  // end negation of impl
912 
913  CNF neg_impl;
914  neg_impl.swapWith(neg);
915  res.addCNF(neg_impl);
916 
917  return res;
918 }

921 void ParExtractorWorker::addToStandAloneAiger(int ctrl_var, const CNF &solution)
922 {
923  const list<vector<int> > &solution_clauses = solution.getClauses();
924 
925  // if we do not have any clauses, the result is true:
926  if(solution_clauses.size() == 0)
927  {
928  aiger_add_and(standalone_circuit_, cnfToAig(ctrl_var), 1, 1);
929  return;
930  }
931 
932  // if one clause is empty, the result is false:
933  for(CNF::ClauseConstIter it = solution_clauses.begin(); it != solution_clauses.end(); ++it)
934  {
935  if(it->size() == 0)
936  {
937  aiger_add_and(standalone_circuit_, cnfToAig(ctrl_var), 0, 0);
938  return;
939  }
940  }
941 
942  int and_res = 0;
943  for(CNF::ClauseConstIter it = solution_clauses.begin(); it != solution_clauses.end(); ++it)
944  {
945  int or_res = cnfToAig((*it)[0]);
946  for(size_t lit_cnt = 1; lit_cnt < it->size(); ++lit_cnt)
947  or_res = makeOr(or_res, cnfToAig((*it)[lit_cnt]));
948  if(it == solution_clauses.begin())
949  and_res = or_res;
950  else
951  and_res = makeAnd(and_res, or_res);
952  }
953  aiger_add_and(standalone_circuit_, cnfToAig(ctrl_var), and_res, 1);
954 }

1038 // -------------------------------------------------------------------------------------------
1039 void ParExtractorWorker::insertMissingAndFromTrans(aiger *standalone_circuit)
1040 {
1041  VarManager &VM = VarManager::instance();
1042 
1043  // Step 1: find out which variables are not yet defined in the standalone_circuit_:
1044  set<int> ref_aiger_vars;
1045  for(unsigned and_cnt = 0; and_cnt < standalone_circuit->num_ands; ++and_cnt)
1046  {
1047  ref_aiger_vars.insert(aiger_strip(standalone_circuit->ands[and_cnt].rhs0));
1048  ref_aiger_vars.insert(aiger_strip(standalone_circuit->ands[and_cnt].rhs1));
1049  }
1050  set<int> miss_cnf_vars;
1051  list<int> analyze_further_queue;
1052  const map<int, set<VarInfo> > &deps = AIG2CNF::instance().getTmpDeps();
1053  for(AIG2CNF::DepConstIter it = deps.begin(); it != deps.end(); ++it)
1054  {
1055  if(ref_aiger_vars.count(cnf_var_to_standalone_aig_var_[it->first]))
1056  {
1057  miss_cnf_vars.insert(it->first);
1058  analyze_further_queue.push_back(it->first);
1059  }
1060  }
1061  // Step 2: build the transitive closure of the missing variables:
1062  while(!analyze_further_queue.empty())
1063  {
1064  int analyze = analyze_further_queue.front();
1065  analyze_further_queue.pop_front();
1066  AIG2CNF::DepConstIter d = deps.find(analyze);
1067  MASSERT(d != deps.end(), "Impossible.");
1068  for(set<VarInfo>::const_iterator it = d->second.begin(); it != d->second.end(); ++it)
1069  {
1070  if(it->getKind() == VarInfo::TMP && miss_cnf_vars.count(it->getLitInCNF()) == 0)
1071  {
1072  miss_cnf_vars.insert(it->getLitInCNF());
1073  analyze_further_queue.push_back(it->getLitInCNF());
1074  }
1075  }
1076  }
1077 
1078  // Step 3: add missing variables by copying them from the transition relation
1079  const char *error = NULL;
1080  aiger *aig = aiger_init();
1081  const string &file = Options::instance().getAigInFileName();
1082  error = aiger_open_and_read_from_file (aig, file.c_str());
1083  MASSERT(error == NULL, "Could not open AIGER file " << file << " (" << error << ").");
1084  for(unsigned and_cnt = 0; and_cnt < aig->num_ands; ++and_cnt)
1085  {
1086  int lhs_cnf = VM.aigLitToCnfLit(aig->ands[and_cnt].lhs);
1087  if(miss_cnf_vars.count(lhs_cnf))
1088  {
1089  unsigned r0 = cnfToAig(VM.aigLitToCnfLit(aig->ands[and_cnt].rhs0));
1090  unsigned r1 = cnfToAig(VM.aigLitToCnfLit(aig->ands[and_cnt].rhs1));
1091  aiger_add_and(standalone_circuit, cnfToAig(lhs_cnf), r0, r1);
1092  }
1093  }
1094  aiger_reset(aig);
1095 }
