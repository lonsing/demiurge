63 vector<int> Utils::extract(const vector<int> &cube_or_clause, VarInfo::VarKind kind)
64 {
65  const vector<int> &rel_vars = VarManager::instance().getVarsOfType(kind);
66  vector<int> res;
67  res.reserve(cube_or_clause.size());
68  for(size_t c1 = 0; c1 < cube_or_clause.size(); ++c1)
69  {
70  int var = cube_or_clause[c1];
71  if(var < 0)
72  var = -var;
73  for(size_t c2 = 0; c2 < rel_vars.size(); ++c2)
74  {
75  if(var == rel_vars[c2])
76  {
77  res.push_back(cube_or_clause[c1]);
78  break;
79  }
80  }
81  }
82  return res;
83 }

194 void Utils::swapPresentToNext(vector<int> &vec)
195 {
196  VarManager &VM = VarManager::instance();
197  vector<int> olt_to_new_temp_var_map(VM.getMaxCNFVar()+1, 0);
198  const vector<int>& ps_vars = VM.getVarsOfType(VarInfo::PRES_STATE);
199  const vector<int>& ns_vars = VM.getVarsOfType(VarInfo::NEXT_STATE);
200  MASSERT(ps_vars.size() == ns_vars.size(), "Sizes do not match.");
201  for(size_t cnt = 0; cnt < ps_vars.size(); ++cnt)
202  olt_to_new_temp_var_map[ps_vars[cnt]] = ns_vars[cnt];
203 
204  for(size_t lit_cnt = 0; lit_cnt < vec.size(); ++lit_cnt)
205  {
206  int old_var = (vec[lit_cnt] < 0) ? -vec[lit_cnt] : vec[lit_cnt];
207  if(vec[lit_cnt] < 0)
208  vec[lit_cnt] = -olt_to_new_temp_var_map[old_var];
209  else
210  vec[lit_cnt] = olt_to_new_temp_var_map[old_var];
211  }
212 }

250 {
251  if(subset.size() > superset.size())
252  return false;
253  for(size_t cnt1 = 0; cnt1 < subset.size(); ++cnt1)
254  {
255  bool found = false;
256  int to_find = subset[cnt1];
257  for(size_t cnt2 = 0; cnt2 < superset.size(); ++cnt2)
258  {
259  if(superset[cnt2] == to_find)
260  {
261  found = true;
262  break;
263  }
264  }
265  if(!found)
266  return false;
267  }
268  return true;

272 bool Utils::compressStateCNF(CNF &cnf, bool hardcore)
273 {
274  if(cnf.getNrOfClauses() == 0)
275  return false;
276 
277  // size_t or_cl = cnf.getNrOfClauses();
278  // size_t or_lits = cnf.getNrOfLits();
279 
280  bool something_changed = false;
281 
282  SatSolver *solver = Options::instance().getSATSolver(false, true);
283 
284  if(hardcore)
285  {
286  // remove not only literals from clauses before we turn to removing clauses
287  list<vector<int> > clauses = cnf.getClauses();
288  vector<int> none;
289  solver->startIncrementalSession(VarManager::instance().getVarsOfType(VarInfo::PRES_STATE),
290  false);
291  solver->incAddCNF(cnf);
292  cnf.clear();
293  for(CNF::ClauseIter it = clauses.begin(); it != clauses.end(); ++it)
294  {
295  vector<int> neg_clause(*it);
296  negateLiterals(neg_clause);
297  vector<int> smaller_neg_clause;
298  bool sat = solver->incIsSatModelOrCore(neg_clause,none,smaller_neg_clause);
299  MASSERT(!sat, "Impossible.");
300  cnf.addNegCubeAsClause(smaller_neg_clause);
301  }
302  }
303 
304  solver->startIncrementalSession(VarManager::instance().getVarsOfType(VarInfo::PRES_STATE),
305  false);
306  CNF still_to_process(cnf);
307  cnf.clear();
308  vector<int> first = still_to_process.removeSmallest();
309  cnf.addClause(first);
310  solver->incAddClause(first);
311  while(still_to_process.getNrOfClauses() != 0)
312  {
313  vector<int> check_clause = still_to_process.removeSmallest();
314  vector<int> check_cube(check_clause);
315  negateLiterals(check_cube);
316  if(solver->incIsSat(check_cube))
317  {
318  solver->incAddClause(check_clause);
319  cnf.addClause(check_clause);
320  }
321  else
322  something_changed = true;
323  }
324  delete solver;
325  solver = NULL;
326  // if(something_changed)
327  // {
328  // size_t new_cl = cnf.getNrOfClauses();
329  // size_t new_lits = cnf.getNrOfLits();
330  // L_DBG("compressCNFs(): " << or_cl << " --> " << new_cl << " clauses.");
331  // L_DBG("compressCNFs(): " << or_lits << " --> " << new_lits << " literals.");
332  // }
333  return something_changed;
334 
335 }

363 void Utils::debugCheckWinReg(const CNF &winning_region, const CNF &neg_winning_region)
364 {
365 #ifndef NDEBUG
366  L_DBG("Checking winning region for correctness ...");
367 
368  // The initial state must be contained in the winning region:
369  SatSolver *sat_solver = Options::instance().getSATSolver();
370  CNF check(winning_region);
371  check.addCNF(AIG2CNF::instance().getInitial());
372  MASSERT(sat_solver->isSat(check), "Initial state is not winning.");
373 
374  // All states of the winning region must be safe
375  check.clear();
376  check.addCNF(winning_region);
377  check.addCNF(AIG2CNF::instance().getUnsafeStates());
378  MASSERT(!sat_solver->isSat(check), "Winning region is not safe.");
379 
380  // from the winning region, we can enforce to stay in the winning region:
381  // exists x,i: forall c: exists x': W(x) & T(x,i,c,x') & !W(x) must be unsat
382  QBFSolver *qbf_solver = Options::instance().getQBFSolver();
383  check.clear();
384  check.addCNF(neg_winning_region);
385  check.swapPresentToNext();
386  check.addCNF(winning_region);
387  check.addCNF(AIG2CNF::instance().getTrans());
388  vector<pair<VarInfo::VarKind, QBFSolver::Quant> > check_quant;
389  check_quant.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
390  check_quant.push_back(make_pair(VarInfo::INPUT, QBFSolver::E));
391  check_quant.push_back(make_pair(VarInfo::CTRL, QBFSolver::A));
392  check_quant.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
393  check_quant.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
394  MASSERT(!qbf_solver->isSat(check_quant, check), "Winning region is not inductive.");
395 
396  delete sat_solver;
397  sat_solver = NULL;
398  delete qbf_solver;
399  qbf_solver = NULL;
400 
401  L_DBG("All checks passed.");
402 #endif
403 }

406 void Utils::debugPrintCurrentMemUsage()
407 {
408  using std::ios_base;
409  using std::ifstream;
410  using std::string;
411 
412  double vm_usage = 0.0;
413  double resident_set = 0.0;
414 
415  // 'file' stat seems to give the most reliable results
416  //
417  ifstream stat_stream("/proc/self/stat",ios_base::in);
418 
419  // dummy vars for leading entries in stat that we don't care about
420  //
421  string pid, comm, state, ppid, pgrp, session, tty_nr;
422  string tpgid, flags, minflt, cminflt, majflt, cmajflt;
423  string utime, stime, cutime, cstime, priority, nice;
424  string O, itrealvalue, starttime;
425 
426  // the two fields we want
427  //
428  unsigned long vsize;
429  long rss;
430 
431  stat_stream >> pid >> comm >> state >> ppid >> pgrp >> session >> tty_nr
432  >> tpgid >> flags >> minflt >> cminflt >> majflt >> cmajflt
433  >> utime >> stime >> cutime >> cstime >> priority >> nice
434  >> O >> itrealvalue >> starttime >> vsize >> rss;
435 
436  stat_stream.close();
437 
438  long page_size_kb = sysconf(_SC_PAGE_SIZE) / 1024;
439  resident_set = rss * page_size_kb;
440  L_DBG("Resident set: " << resident_set << " kB.");
441  vm_usage = vsize / 1024.0;
442  L_DBG("Virtual Memory: " << vm_usage << " kB.");
443 }
