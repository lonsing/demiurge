228 {
229  VarManager &VM = VarManager::instance();

241  vec[lit_cnt] = -olt_to_new_temp_var_map[old_var];
242  else
243  vec[lit_cnt] = olt_to_new_temp_var_map[old_var];

317  if(hardcore)

371 void Utils::negateStateCNF(CNF &cnf)
372 {
373  CNF orig_cnf(cnf);
374  CNF cnf_to_learn(cnf);
375  VarManager::instance().push();
376  cnf_to_learn.negate();
377  cnf.clear();
378  const vector<int> &vars = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE);
379 
380  SatSolver *find_solver = Options::instance().getSATSolver(false, true);
381  find_solver->startIncrementalSession(vars, false);
382  find_solver->incAddCNF(orig_cnf);
383  SatSolver *gen_solver = Options::instance().getSATSolver(false, true);
384  gen_solver->startIncrementalSession(vars, false);
385  gen_solver->incAddCNF(cnf_to_learn);
386 
387  vector<int> model;
388  vector<int> none;
389  while(true)
390  {
391  bool sat = find_solver->incIsSatModelOrCore(none, vars, model);
392  if(!sat)
393  break;
394  //Generalize model as long as model implies orig_cnf
395  // as long as model & cnf_to_learn is unsat
396  vector<int> core;
397  sat = gen_solver->incIsSatModelOrCore(model, vars, core);
398  MASSERT(!sat, "Impossible.");
399  cnf.addNegCubeAsClause(core);
400  find_solver->incAddNegCubeAsClause(core);
401  }
402  VarManager::instance().pop();
403  delete find_solver;
404  find_solver = NULL;
405  delete gen_solver;
406  gen_solver = NULL;
407 }

410 void Utils::negateViaAig(CNF &cnf)
411 {
412  set<int> vars;
413  cnf.appendVarsTo(vars);
414  int max_cnf_var = vars.empty() ? 1 : *(vars.rbegin());
415  vector<int> cnf_to_aig(max_cnf_var + 1, 0);
416  vector<int> aig_to_cnf(vars.size() * 2 + 2, 0);
417  int next_free_aig_lit = 2;
418  aiger *raw = aiger_init();
419  for(set<int>::const_iterator it = vars.begin(); it != vars.end(); ++it)
420  {
421  aiger_add_input(raw, next_free_aig_lit, NULL);
422  cnf_to_aig[*it] = next_free_aig_lit;
423  aig_to_cnf[next_free_aig_lit] = *it;
424  aig_to_cnf[next_free_aig_lit+1] = -(*it);
425  next_free_aig_lit += 2;
426  }
427  int last_and = 1;
428  const list<vector<int> > &cl = cnf.getClauses();
429  for(CNF::ClauseConstIter it = cl.begin(); it != cl.end(); ++it)
430  {
431  const vector<int> &c = *it;
432  int last_or = 1;
433  if(c.size() >= 1)
434  last_or = c[0] < 0 ? cnf_to_aig[-c[0]] : cnf_to_aig[c[0]] + 1;
435  for(size_t cnt = 1; cnt < c.size(); ++cnt)
436  {
437  int next_cnf_to_aig = c[cnt] < 0 ? cnf_to_aig[-c[cnt]] : cnf_to_aig[c[cnt]] + 1;
438  aiger_add_and(raw, next_free_aig_lit, last_or, next_cnf_to_aig);
439  last_or = next_free_aig_lit;
440  next_free_aig_lit += 2;
441  }
442  if(last_and == 1)
443  last_and = aiger_not(last_or);
444  else
445  {
446  aiger_add_and(raw, next_free_aig_lit, last_and, aiger_not(last_or));
447  last_and = next_free_aig_lit;
448  next_free_aig_lit += 2;
449  }
450  }
451  aiger_add_output(raw, last_and, NULL);
452 
453  // done constructing AIGER version of cnf, now we optimize with ABC:
454  string tmp_in_file = Options::instance().getUniqueTmpFileName("optimize_win") + ".aig";
455  int succ = aiger_open_and_write_to_file(raw, tmp_in_file.c_str());
456  MASSERT(succ != 0, "Could not write out AIGER file for optimization with ABC.");
457 
458  // optimize circuit:
459  string path_to_abc = Options::instance().getTPDirName() + "/abc/abc/abc";
460  string tmp_out_file = Options::instance().getUniqueTmpFileName("optimized_win") + ".aig";
461  string abc_command = path_to_abc + " -c \"";
462  abc_command += "read_aiger " + tmp_in_file + ";";
463  abc_command += " strash; refactor -zl; rewrite -zl;";
464  abc_command += " strash; refactor -zl; rewrite -zl;";
465  abc_command += " strash; refactor -zl; rewrite -zl;";
466  abc_command += " strash; refactor -zl; rewrite -zl;";
467  abc_command += " strash; refactor -zl; rewrite -zl;";
468  abc_command += " strash; refactor -zl; rewrite -zl;";
469  abc_command += " rewrite -zl; dfraig;";
470  abc_command += " write_aiger -s " + tmp_out_file + "\"";
471  abc_command += " > /dev/null 2>&1";
472  int ret = system(abc_command.c_str());
473  ret = WEXITSTATUS(ret);
474  MASSERT(ret == 0, "ABC terminated with strange return code.");
475 
476  // read back the result:
477  aiger *res = aiger_init();
478  const char *err = aiger_open_and_read_from_file (res, tmp_out_file.c_str());
479  MASSERT(err == NULL, "Could not open optimized AIGER file "
480  << tmp_out_file << " (" << err << ").");
481  std::remove(tmp_in_file.c_str());
482  std::remove(tmp_out_file.c_str());
483 
484  // done optimizing the aiger circuit. Now encode it back onto a CNF:
485  cnf.clear();
486  aig_to_cnf.resize(aig_to_cnf.size() + (res->num_ands * 2), 0);
487  for(unsigned cnt = 0; cnt < res->num_ands; ++cnt)
488  {
489  int nw = VarManager::instance().createFreshTmpVar();
490  aig_to_cnf[res->ands[cnt].lhs] = nw;
491  aig_to_cnf[res->ands[cnt].lhs + 1] = -nw;
492  cnf.add2LitClause(aig_to_cnf[res->ands[cnt].rhs0], -nw);
493  cnf.add2LitClause(aig_to_cnf[res->ands[cnt].rhs1], -nw);
494  cnf.add3LitClause(-aig_to_cnf[res->ands[cnt].rhs0], -aig_to_cnf[res->ands[cnt].rhs1], nw);
495  }
496  cnf.add1LitClause(-aig_to_cnf[res->outputs[0].lit]);
497  aiger_reset(raw);
498  aiger_reset(res);
499 }

503 {

510  solver_cl->startIncrementalSession(VM.getAllNonTempVars(), false);
511  SatSolver *solver_lit = NULL;
512 
513  // Step 1: remove literals from present-state clauses if enabled:
514  if(false) // does not pay off
515  {
516  solver_lit = Options::instance().getSATSolver(false, true);
517  solver_lit->startIncrementalSession(VM.getAllNonTempVars(), false);
518  solver_lit->incAddCNF(ps_cnf);

520  ps_cnf.swapWith(clauses);

522  {

524  negateLiterals(neg_clause);

527  MASSERT(!sat, "Impossible.");
528  ps_cnf.addNegCubeAsClause(smaller_neg_clause);
529  }
530  }
531 
532  // Step 2: remove present-state clauses:
533  CNF still_to_process;
534  still_to_process.swapWith(ps_cnf);
535  while(still_to_process.getNrOfClauses() != 0)
536  {

541  {

544  }
545  }
546 
547  ns_cnf = ps_cnf;
548  ns_cnf.swapPresentToNext();
549 
550  // Step 3: remove literals from next-state clauses if enabled:
551  if(hardcore)
552  {
553  solver_lit = Options::instance().getSATSolver(false, true);
554  solver_lit->startIncrementalSession(VM.getAllNonTempVars(), false);
555  solver_lit->incAddCNF(ps_cnf);
556  solver_lit->incAddCNF(AIG2CNF::instance().getTrans());
557  solver_lit->incAddCNF(ns_cnf);
558  list<vector<int> > clauses;
559  ns_cnf.swapWith(clauses);
560  for(CNF::ClauseIter it = clauses.begin(); it != clauses.end(); ++it)
561  {
562  vector<int> neg_clause(*it);
563  negateLiterals(neg_clause);
564  vector<int> smaller_neg_clause;
565  bool sat = solver_lit->incIsSatModelOrCore(neg_clause, none, smaller_neg_clause);
566  MASSERT(!sat, "Impossible.");
567  ns_cnf.addNegCubeAsClause(smaller_neg_clause);
568  }
569  }
570 
571  // The solver_cl still contains the present-state cnf
572  // Step 4: remove next-state clauses:
573  solver_cl->incAddCNF(AIG2CNF::instance().getTrans());
574  still_to_process = ns_cnf;
575  ns_cnf.clear();
576  while(still_to_process.getNrOfClauses() != 0)
577  {
578  vector<int> check_clause = still_to_process.removeSmallest();
579  vector<int> check_cube(check_clause);
580  negateLiterals(check_cube);
581  if(solver_cl->incIsSat(check_cube))
582  {
583  solver_cl->incAddClause(check_clause);
584  ns_cnf.addClause(check_clause);
585  }
586  }
587 
588  delete solver_cl;
589  delete solver_lit;
590  //L_DBG("compr-present: " << orig_cl << " --> " << ps_cnf.getNrOfClauses() << " clauses.");
591  //L_DBG("compr-present: " << orig_lits << " --> " << ps_cnf.getNrOfLits() << " literals.");
592  //L_DBG("compr-next: " << orig_cl << " --> " << ns_cnf.getNrOfClauses() << " clauses.");
593  //L_DBG("compr-next: " << orig_lits << " --> " << ns_cnf.getNrOfLits() << " literals.");
594 }

654 // -------------------------------------------------------------------------------------------
655 void Utils::debugCheckWinReg(const CNF &winning_region, const CNF &neg_winning_region)
656 {
657 #ifndef NDEBUG
658  L_DBG("Checking winning region for correctness ...");
659 
660  // The initial state must be contained in the winning region:
661  SatSolver *sat_solver = Options::instance().getSATSolver();
662  CNF check(winning_region);
663  check.addCNF(AIG2CNF::instance().getInitial());
664  MASSERT(sat_solver->isSat(check), "Initial state is not winning.");
665 
666  // All states of the winning region must be safe
667  check.clear();
668  check.addCNF(winning_region);
669  check.addCNF(AIG2CNF::instance().getUnsafeStates());
670  MASSERT(!sat_solver->isSat(check), "Winning region is not safe.");
671 
672  // from the winning region, we can enforce to stay in the winning region:
673  // exists x,i: forall c: exists x': W(x) & T(x,i,c,x') & !W(x) must be unsat
674  bool check_with_qbf = false;
675  if(check_with_qbf)
676  {
677  QBFSolver *qbf_solver = Options::instance().getQBFSolver();
678  check.clear();
679  check.addCNF(neg_winning_region);
680  check.swapPresentToNext();
681  check.addCNF(winning_region);
682  check.addCNF(AIG2CNF::instance().getTrans());
683  vector<pair<VarInfo::VarKind, QBFSolver::Quant> > check_quant;
684  check_quant.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
685  check_quant.push_back(make_pair(VarInfo::INPUT, QBFSolver::E));
686  check_quant.push_back(make_pair(VarInfo::CTRL, QBFSolver::A));
687  check_quant.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
688  check_quant.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
689  MASSERT(!qbf_solver->isSat(check_quant, check), "Winning region is not inductive.");
690  delete qbf_solver;
691  qbf_solver = NULL;
692  }
693  else
694  {
695  // SAT-based check:
696  const vector<int> &in = VarManager::instance().getVarsOfType(VarInfo::INPUT);
697  const vector<int> &pres = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE);
698  const vector<int> &ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL);
699  vector<int> sic;
700  sic.reserve(pres.size() + in.size() + ctrl.size());
701  sic.insert(sic.end(), pres.begin(), pres.end());
702  sic.insert(sic.end(), in.begin(), in.end());
703  sic.insert(sic.end(), ctrl.begin(), ctrl.end());
704  vector<int> si;
705  si.reserve(pres.size() + in.size());
706  si.insert(si.end(), pres.begin(), pres.end());
707  si.insert(si.end(), in.begin(), in.end());
708  CNF check_cnf(neg_winning_region);
709  check_cnf.swapPresentToNext();
710  check_cnf.addCNF(winning_region);
711  check_cnf.addCNF(AIG2CNF::instance().getTrans());
712  SatSolver *check_solver = Options::instance().getSATSolver(false, true);
713  check_solver->startIncrementalSession(sic, false);
714  check_solver->incAddCNF(check_cnf);
715  CNF gen_cnf(winning_region);
716  gen_cnf.swapPresentToNext();
717  gen_cnf.renameTmps();
718  gen_cnf.addCNF(winning_region);
719  gen_cnf.addCNF(AIG2CNF::instance().getTrans());
720  SatSolver *gen_solver = Options::instance().getSATSolver(false, true);
721  gen_solver->startIncrementalSession(sic, false);
722  gen_solver->incAddCNF(gen_cnf);
723 
724  vector<int> model_or_core;
725  while(true)
726  {
727  bool sat = check_solver->incIsSatModelOrCore(vector<int>(), si, model_or_core);
728  if(!sat)
729  break;
730  vector<int> state_input = model_or_core;
731  vector<int> state = Utils::extract(state_input, VarInfo::PRES_STATE);
732  vector<int> input = Utils::extract(state_input, VarInfo::INPUT);
733 
734  sat = gen_solver->incIsSatModelOrCore(state, input, ctrl, model_or_core);
735  MASSERT(sat, "Winning region is not inductive.");
736 
737  vector<int> resp(model_or_core);
738  sat = check_solver->incIsSatModelOrCore(state_input, resp, vector<int>(), model_or_core);
739  MASSERT(sat == false, "Impossible.");
740  check_solver->incAddNegCubeAsClause(model_or_core);
741  }
742  delete check_solver;
743  check_solver = NULL;
744  delete gen_solver;
745  gen_solver = NULL;
746  }
747 
748  delete sat_solver;
749  sat_solver = NULL;
750 
751  L_DBG("All checks passed.");
752 #endif
753 }
