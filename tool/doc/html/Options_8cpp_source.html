291  return true;

293  }
294  else if(arg == "-e")

296  ++arg_count;

298  {
299  cerr << "Option -t must be followed by a SAT solver name." << endl;

301  }

303  StringUtils::toLowerCaseIn(circuit_sat_solver_);

308  cerr << "Unknown SAT solver '" << circuit_sat_solver_ <<"'." << endl;

312  else if(arg.find("--timeout_hint=") == 0)

317  else if(arg == "-k")
318  {

320  if(arg_count >= argc)
321  {

374 // -------------------------------------------------------------------------------------------
375 BackEnd* Options::getBackEnd() const
376 {
377  if(back_end_ == "epr")
378  return new EPRSynthesizer();
379 
380  if(back_end_ == "learn")
381  {
382  if(mode_ == 0 || mode_ == 1 || mode_ == 4)
383  return new LearnSynthQBF(getCircuitExtractor());
384  else if(mode_ == 2 || mode_ == 3 || mode_ == 5 || mode_ == 6)
385  return new LearnSynthQBFInd(getCircuitExtractor());
386  else // m == 7, 8, 9, or 10
387  return new LearnSynthQBFInc(getCircuitExtractor());
388  }
389  if(back_end_ == "learn_sat")
390  return new LearnSynthSAT(getCircuitExtractor());
391  if(back_end_.find("lp") == 0)
392  {
393  string rest = back_end_.substr(2);
394  size_t nr_of_threads = thread::hardware_concurrency();
395  if(rest != "")
396  {
397  istringstream iss(rest);
398  iss >> nr_of_threads;
399  }
400  return new ParallelLearner(nr_of_threads, getCircuitExtractor());
401  }
402  if(back_end_ == "templ")
403  return new TemplateSynth(getCircuitExtractor());
404  if(back_end_ == "ifm")
405  return new IFM13Synth(getCircuitExtractor());
406  if(back_end_ == "load")
407  return new LoadSynth(getCircuitExtractor());
408  L_ERR("Unknown back-end '" << back_end_ <<"'.");
409  exit(-1);
410 }

431 CNFImplExtractor* Options::getCircuitExtractor() const
432 {
433  if(real_only_)
434  return NULL;
435  if(circuit_extraction_name_ == "qbfcert")
436  return new QBFCertImplExtractor();

440  {
441  string rest = circuit_extraction_name_.substr(2);
442  size_t nr_of_threads = thread::hardware_concurrency();
443  if(rest != "")
444  {
445  istringstream iss(rest);

451  return new InterpolImplExtractor();
452  if(circuit_extraction_name_ == "store")
453  return new StoreImplExtractor();

455  exit(-1);
