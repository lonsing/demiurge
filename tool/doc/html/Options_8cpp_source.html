68 bool Options::parse(int argc, char **argv)
69 {
70  for(int arg_count = 1; arg_count < argc; ++arg_count)
71  {
72  string arg(argv[arg_count]);
73  if(arg == "--help" || arg == "-h")
74  {
75  printHelp();
76  return true;
77  }
78  else if(arg == "--version" || arg == "-v")
79  {
80  cout << "demiurge version " << Options::VERSION << endl;
81  return true;
82  }
83  else if(arg == "--real_only" || arg == "-r")
84  {
85  real_only_ = true;
86  }
87  else if(arg.find("--in=") == 0)
88  {
89  aig_in_file_name_ = arg.substr(5, string::npos);
90  }
91  else if(arg == "-i")
92  {
93  ++arg_count;
94  if(arg_count >= argc)
95  {
96  cerr << "Option -i must be followed by a filename." << endl;
97  return true;
98  }
99  aig_in_file_name_ = string(argv[arg_count]);
100  }
101  else if(arg.find("--out=") == 0)
102  {
103  aig_out_file_name_ = arg.substr(6, string::npos);
104  }
105  else if(arg == "-o")
106  {
107  ++arg_count;
108  if(arg_count >= argc)
109  {
110  cerr << "Option -o must be followed by a filename." << endl;
111  return true;
112  }
113  aig_out_file_name_ = string(argv[arg_count]);
114  }
115  else if(arg.find("--print=") == 0)
116  {
117  print_string_ = arg.substr(8, string::npos);
118  }
119  else if(arg == "-p")
120  {
121  ++arg_count;
122  if(arg_count >= argc)
123  {
124  cerr << "Option -p must be followed by a string indicating what to print." << endl;
125  return true;
126  }
127  print_string_ = string(argv[arg_count]);
128  }
129  else if(arg.find("--backend=") == 0)
130  {
131  back_end_ = arg.substr(10, string::npos);
132  StringUtils::toLowerCaseIn(back_end_);
133  }
134  else if(arg == "-b")
135  {
136  ++arg_count;
137  if(arg_count >= argc)
138  {
139  cerr << "Option -b must be followed by a back-end name." << endl;
140  return true;
141  }
142  back_end_ = string(argv[arg_count]);
143  StringUtils::toLowerCaseIn(back_end_);
144  }
145  else if(arg.find("--mode=") == 0)
146  {
147  istringstream iss(arg.substr(7, string::npos));
148  iss >> mode_;
149  }
150  else if(arg == "-m")
151  {
152  ++arg_count;
153  if(arg_count >= argc)
154  {
155  cerr << "Option -m must be followed by an integer number." << endl;
156  return true;
157  }
158  istringstream iss(argv[arg_count]);
159  iss >> mode_;
160  }
161  else if(arg.find("--qbf_sv=") == 0)
162  {
163  qbf_solver_ = arg.substr(9, string::npos);
164  StringUtils::toLowerCaseIn(qbf_solver_);
165  if(qbf_solver_ != "depqbf_ext" &&
166  qbf_solver_ != "depqbf_api" &&
167  qbf_solver_ != "blo_dep_api" &&
168  qbf_solver_ != "qube_ext")
169  {
170  cerr << "Unknown QBF solver '" << qbf_solver_ <<"'." << endl;
171  return true;
172  }
173  }
174  else if(arg == "-q")
175  {
176  ++arg_count;
177  if(arg_count >= argc)
178  {
179  cerr << "Option -q must be followed by a QBF solver name." << endl;
180  return true;
181  }
182  qbf_solver_ = string(argv[arg_count]);
183  StringUtils::toLowerCaseIn(qbf_solver_);
184  if(qbf_solver_ != "depqbf_ext" &&
185  qbf_solver_ != "depqbf_api" &&
186  qbf_solver_ != "blo_dep_api" &&
187  qbf_solver_ != "qube_ext")
188  {
189  cerr << "Unknown QBF solver '" << qbf_solver_ <<"'." << endl;
190  return true;
191  }
192  }
193  else if(arg.find("--sat_sv=") == 0)
194  {
195  sat_solver_ = arg.substr(9, string::npos);
196  StringUtils::toLowerCaseIn(sat_solver_);
197  if(sat_solver_ != "lin_api" &&
198  sat_solver_ != "min_api" &&
199  sat_solver_ != "pic_api")
200  {
201  cerr << "Unknown SAT solver '" << sat_solver_ <<"'." << endl;
202  return true;
203  }
204  }
205  else if(arg == "-s")
206  {
207  ++arg_count;
208  if(arg_count >= argc)
209  {
210  cerr << "Option -s must be followed by a SAT solver name." << endl;
211  return true;
212  }
213  sat_solver_ = string(argv[arg_count]);
214  StringUtils::toLowerCaseIn(sat_solver_);
215  if(sat_solver_ != "lin_api" &&
216  sat_solver_ != "min_api" &&
217  sat_solver_ != "pic_api")
218  {
219  cerr << "Unknown SAT solver '" << sat_solver_ <<"'." << endl;
220  return true;
221  }
222  }
223  }
224  if(aig_in_file_name_ == "")
225  {
226  cerr << "No input file given." << endl;
227  return true;
228  }
229  initLogger();
230  if(aig_out_file_name_ == "")
231  {
232  L_WRN("No output file given. Defaulting to \"res_"<< aig_in_file_name_ << "\".");
233  aig_out_file_name_ = "res_" + aig_in_file_name_;
234  }
235  return false;
236 }

256 // -------------------------------------------------------------------------------------------
257 BackEnd* Options::getBackEnd() const
258 {
259  if(back_end_ == "learn")
260  {
261  if(mode_ == 0 || mode_ == 1 || mode_ == 4)
262  return new LearnSynthQBF();
263  else if(mode_ == 2 || mode_ == 3 || mode_ == 5 || mode_ == 6)
264  return new LearnSynthQBFInd();
265  else // m == 7 or 8 (experimental)
266  return new LearnSynthQBFInc();
267  }
268  if(back_end_ == "learn_sat")
269  return new LearnSynthSAT();
270  if(back_end_.find("lp") == 0)
271  {
272  string rest = back_end_.substr(2);
273  size_t nr_of_threads = thread::hardware_concurrency();
274  if(rest != "")
275  {
276  istringstream iss(rest);
277  iss >> nr_of_threads;
278  }
279  return new ParallelLearner(nr_of_threads);
280  }
281  if(back_end_ == "templ")
282  return new TemplateSynth();
283  if(back_end_ == "epr")
284  return new EPRSynthesizer();
285  if(back_end_ == "ifm")
286  return new IFM13Synth();
287  L_ERR("Unknown back-end '" << back_end_ <<"'.");
288  exit(-1);
289 }

465 void Options::initLogger() const
466 {
467  Logger &logger = Logger::instance();
468  if(print_string_.find("E") != string::npos || print_string_.find("e") != string::npos)
469  logger.enable(Logger::ERR);
470  else
471  logger.disable(Logger::ERR);
472  if(print_string_.find("W") != string::npos || print_string_.find("w") != string::npos)
473  logger.enable(Logger::WRN);
474  else
475  logger.disable(Logger::WRN);
476  if(print_string_.find("R") != string::npos || print_string_.find("r") != string::npos)
477  logger.enable(Logger::RES);
478  else
479  logger.disable(Logger::RES);
480  if(print_string_.find("I") != string::npos || print_string_.find("i") != string::npos)
481  logger.enable(Logger::INF);
482  else
483  logger.disable(Logger::INF);
484  if(print_string_.find("D") != string::npos || print_string_.find("d") != string::npos)
485  logger.enable(Logger::DBG);
486  else
487  logger.disable(Logger::DBG);
488  if(print_string_.find("L") != string::npos || print_string_.find("l") != string::npos)
489  logger.enable(Logger::LOG);
490  else
491  logger.disable(Logger::LOG);
492 }
