96 {

104 // -------------------------------------------------------------------------------------------
105 void QBFCertImplExtractor::extractCircuit(const CNF &winning_region,
106  const CNF &neg_winning_region)
107 {
108  // We would like to find skolem functions for the c signals in:
109  // forall x,i: exists c,x': (NOT W(x)) OR ( T(x,i,c,x') AND W(x')),
110  // where W(x) denotes the winning region.
111 
112  // Instead, we compute Herbrand functions for the c signals in:
113  // exists x,i: forall c: exists x': W(x) AND T(x,i,c,x') AND NOT W(x')),
114  // where W(x) denotes the winning region.
115 
116  CNF strategy(neg_winning_region);
117  strategy.swapPresentToNext();
118  strategy.addCNF(AIG2CNF::instance().getTrans());
119  strategy.addCNF(winning_region);
120 
121  // build the quantifier prefix:
122  vector<pair<VarInfo::VarKind, DepQBFExt::Quant> > quant;
123  quant.push_back(make_pair(VarInfo::PRES_STATE, DepQBFExt::E));
124  quant.push_back(make_pair(VarInfo::INPUT, DepQBFExt::E));
125  quant.push_back(make_pair(VarInfo::CTRL, DepQBFExt::A));
126  quant.push_back(make_pair(VarInfo::NEXT_STATE, DepQBFExt::E));
127  quant.push_back(make_pair(VarInfo::TMP, DepQBFExt::E));
128 
129  // now we compute functions for the control signals c:
130  DepQBFExt solver;
131 #ifndef NDEBUG
132  MASSERT(solver.isSat(quant,strategy) == false, "Error in winning region.");
133 #endif
134  string answer = solver.qbfCert(quant, strategy);
135 
136  // now, we only need to parse the returned aiger file:
137  vector<vector<int> > and_gates;
138  size_t max_idx = parseQBFCertAnswer(answer, and_gates);
139 
140  L_INF("Writing the result file ...");
141  dumpResAigerLib(max_idx, and_gates);
142 
143 }

146 size_t QBFCertImplExtractor::parseQBFCertAnswer(const string &answer,
147  vector<vector<int> > &and_gates) const
148 {
149  vector<string> lines;
150  StringUtils::splitLines(answer, lines, false);
151  MASSERT(lines.size() > 0, "QBFCert parse error.");
152  istringstream header(lines[0]);
153  size_t max_idx = 0, nr_in = 0, nr_l = 0, nr_out = 0, nr_gates = 0;
154  string aag;
155  header >> aag >> max_idx >> nr_in >> nr_l >> nr_out >> nr_gates;
156  MASSERT(lines.size() >= nr_in + nr_out + nr_gates, "QBFCert parse error.");
157  MASSERT(nr_l == 0, "QBFCert parse error.");
158 
159  // let's build up a map to rename the variable indices:
160  vector<int> rename_map(2*(max_idx + 1), -1);
161  rename_map[0] = 0;
162  rename_map[1] = 1;
163 
164  VarManager& VM = VarManager::instance();
165  size_t next_line = 1;
166  for(size_t cnt = 0; cnt < nr_in + nr_out; ++cnt)
167  {
168  istringstream nr(lines[next_line++]);
169  int old_nr = 0;
170  nr >> old_nr;
171  int new_nr = VM.getInfo(old_nr >> 1).getLitInAIG();
172  rename_map[old_nr] = new_nr;
173  rename_map[old_nr+1] = new_nr + 1;
174  }
175 
176  // now we are applying the map:
177  const vector<int> &ctrl_cnf_vars = VM.getVarsOfType(VarInfo::CTRL);
178  and_gates.reserve(nr_gates + ctrl_cnf_vars.size());
179  int next_free_aig_var = VM.getMaxAIGVar()*2 + 2;
180  L_DBG("Resulting AND gates:");
181  for(size_t cnt = 0; cnt < nr_gates; ++cnt)
182  {
183  istringstream triple(lines[next_line++]);
184  vector<int> and_gate(3,0);
185  triple >> and_gate[0] >> and_gate[1] >> and_gate[2];
186  for(size_t val_cnt = 0; val_cnt < 3; ++val_cnt)
187  {
188  int var_to_rename = and_gate[val_cnt];
189  if(rename_map[var_to_rename] == -1)
190  {
191  if((var_to_rename & 1) != 0)
192  var_to_rename -= 1;
193  rename_map[var_to_rename] = next_free_aig_var;
194  rename_map[var_to_rename+1] = next_free_aig_var+1;
195  next_free_aig_var += 2;
196  }
197  and_gate[val_cnt] = rename_map[and_gate[val_cnt]];
198  }
199  and_gates.push_back(and_gate);
200  L_DBG(and_gate[0] << " " << and_gate[1] << " " << and_gate[2]);
201  }
202 
203  // if certain control signals are not defined by AND gates, then this means
204  // that we can set them arbitrarily:
205  for(size_t cnt = 0; cnt < ctrl_cnf_vars.size(); ++cnt)
206  {
207  int aig_var = VM.getInfo(ctrl_cnf_vars[cnt]).getLitInAIG();
208  bool found = false;
209  for(size_t cnt2 = 0; cnt2 < and_gates.size(); ++cnt2)
210  {
211  if(and_gates[cnt2][0] == aig_var)
212  {
213  found = true;
214  break;
215  }
216  }
217  if(!found)
218  {
219  L_DBG("AIGER-literal " << aig_var << " could be chosen arbitrarily.");
220  vector<int> and_gate(3,0);
221  and_gate[0] = aig_var;
222  and_gates.push_back(and_gate);
223  }
224  }
225 
226  return (next_free_aig_var - 2) >> 1;
227 }

230 void QBFCertImplExtractor::dumpResAigerLib(size_t max_var_idx,
231  const vector<vector<int> > &and_gates)
232 {
233  const string &in_file = Options::instance().getAigInFileName();
234  const string &out_file = Options::instance().getAigOutFileName();
235  const char *error = NULL;
236  aiger *aig = aiger_init();
237  error = aiger_open_and_read_from_file (aig, in_file.c_str());
238  MASSERT(error == NULL, "Could not open AIGER file " << in_file << " (" << error << ").");
239 
240  // remove controllable inputs:
241  aiger_private * priv = (aiger_private*) (aig);
242  for(unsigned cnt = 0; cnt < aig->num_inputs; ++cnt)
243  {
244  string name;
245  if(aig->inputs[cnt].name != NULL)
246  name = string(aig->inputs[cnt].name);
247  StringUtils::toLowerCaseIn(name);
248  if(name.find("controllable_") == 0)
249  {
250  free(aig->inputs[cnt].name);
251  for(unsigned cnt2 = cnt; cnt2+1 < aig->num_inputs; ++cnt2)
252  aig->inputs[cnt2] = aig->inputs[cnt2+1];
253  aig->num_inputs--;
254  priv->size_inputs--;
255  cnt--;
256  }
257  }
258 
259  // add AND gates:
260  unsigned new_size = aig->num_ands + and_gates.size();
261  aig->ands = static_cast<aiger_and*>(realloc(aig->ands, new_size * sizeof(aiger_and)));
262  for(unsigned cnt = 0; cnt < and_gates.size(); ++cnt)
263  {
264  unsigned idx = aig->num_ands + cnt;
265  aig->ands[idx].lhs = and_gates[cnt][0];
266  aig->ands[idx].rhs0 = and_gates[cnt][1];
267  aig->ands[idx].rhs1 = and_gates[cnt][2];
268  }
269  aig->num_ands = new_size;
270  priv->size_ands = new_size;
271  unsigned new_max_v = aig->num_inputs + aig->num_latches + aig->num_ands;
272  MASSERT(max_var_idx == new_max_v, "Error in max var index.");
273  aig->maxvar = new_max_v;
274 
275  int success = aiger_open_and_write_to_file(aig, out_file.c_str());
276  MASSERT(success, "Could not write result file.");
277  aiger_reset(aig);
278 }
