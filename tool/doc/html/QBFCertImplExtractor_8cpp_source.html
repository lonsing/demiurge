63 void QBFCertImplExtractor::run(const CNF &winning_region, const CNF &neg_winning_region)
64 {
65  // We would like to find skolem functions for the c signals in:
66  // forall x,i: exists c,x': (NOT W(x)) OR ( T(x,i,c,x') AND W(x')),
67  // where W(x) denotes the winning region.
68 
69  // Instead, we compute Herbrand functions for the c signals in:
70  // exists x,i: forall c: exists x': W(x) AND T(x,i,c,x') AND NOT W(x')),
71  // where W(x) denotes the winning region.
72 
73  CNF strategy(neg_winning_region);
74  strategy.swapPresentToNext();
75  strategy.addCNF(AIG2CNF::instance().getTrans());
76  strategy.addCNF(winning_region);
77 
78  // build the quantifier prefix:
79  vector<pair<VarInfo::VarKind, DepQBFExt::Quant> > quant;
80  quant.push_back(make_pair(VarInfo::INPUT, DepQBFExt::E));
81  quant.push_back(make_pair(VarInfo::PRES_STATE, DepQBFExt::E));
82  quant.push_back(make_pair(VarInfo::CTRL, DepQBFExt::A));
83  quant.push_back(make_pair(VarInfo::NEXT_STATE, DepQBFExt::E));
84  quant.push_back(make_pair(VarInfo::TMP, DepQBFExt::E));
85 
86  // now we compute functions for the control signals c:
87  DepQBFExt solver;
88 #ifndef NDEBUG
89  MASSERT(solver.isSat(quant,strategy) == false, "Error in winning region.");
90 #endif
91 
92  PointInTime qbfcert_start_time = Stopwatch::start();
93  aiger* answer = solver.qbfCert(quant, strategy);
94  qbfcert_real_time_ = Stopwatch::getRealTimeSec(qbfcert_start_time);
95 
96  // The answer contains all existentially quantified variables
97  // (INPUT, PRES_STATE, NEXT_STATE, TMP) as inputs (in this order). We need to remove them.
98  // We also need to remove the error-state variable as input.
99  size_t nr_in = VarManager::instance().getVarsOfType(VarInfo::INPUT).size();
100  nr_in += VarManager::instance().getVarsOfType(VarInfo::PRES_STATE).size() - 1;
101 
102  // remove error-state from inputs:
103  size_t err_idx = VarManager::instance().getVarsOfType(VarInfo::INPUT).size();
104  unsigned error_state_in_aig = answer->inputs[err_idx].lit;
105  unsigned neg_error_state_in_aig = error_state_in_aig + 1;
106  for(; err_idx < nr_in + 1; ++err_idx)
107  answer->inputs[err_idx].lit = answer->inputs[err_idx+1].lit;
108  // remove all other inputs:
109  answer->num_inputs = nr_in;
110 
111  // make sure the error-state does not occur as input to any AND gate:
112  for(unsigned cnt = 0; cnt < answer->num_ands; ++cnt)
113  {
114  if(answer->ands[cnt].rhs0 == error_state_in_aig)
115  answer->ands[cnt].rhs0 = 0;
116  if(answer->ands[cnt].rhs1 == error_state_in_aig)
117  answer->ands[cnt].rhs1 = 0;
118  if(answer->ands[cnt].rhs0 == neg_error_state_in_aig)
119  answer->ands[cnt].rhs0 = 1;
120  if(answer->ands[cnt].rhs1 == neg_error_state_in_aig)
121  answer->ands[cnt].rhs1 = 1;
122  }
123 
124  // if ctrl-variables are not assigned by the aiger-circuit (this happens sometimes)
125  // they must be irrelevant, and are assigned to FALSE:
126  unsigned nr_ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL).size();
127  if(answer->num_outputs < nr_ctrl)
128  {
129  L_DBG("QBFCert did not assign all ctrl-signals. Setting them to 0.");
130  }
131  while(answer->num_outputs < nr_ctrl)
132  aiger_add_output(answer, 0, NULL);
133 
134  aiger_reencode(answer);
135  size_before_abc_ = answer->num_ands;
136  PointInTime abc_start_time = Stopwatch::start();
137  aiger *optimized = optimizeWithABC(answer);
138  abc_real_time_ = Stopwatch::getRealTimeSec(abc_start_time);
139  size_after_abc_ = optimized->num_ands;
140  aiger_reset(answer);
141  size_final_ = insertIntoSpec(optimized);
142  aiger_reset(optimized);
143 }
