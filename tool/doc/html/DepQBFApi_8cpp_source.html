56 bool DepQBFApi::isSat(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
57  const CNF& cnf)
58 {
59  if(use_bloqqer_)
60  {
61  initBloqqer(quantifier_prefix, cnf);
62  bloqqer_preprocess();
63  int res = bloqqer_solve();
64  //debugCheckBloqqerVerdict(res, quantifier_prefix, cnf);
65  bloqqer_release();
66  if(res == 10)
67  return true;
68  else if(res == 20)
69  return false;
70  MASSERT(false, "Strange return code from Bloqqer/DepQBF.");
71  }
72  else
73  {
74  QDPLL *solver = qdpll_create();
75  initDepQBF(solver, quantifier_prefix, cnf);
76  QDPLLResult res = qdpll_sat(solver);
77  qdpll_delete(solver);
78  if(res == QDPLL_RESULT_SAT)
79  return true;
80  if(res == QDPLL_RESULT_UNSAT)
81  return false;
82  MASSERT(false, "Strange return code from DepQBF.");
83  }
84  return false;
85 }

120 bool DepQBFApi::isSatModel(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
121  const CNF& cnf,
122  vector<int> &model)
123 {
124  if(use_bloqqer_)
125  {
126  initBloqqer(quantifier_prefix, cnf);
127  bloqqer_preprocess();
128  int res = bloqqer_solve();
129  if(res == 20)
130  {
131  debugCheckBloqqerVerdict(res, quantifier_prefix, cnf);
132  bloqqer_release();
133  return false;
134  }
135  MASSERT(res == 10, "Strange return code from Bloqqer/DepQBF.");
136  VarManager &VM = VarManager::instance();
137  size_t model_size = 0;
138  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
139  {
140  model_size += VM.getVarsOfType(quantifier_prefix[q_cnt].first).size();
141  if(q_cnt + 1 < quantifier_prefix.size())
142  if(quantifier_prefix[q_cnt].second != quantifier_prefix[q_cnt+1].second)
143  break;
144  }
145  model.clear();
146  model.reserve(model_size);
147  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
148  {
149  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[q_cnt].first);
150  for(size_t v_cnt = 0; v_cnt < vars.size(); ++v_cnt)
151  {
152  VarValue val = bloqqer_getvalue(vars[v_cnt]);
153  MASSERT(val != UNKN, "Bloqqer return an UNKN as value");
154  if(val == NEG)
155  model.push_back(-vars[v_cnt]);
156  else if(val == POS)
157  model.push_back(vars[v_cnt]);
158  }
159  if(q_cnt + 1 < quantifier_prefix.size())
160  if(quantifier_prefix[q_cnt].second != quantifier_prefix[q_cnt+1].second)
161  break;
162  }
163  //debugCheckBloqqerModel(model, quantifier_prefix, cnf);
164  bloqqer_release();
165  return true;
166  }
167  else
168  {
169  QDPLL *solver = qdpll_create();
170  initDepQBF(solver, quantifier_prefix, cnf);
171  QDPLLResult res = qdpll_sat(solver);
172  if(res == QDPLL_RESULT_UNSAT)
173  {
174  qdpll_delete(solver);
175  return false;
176  }
177  MASSERT(res != QDPLL_RESULT_UNKNOWN, "Strange return code from DepQBF.");
178  VarManager &VM = VarManager::instance();
179  size_t model_size = 0;
180  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
181  {
182  model_size += VM.getVarsOfType(quantifier_prefix[q_cnt].first).size();
183  if(q_cnt + 1 < quantifier_prefix.size())
184  if(quantifier_prefix[q_cnt].second != quantifier_prefix[q_cnt+1].second)
185  break;
186  }
187  model.clear();
188  model.reserve(model_size);
189  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
190  {
191  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[q_cnt].first);
192  for(size_t v_cnt = 0; v_cnt < vars.size(); ++v_cnt)
193  {
194  QDPLLAssignment val = qdpll_get_value(solver, vars[v_cnt]);
195  if(val == QDPLL_ASSIGNMENT_FALSE)
196  model.push_back(-vars[v_cnt]);
197  else if(val == QDPLL_ASSIGNMENT_TRUE)
198  model.push_back(vars[v_cnt]);
199  }
200  if(q_cnt + 1 < quantifier_prefix.size())
201  if(quantifier_prefix[q_cnt].second != quantifier_prefix[q_cnt+1].second)
202  break;
203  }
204  qdpll_delete(solver);
205  return true;
206  }
207 }

307 void DepQBFApi::initBloqqer(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
308  const CNF& cnf)
309 {
310  VarManager &VM = VarManager::instance();
311 
312  bloqqer_init(VM.getMaxCNFVar(), cnf.getNrOfClauses());
313  for(size_t level = 0; level < quantifier_prefix.size(); ++level)
314  {
315  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[level].first);
316  for(size_t var_nr = 0; var_nr < vars.size(); ++var_nr)
317  {
318  if(quantifier_prefix[level].second == E)
319  bloqqer_decl_var(vars[var_nr]);
320  else
321  bloqqer_decl_var(-vars[var_nr]);
322  }
323  }
324 
325  const list<vector<int> > &clauses = cnf.getClauses();
326  for(CNF::ClauseConstIter it = clauses.begin(); it != clauses.end(); ++it)
327  {
328  for(size_t lit_cnt = 0; lit_cnt < it->size(); ++lit_cnt)
329  bloqqer_add((*it)[lit_cnt]);
330  bloqqer_add(0);
331  }
332 }

336  const CNF& cnf)

371 void DepQBFApi::initDepQBF(QDPLL *solver,
372  const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
373  const CNF& cnf)
374 {
375  VarManager &VM = VarManager::instance();
376  qdpll_adjust_vars(solver, VM.getMaxCNFVar());
377  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
378  {
379  if(quantifier_prefix[q_cnt].second == E)
380  qdpll_new_scope(solver, QDPLL_QTYPE_EXISTS);
381  else
382  qdpll_new_scope(solver, QDPLL_QTYPE_FORALL);
383  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[q_cnt].first);
384  for(size_t v_cnt = 0; v_cnt < vars.size(); ++v_cnt)
385  qdpll_add(solver, vars[v_cnt]);
386  qdpll_add(solver, 0);
387  }
388  const list<vector<int> > &clauses = cnf.getClauses();
389  for(CNF::ClauseConstIter it = clauses.begin(); it != clauses.end(); ++it)
390  {
391  for(size_t lit_cnt = 0; lit_cnt < it->size(); ++lit_cnt)
392  qdpll_add(solver, (*it)[lit_cnt]);
393  qdpll_add(solver, 0);
394  }

434 void DepQBFApi::debugCheckBloqqerVerdict(int bloqqer_verdict,
435  const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
436  const CNF& cnf)
437 {
438 #ifndef NDEBUG
439  DepQBFApi checker(false);
440  bool check_sat = checker.isSat(quantifier_prefix, cnf);
441  if(((!check_sat) && (bloqqer_verdict == 10)) || (check_sat && (bloqqer_verdict == 20)))
442  {
443  L_ERR("Error in DepQBFApi when using Bloqqer.");
444  if((!check_sat) && (bloqqer_verdict == 10))
445  {
446  L_ERR("Bloqqer said SAT but it should be UNSAT.");
447  }
448  else
449  {
450  L_ERR("Bloqqer said UNSAT but it should be SAT.");
451  }
452  DepQBFExt printer;
453  printer.dumpQBF(quantifier_prefix, cnf, "tmp/incorrect.qdimacs");
454  L_ERR("QBF problem has been written to tmp/incorrect.qdimacs");
455  L_ERR("The formula after pre-processing:");
456  bloqqer_print();
457  MASSERT(false, "Error in DepQBFApi when using Bloqqer.");
458  }
459 #endif
460 }
