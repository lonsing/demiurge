69 bool RareqsApi::isSat(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
70  const CNF& cnf)
71 {
72  vector<int> model;
73  VarManager &VM = VarManager::instance();
74  vector<pair<vector<int>, RAReQSQuantifierType> > pr;
75  for(size_t cnt = 0; cnt < quantifier_prefix.size(); ++cnt)
76  {
77  vector<int> merged_vars;
78  while(true)
79  {
80  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[cnt].first);
81  merged_vars.insert(merged_vars.end(), vars.begin(), vars.end());
82  if(cnt == quantifier_prefix.size() - 1)
83  break;
84  if(quantifier_prefix[cnt].second != quantifier_prefix[cnt+1].second)
85  break;
86  ++cnt;
87  }
88  if(quantifier_prefix[cnt].second == E)
89  pr.push_back(make_pair(merged_vars, RAReQS_EXISTS));
90  else
91  pr.push_back(make_pair(merged_vars, RAReQS_FORALL));
92  }
93  return RAReQSisSatModel(pr, cnf.getClauses(), model);
94 }
