85 void CNFImplExtractor::extractCircuit(const CNF &winning_region)
86 {
87  PointInTime start_time = Stopwatch::start();
88  CNF simplified_winning_region(winning_region);
89  Utils::compressStateCNF(simplified_winning_region, true);
90  CNF neg_winning_region(simplified_winning_region);
91  neg_winning_region.negate();
92  // We can also compute a negation without introducing temporary variables, but this is
93  // expensive:
94  //Utils::negateStateCNF(neg_winning_region);
95  //Utils::compressStateCNF(neg_winning_region, false);
96  run(simplified_winning_region, neg_winning_region);
97  extraction_cpu_time_ = Stopwatch::getCPUTimeSec(start_time);
98  extraction_real_time_ = Stopwatch::getRealTimeSec(start_time);
99 }

112 aiger* CNFImplExtractor::optimizeWithABC(aiger *circuit) const
113 {
114  //string dbg_in_file = Options::instance().getUniqueTmpFileName("optimize_me") + ".aag";
115  //int succ = aiger_open_and_write_to_file(circuit, dbg_in_file.c_str());
116  string tmp_in_file = Options::instance().getUniqueTmpFileName("optimize_me") + ".aig";
117  int succ = aiger_open_and_write_to_file(circuit, tmp_in_file.c_str());
118  MASSERT(succ != 0, "Could not write out AIGER file for optimization with ABC.");
119 
120  // optimize circuit:
121  string path_to_abc = Options::instance().getTPDirName() + "/abc/abc/abc";
122  string tmp_out_file = Options::instance().getUniqueTmpFileName("optimized") + ".aig";
123  string abc_command = path_to_abc + " -c \"";
124  abc_command += "read_aiger " + tmp_in_file + ";";
125  abc_command += " strash; refactor -zl; rewrite -zl;";
126  if(circuit->num_ands < 1000000)
127  abc_command += " strash; refactor -zl; rewrite -zl;";
128  if(circuit->num_ands < 200000)
129  abc_command += " strash; refactor -zl; rewrite -zl;";
130  if(circuit->num_ands < 200000)
131  abc_command += " dfraig; rewrite -zl; dfraig;";
132  abc_command += " write_aiger -s " + tmp_out_file + "\"";
133  abc_command += " > /dev/null 2>&1";
134  int ret = system(abc_command.c_str());
135  ret = WEXITSTATUS(ret);
136  MASSERT(ret == 0, "ABC terminated with strange return code.");
137 
138  // read back the result:
139  aiger *res = aiger_init();
140  const char *err = aiger_open_and_read_from_file (res, tmp_out_file.c_str());
141  MASSERT(err == NULL, "Could not open optimized AIGER file "
142  << tmp_out_file << " (" << err << ").");
143  remove(tmp_in_file.c_str());
144  remove(tmp_out_file.c_str());
145  return res;
146 }

149 size_t CNFImplExtractor::insertIntoSpec(aiger *standalone_circuit) const
150 {
151  const vector<int> &ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL);
152  const vector<int> &in = VarManager::instance().getVarsOfType(VarInfo::INPUT);
153  const vector<int> &pres = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE);
154  MASSERT(standalone_circuit->num_inputs == pres.size() + in.size() - 1, "Wrong inputs.");
155  MASSERT(standalone_circuit->num_outputs == ctrl.size(), "Wrong outputs.");
156 
157  // 1.: build the literal-renaming map
158  int max_lit = 2*(standalone_circuit->maxvar + 1);
159  vector<int> rename_map(max_lit + 1, -1);
160  rename_map[0] = 0;
161  rename_map[1] = 1;
162 
163  // 1a.: renaming of the inputs of the stand-alone circuit:
164  vector<int> ip;
165  ip.reserve(in.size() + pres.size());
166  ip.insert(ip.end(), in.begin(), in.end());
167  ip.insert(ip.end(), pres.begin(), pres.end());
168  int in_cnt = 0;
169  for(size_t cnt = 0; cnt < ip.size(); ++cnt)
170  {
171  if(ip[cnt] != VarManager::instance().getPresErrorStateVar())
172  {
173  unsigned from_aig_lit = standalone_circuit->inputs[in_cnt].lit;
174  unsigned to_aig_lit = VarManager::instance().getInfo(ip[cnt]).getLitInAIG();
175  rename_map[from_aig_lit] = to_aig_lit;
176  rename_map[from_aig_lit+1] = to_aig_lit+1;
177  in_cnt++;
178  }
179  }
180 
181  // 1b.: renaming of the outputs of the AND gates:
182  int to_aig_lit = VarManager::instance().getMaxAIGVar() * 2 + 2;
183  for(unsigned cnt = 0; cnt < standalone_circuit->num_ands; ++cnt)
184  {
185  unsigned from_aig_lit = standalone_circuit->ands[cnt].lhs;
186  rename_map[from_aig_lit] = to_aig_lit;
187  rename_map[from_aig_lit+1] = to_aig_lit+1;
188  to_aig_lit += 2;
189  }
190  unsigned new_max_aig_var = (to_aig_lit - 2)/2;
191 
192  // 1c.: some of the outputs can be connected to the spec-signals by renaming:
193  // If a control signal (an output of the stand-alone circuit) is an unnegated output of an
194  // AND gate, then we can connect it to the spec signal simply by renaming:
195  vector<unsigned> connect_out_indices;
196  set<unsigned> already_renamed;
197  connect_out_indices.reserve(standalone_circuit->num_outputs);
198  for(unsigned cnt = 0; cnt < standalone_circuit->num_outputs; ++cnt)
199  {
200  unsigned output_lit = standalone_circuit->outputs[cnt].lit;
201  if(!aiger_sign(output_lit) &&
202  output_lit > (standalone_circuit->num_inputs*2+1) &&
203  already_renamed.count(output_lit) == 0)
204  {
205  // this output of the stand-alone circuit is unnegated and it is defined by an AND
206  // gate.
207  int to_lit = VarManager::instance().getInfo(ctrl[cnt]).getLitInAIG();
208  rename_map[output_lit] = to_lit;
209  rename_map[output_lit+1] = to_lit+1;
210  already_renamed.insert(output_lit);
211  }
212  else
213  connect_out_indices.push_back(cnt);
214  }
215 
216  // 2.: Insert the stand-alone circuit into the spec:
217  const string &in_file = Options::instance().getAigInFileName();
218  const string &out_file = Options::instance().getAigOutFileName();
219  const char *error = NULL;
220  aiger *aig = aiger_init();
221  error = aiger_open_and_read_from_file (aig, in_file.c_str());
222  MASSERT(error == NULL, "Could not open AIGER file " << in_file << " (" << error << ").");
223 
224  // 2.a.: remove controllable inputs:
225  aiger_private * priv = (aiger_private*) (aig);
226  for(unsigned cnt = 0; cnt < aig->num_inputs; ++cnt)
227  {
228  string name;
229  if(aig->inputs[cnt].name != NULL)
230  name = string(aig->inputs[cnt].name);
231  StringUtils::toLowerCaseIn(name);
232  if(name.find("controllable_") == 0)
233  {
234  free(aig->inputs[cnt].name);
235  for(unsigned cnt2 = cnt; cnt2+1 < aig->num_inputs; ++cnt2)
236  aig->inputs[cnt2] = aig->inputs[cnt2+1];
237  aig->num_inputs--;
238  priv->size_inputs--;
239  cnt--;
240  }
241  }
242 
243  // 2.b.: add AND gates from standalone_circuit:
244  unsigned new_size = aig->num_ands + standalone_circuit->num_ands + connect_out_indices.size();
245  aig->ands = static_cast<aiger_and*>(realloc(aig->ands, new_size * sizeof(aiger_and)));
246  unsigned next_and_idx = aig->num_ands;
247  for(unsigned cnt = 0; cnt < standalone_circuit->num_ands; ++cnt)
248  {
249  aig->ands[next_and_idx].lhs = rename_map[standalone_circuit->ands[cnt].lhs];
250  aig->ands[next_and_idx].rhs0 = rename_map[standalone_circuit->ands[cnt].rhs0];
251  aig->ands[next_and_idx].rhs1 = rename_map[standalone_circuit->ands[cnt].rhs1];
252  next_and_idx++;
253  }
254 
255  // 2.c.: add additional AND gates to connect the output of the standalone circuit
256  // to the controllable inputs:
257  for(size_t cnt = 0; cnt < connect_out_indices.size(); ++cnt)
258  {
259  unsigned output_lit = standalone_circuit->outputs[connect_out_indices[cnt]].lit;
260  // this output of the stand-alone circuit is negated of not defined by an AND
261  // gate. Hence, we need to connect it with an additional AND gate.
262  int to_lit = VarManager::instance().getInfo(ctrl[connect_out_indices[cnt]]).getLitInAIG();
263  aig->ands[next_and_idx].lhs = to_lit;
264  aig->ands[next_and_idx].rhs0 = rename_map[output_lit];
265  aig->ands[next_and_idx].rhs1 = 1;
266  next_and_idx++;
267  }
268 
269  // 3.: Update the size of the regions:
270  aig->num_ands = new_size;
271  priv->size_ands = new_size;
272  aig->maxvar = new_max_aig_var;
273 
274  // 4.: Write out the result:
275  if(out_file == "stdout")
276  {
277  int success = aiger_write_to_file(aig, aiger_ascii_mode, stdout);
278  MASSERT(success, "Could not write result file.");
279  }
280  else
281  {
282  int success = aiger_open_and_write_to_file(aig, out_file.c_str());
283  MASSERT(success, "Could not write result file.");
284  }
285  aiger_reset(aig);
286  return standalone_circuit->num_ands + connect_out_indices.size();
287 }
