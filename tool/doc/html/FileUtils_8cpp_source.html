42 bool FileUtils::readFile(const string &file_name, string &file_content)
43 {
44  ifstream is(file_name.c_str());
45  if(is.fail())
46  return false;
47  is.seekg(0, std::ios::end);
48  if(is.fail() || is.bad())
49  {
50  is.close();
51  return false;
52  }
53  size_t length = is.tellg();
54  is.seekg(0, std::ios::beg);
55  if(is.fail() || is.bad())
56  {
57  is.close();
58  return false;
59  }
60  char *buffer = new char[length];
61  MASSERT(buffer != NULL, "Out of memory.");
62  is.read(buffer, length);
63  is.close();
64  file_content.append(buffer, length);
65  delete[] buffer;
66  buffer = NULL;
67  return true;
68 }
