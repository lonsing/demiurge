206 bool LearnSynthSAT::computeWinningRegionPlain()
207 {
208  AIG2CNF& A2C = AIG2CNF::instance();
209 
210  winning_region_.clear();
211  winning_region_ = A2C.getSafeStates();
212  winning_region_large_.clear();
213  winning_region_large_ = A2C.getSafeStates();
214 
215  // solver_i_ contains F & T & !F':
216  solver_i_->doMinCores(true);
217  solver_i_->doRandModels(false);
218  solver_i_->startIncrementalSession(sic_, false);
219  solver_i_->incAddCNF(A2C.getNextUnsafeStates());
220  solver_i_->incAddCNF(A2C.getTrans());
221  solver_i_->incAddCNF(A2C.getSafeStates());
222 
223  // solver_ctrl_ contains F & T & F':
224  solver_ctrl_->doMinCores(true);
225  solver_ctrl_->doRandModels(false);
226  solver_ctrl_->startIncrementalSession(sicn_, false);
227  solver_ctrl_->incAddCNF(A2C.getNextSafeStates());
228  solver_ctrl_->incAddCNF(A2C.getTrans());
229  solver_ctrl_->incAddCNF(A2C.getSafeStates());
230  bool precise = true;
231 
232  size_t it_cnt = 0;
233  size_t clauses_added = 0;
234  size_t reset_c_cnt = 0;
235  vector<int> model_or_core;
236  while(true)
237  {
238  ++it_cnt;
239 
240  // First, we check if there is some transition with some values for the inputs i and
241  // and c from F to !G' (G is a copy of F that is updated only lazily).
242  statistics_.notifyBeforeComputeCandidate();
243  bool sat = solver_i_->incIsSatModelOrCore(vector<int>(), si_, model_or_core);
244  statistics_.notifyAfterComputeCandidate();
245 
246  if(!sat)
247  {
248  // No such transition exists.
249  if(precise)
250  {
251  // G = F. This means that there exists no possibility for the antagonist to leave F.
252  // Hence, F is a valid winning region. We can stop.
253  return true;
254  }
255  // G != F. Hence, the reason why no such transition exists could be that G is not up-to
256  // data. We set G:=F and start again. For that, we have to restart the incremental
257  // of session solver_i_.
258  statistics_.notifyRestart();
259  L_DBG("Need to restart with fresh U (iteration " << it_cnt << ")");
260  // get rid of all temporary variables introduced during negations:
261  VarManager::instance().resetToLastPush();
262  solver_i_->startIncrementalSession(sic_, false);
263  Utils::compressStateCNF(winning_region_);
264  CNF leave_win(winning_region_);
265  leave_win.swapPresentToNext();
266  leave_win.negate();
267  leave_win.addCNF(AIG2CNF::instance().getTrans());
268  leave_win.addCNF(winning_region_);
269  //leave_win.addCNF(winning_region_large_);
270  solver_i_->incAddCNF(leave_win);
271  precise = true;
272  continue;
273  }
274 
275  vector<int> state_input = model_or_core;
276  vector<int> state = Utils::extract(state_input, VarInfo::PRES_STATE);
277  vector<int> input = Utils::extract(state_input, VarInfo::INPUT);
278 
279  // There exists a transition from F to !G' with some input i. Let's now see if the
280  // protagonist can find some c such that leaving F is avoided:
281  statistics_.notifyBeforeCheckCandidate();
282  sat = solver_ctrl_->incIsSatModelOrCore(state, input, c_, model_or_core);
283  if(!sat)
284  {
285  // No such response exists. This means that we have found a counterexample-state.
286  // The solver solver_ctrl_ already computed a generalization of this
287  // counterexample-state in form of an unsatisfiable core. The unsatisfiable core
288  // represents the set of all state for which input i enforces that F is left.
289  // (This is weaker than what we compute with the QBF-solver in LearnSynthQBF, but
290  // it appears to be good enough in our experiments, and the computation is fast.)
291 
292  // We can try to reduce it further (does not pay of):
293  //vector<int> first_core(model_or_core);
294  //solver_ctrl_->incAddNegCubeAsClause(model_or_core);
295  //Utils::swapPresentToNext(model_or_core);
296  //solver_ctrl_->incAddNegCubeAsClause(model_or_core);
297  //sat = solver_ctrl_->incIsSatModelOrCore(first_core, input, c, model_or_core);
298 
299  if(Utils::containsInit(model_or_core))
300  return false;
301  // We compute the corresponding blocking clause, and update the winning region and the
302  // solvers:
303  vector<int> blocking_clause(model_or_core);
304  Utils::negateLiterals(blocking_clause);
305  winning_region_.addClauseAndSimplify(blocking_clause);
306  //winning_region_large_.addClauseAndSimplify(blocking_clause);
307  solver_i_->incAddClause(blocking_clause);
308 
309  // reseting the solver_ctrl_ seems to help.
310  // TODO: find out good numbers for the heuristics when to restart
311  if(clauses_added > winning_region_.getNrOfClauses() + 100)
312  {
313  if(reset_c_cnt % 1000 == 999)
314  Utils::compressStateCNF(winning_region_, true);
315  else if(reset_c_cnt % 100 == 99)
316  Utils::compressStateCNF(winning_region_, false);
317  solver_ctrl_->startIncrementalSession(sicn_, false);
318  CNF next_win(winning_region_);
319  next_win.swapPresentToNext();
320  solver_ctrl_->incAddCNF(winning_region_);
321  solver_ctrl_->incAddCNF(A2C.getTrans());
322  solver_ctrl_->incAddCNF(next_win);
323  clauses_added = winning_region_.getNrOfClauses();
324  reset_c_cnt++;
325  }
326  else
327  {
328  solver_ctrl_->incAddClause(blocking_clause);
329  Utils::swapPresentToNext(blocking_clause);
330  solver_ctrl_->incAddClause(blocking_clause);
331  clauses_added++;
332  }
333  precise = false;
334  statistics_.notifyAfterCheckCandidateFound(s_.size(), blocking_clause.size());
335  }
336  else
337  {
338  // There exists a response of the protagonist to this input vector. Hence, this input
339  // vector is useless for the antagonist in trying to go from F to !G'. We need to
340  // exclude it from solver_i_ so that the same state-input pair is not tried again.
341  // However, instead of excluding this one state-input pair only, we generalize it using
342  // an unsatisfiable core. The core gives us all state-input combinations for which
343  // the control vector found by solver_ctrl_ prevents that we go from F to !G'.
344  statistics_.notifyAfterCheckCandidateFailed();
345  vector<int> ctrl(model_or_core);
346  statistics_.notifyBeforeRefine();
347  sat = solver_i_->incIsSatModelOrCore(state_input, ctrl, vector<int>(), model_or_core);
348  MASSERT(sat == false, "Impossible.");
349  solver_i_->incAddNegCubeAsClause(model_or_core);
350  statistics_.notifyAfterRefine(si_.size(), model_or_core.size());
351  }
352  }
