92 void VarManager::initFromAig(aiger *aig, const vector<bool>& refs)
93 {
94  clear();
95  // build map from AIG literals to CNF literals:
96  aig_to_cnf_lit_map_.resize(2*(aig->maxvar+1), 0);
97  max_cnf_var_ = 0;
98  if (refs[0] || refs[1])
99  {
100  aig_to_cnf_lit_map_[0] = -1;
101  aig_to_cnf_lit_map_[1] = 1;
102  max_cnf_var_++;
103  VarInfo info(VarInfo::TMP, 1, 1, "TRUE");
104  vars_.insert(pair<int,VarInfo>(1, info));
105  }
106  for (unsigned lit = 2; lit <= 2*aig->maxvar; lit += 2)
107  {
108  if (refs[lit] || refs[lit+1])
109  {
110  aig_to_cnf_lit_map_[lit] = ++max_cnf_var_;
111  aig_to_cnf_lit_map_[lit+1] = -max_cnf_var_;
112  VarInfo info(VarInfo::TMP, max_cnf_var_, lit);
113  vars_.insert(pair<int,VarInfo>(max_cnf_var_, info));
114  }
115  }
116  max_aig_var_ = aig->maxvar;
117 
118  // set type and name properly:
119  inputs_.reserve(aig->num_inputs);
120  for(unsigned cnt = 0; cnt < aig->num_inputs; ++cnt)
121  {
122  unsigned lit = aig->inputs[cnt].lit;
123  if (refs[lit] || refs[lit+1])
124  {
125  int cnf_in_var = aig_to_cnf_lit_map_[lit];
126  MASSERT(vars_.find(cnf_in_var) != vars_.end(), "Var not found");
127  string aig_name;
128  ostringstream name;
129  if(aig->inputs[cnt].name != NULL)
130  aig_name = aig->inputs[cnt].name;
131  string aig_name_lower = StringUtils::toLowerCase(aig_name);
132  if(aig_name_lower.find("controllable_") == 0)
133  {
134  name << "c_" << cnt << "(" << aig_name << ")";
135  vars_.find(cnf_in_var)->second.setName(name.str());
136  vars_.find(cnf_in_var)->second.setKind(VarInfo::CTRL);
137  ctrl_vars_.push_back(cnf_in_var);
138  }
139  else
140  {
141  name << "i_" << cnt << "(" << aig_name << ")";
142  vars_.find(cnf_in_var)->second.setName(name.str());
143  vars_.find(cnf_in_var)->second.setKind(VarInfo::INPUT);
144  inputs_.push_back(cnf_in_var);
145  }
146  }
147  }
148  unsigned output_var = aig->outputs[0].lit;
149  output_var &= ~1;
150  if((aig->outputs[0].lit & 1) == 0)
151  vars_.find(aig_to_cnf_lit_map_[output_var])->second.setName("error_output");
152  else
153  vars_.find(aig_to_cnf_lit_map_[output_var])->second.setName("neg_error_output");
154  pres_state_vars_.reserve(aig->num_latches+1);
155  next_state_vars_.reserve(aig->num_latches+1);
156  // we latch the error_output to make it part of our state space:
157  max_cnf_var_++;
158  VarInfo pe_info(VarInfo::PRES_STATE, max_cnf_var_, 0, "pres_error");
159  vars_.insert(pair<int,VarInfo>(max_cnf_var_, pe_info));
160  pres_state_vars_.push_back(max_cnf_var_);
161  max_cnf_var_++;
162  VarInfo ne_info(VarInfo::NEXT_STATE, max_cnf_var_, 0, "next_error");
163  vars_.insert(pair<int,VarInfo>(max_cnf_var_, ne_info));
164  next_state_vars_.push_back(max_cnf_var_);
165 
166 
167  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
168  {
169  // present state:
170  unsigned lit = aig->latches[cnt].lit;
171  int cnf_state_var = aig_to_cnf_lit_map_[lit];
172  vars_.find(cnf_state_var)->second.setKind(VarInfo::PRES_STATE);
173  ostringstream name;
174  name << "x_" << cnt;
175  if(aig->latches[cnt].name != NULL)
176  name << "(" << string(aig->latches[cnt].name) << ")";
177  vars_.find(cnf_state_var)->second.setName(name.str());
178  pres_state_vars_.push_back(cnf_state_var);
179 
180  // next state:
181  // we create a new variable to have a new name for sure:
182  lit = aig->latches[cnt].next;
183  ostringstream next_name;
184  next_name << "x'_" << cnt;
185  if(aig->latches[cnt].name != NULL)
186  next_name << "(" << string(aig->latches[cnt].name) << ")";
187  max_cnf_var_++;
188  VarInfo ns_info(VarInfo::NEXT_STATE, max_cnf_var_, lit, next_name.str());
189  vars_.insert(pair<int,VarInfo>(max_cnf_var_, ns_info));
190  next_state_vars_.push_back(max_cnf_var_);
191  }
192 
193  // all remaining variables must be temporary variables.
194  tmp_vars_.reserve(1000000);
195  for(VarsConstIter it = vars_.begin(); it != vars_.end(); ++it)
196  {
197  if(it->second.getKind() == VarInfo::TMP)
198  tmp_vars_.push_back(it->first);
199  }
200 }

281 const vector<int>& VarManager::getVarsOfType(VarInfo::VarKind var_kind) const
282 {
283  if(var_kind == VarInfo::INPUT)
284  return inputs_;
285  else if(var_kind == VarInfo::CTRL)
286  return ctrl_vars_;
287  else if(var_kind == VarInfo::PRES_STATE)
288  return pres_state_vars_;
289  else if(var_kind == VarInfo::NEXT_STATE)
290  return next_state_vars_;
291  else if(var_kind == VarInfo::TEMPL_PARAMS)
292  return templs_params_;
293  else if(var_kind == VarInfo::PREV)
294  return prev_vars_;
295  else //(var_kind == VarInfo::TMP)
296  return tmp_vars_;
297 }

300 vector<int> VarManager::getAllNonTempVars() const
301 {
302  vector<int> res;
303  size_t size = inputs_.size();
304  size += ctrl_vars_.size();
305  size += pres_state_vars_.size();
306  size += next_state_vars_.size();
307  size += templs_params_.size();
308  size += prev_vars_.size();
309  res.reserve(size);
310  res.insert(res.end(), inputs_.begin(), inputs_.end());
311  res.insert(res.end(), ctrl_vars_.begin(), ctrl_vars_.end());
312  res.insert(res.end(), pres_state_vars_.begin(), pres_state_vars_.end());
313  res.insert(res.end(), next_state_vars_.begin(), next_state_vars_.end());
314  res.insert(res.end(), templs_params_.begin(), templs_params_.end());
315  res.insert(res.end(), prev_vars_.begin(), prev_vars_.end());
316  return res;
317 }

344 string VarManager::toString()
345 {
346  ostringstream oss;
347  map<VarInfo::VarKind, string> var_kind_names;
348  var_kind_names[VarInfo::PRES_STATE] = "PS";
349  var_kind_names[VarInfo::NEXT_STATE] = "NS";
350  var_kind_names[VarInfo::INPUT] = "IN";
351  var_kind_names[VarInfo::CTRL] = "CTRL";
352  var_kind_names[VarInfo::TMP] = "TMP";
353  var_kind_names[VarInfo::TEMPL_PARAMS] = "PARAM";
354  var_kind_names[VarInfo::PREV] = "PREV";
355  oss << "Variables:" << endl;
356  oss << "==========" << endl;
357  for(VarsConstIter it = vars_.begin(); it != vars_.end(); ++it)
358  {
359  const VarInfo &info = it->second;
360  oss << "- CNF: "<< info.getLitInCNF();
361  oss << ", Kind:" << var_kind_names[info.getKind()];
362  oss << ", AIG:" << info.getLitInAIG();
363  oss << ", name:" << info.getName() << endl;
364  }
365  return oss.str();
366 }
