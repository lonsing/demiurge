245 bool ParallelLearner::run()
246 {
247  VarManager::instance().push();
248  L_INF("Starting to compute a winning region using " << nr_of_threads_ << " threads ...");
249  statistics_.notifyWinRegStart();
250  winning_region_.clear();
251  winning_region_.addCNF(AIG2CNF::instance().getSafeStates());
252 
253  // Start the threads:
254  vector<thread> explorer_threads;
255  explorer_threads.reserve(clause_explorers_.size() - 1);
256  for(size_t cnt = 1; cnt < clause_explorers_.size(); ++cnt)
257  explorer_threads.push_back(thread(&ClauseExplorerSAT::exploreClauses,
258  clause_explorers_[cnt]));
259 
260  vector<thread> gen_threads;
261  gen_threads.reserve(ce_generalizers_.size());
262  for(size_t cnt = 0; cnt < ce_generalizers_.size(); ++cnt)
263  gen_threads.push_back(thread(&CounterGenSAT::generalizeCounterexamples,
264  ce_generalizers_[cnt]));
265 
266  vector<thread> minimizer_threads;
267  minimizer_threads.reserve(clause_minimizers_.size());
268  for(size_t cnt = 0; cnt < clause_minimizers_.size(); ++cnt)
269  minimizer_threads.push_back(thread(&ClauseMinimizerQBF::minimizeClauses,
270  clause_minimizers_[cnt]));
271 
272  vector<thread> ifm_threads;
273  ifm_threads.reserve(ifm_explorers_.size());
274  for(size_t cnt = 0; cnt < ifm_explorers_.size(); ++cnt)
275  ifm_threads.push_back(thread(&IFM13Explorer::exploreClauses,
276  ifm_explorers_[cnt]));
277 
278  //The main thread executes the first explorer:
279  MASSERT(clause_explorers_.size() > 0, "There must be at least one explorer thread");
280  clause_explorers_[0]->exploreClauses();
281 
282  // Wait until the threads are finished:
283  for(size_t cnt = 0; cnt < explorer_threads.size(); ++cnt)
284  explorer_threads[cnt].join();
285  for(size_t cnt = 0; cnt < gen_threads.size(); ++cnt)
286  gen_threads[cnt].join();
287  for(size_t cnt = 0; cnt < minimizer_threads.size(); ++cnt)
288  minimizer_threads[cnt].join();
289  for(size_t cnt = 0; cnt < ifm_threads.size(); ++cnt)
290  ifm_threads[cnt].join();
291 
292  // Merge statistics:
293  statistics_.notifyWinRegEnd();
294  for(size_t cnt = 0; cnt < clause_explorers_.size(); ++cnt)
295  statistics_.mergeWith(clause_explorers_[cnt]->getStatistics());
296  for(size_t cnt = 0; cnt < ce_generalizers_.size(); ++cnt)
297  statistics_.mergeWith(ce_generalizers_[cnt]->getStatistics());
298 
299  // Extract a circuit:
300  if(result_ == UNREALIZABLE)
301  {
302  L_RES("The specification is unrealizable.");
303  statistics_.logStatistics();
304  return false;
305  }
306  L_RES("The specification is realizable.");
307  Utils::debugCheckWinReg(winning_region_);
308  if(Options::instance().doRealizabilityOnly())
309  {
310  statistics_.logStatistics();
311  return true;
312  }
313 
314  L_INF("Starting to extract a circuit ...");
315  statistics_.notifyRelDetStart();
316  QBFCertImplExtractor extractor;
317  extractor.extractCircuit(winning_region_);
318  statistics_.notifyRelDetEnd();
319  L_INF("Synthesis done.");
320  statistics_.logStatistics();
321  return true;
322 
323 }

405 void ParallelLearner::computePreviousTrans()
406 {
407  // build previous-state copy of the transition relation:
408  AIG2CNF& A2C = AIG2CNF::instance();
409  VarManager &VM = VarManager::instance();
410  current_to_previous_map_.resize(VM.getMaxCNFVar()+1, 0);
411  const vector<int> &s = VM.getVarsOfType(VarInfo::PRES_STATE);
412  const vector<int> &s_next = VM.getVarsOfType(VarInfo::NEXT_STATE);
413  const vector<int> &c = VM.getVarsOfType(VarInfo::CTRL);
414  const vector<int> &i = VM.getVarsOfType(VarInfo::INPUT);
415  vector<int> t = VM.getVarsOfType(VarInfo::TMP);
416 
417  for(size_t v_cnt = 0; v_cnt < s.size(); ++v_cnt)
418  current_to_previous_map_[s[v_cnt]] = VM.createFreshPrevVar();
419  for(size_t v_cnt = 0; v_cnt < c.size(); ++v_cnt)
420  current_to_previous_map_[c[v_cnt]] = VM.createFreshPrevVar();
421  for(size_t v_cnt = 0; v_cnt < i.size(); ++v_cnt)
422  current_to_previous_map_[i[v_cnt]] = VM.createFreshPrevVar();
423  for(size_t v_cnt = 0; v_cnt < t.size(); ++v_cnt)
424  current_to_previous_map_[t[v_cnt]] = VM.createFreshTmpVar();
425  for(size_t v_cnt = 0; v_cnt < s_next.size(); ++v_cnt)
426  current_to_previous_map_[s_next[v_cnt]] = s[v_cnt];
427 
428  current_state_is_initial_ = VM.createFreshTmpVar();
429  list<vector<int> > prev_trans_clauses = A2C.getTrans().getClauses();
430  for(CNF::ClauseIter it = prev_trans_clauses.begin(); it != prev_trans_clauses.end(); ++it)
431  {
432  presentToPrevious(*it);
433  it->push_back(current_state_is_initial_);
434  }
435  prev_trans_or_initial_.swapWith(prev_trans_clauses);
436  // if one of the state variables is true, then current_state_is_initial must be false:
437  for(size_t cnt = 0; cnt < s.size(); ++cnt)
438  prev_trans_or_initial_.add2LitClause(-s[cnt], -current_state_is_initial_);
439 }

578 void ClauseExplorerSAT::exploreClauses()
579 {
580  const AIG2CNF& A2C = AIG2CNF::instance();
581 
582  solver_i_->startIncrementalSession(vars_to_keep_, false);
583  solver_i_->incAddCNF(A2C.getNextUnsafeStates());
584  solver_i_->incAddCNF(A2C.getTrans());
585  solver_i_->incAddCNF(A2C.getSafeStates());
586 
587  solver_ctrl_->startIncrementalSession(vars_to_keep_, false);
588  solver_ctrl_->incAddCNF(A2C.getNextSafeStates());
589  solver_ctrl_->incAddCNF(A2C.getTrans());
590  solver_ctrl_->incAddCNF(A2C.getSafeStates());
591 
592  if(use_ind_)
593  {
594  solver_ctrl_ind_->startIncrementalSession(vars_to_keep_, true);
595  solver_ctrl_ind_->incAddCNF(A2C.getNextSafeStates());
596  solver_ctrl_ind_->incAddCNF(A2C.getTrans());
597  solver_ctrl_ind_->incAddCNF(A2C.getSafeStates());
598  solver_ctrl_ind_->incAddCNF(prev_trans_or_initial_);
599  solver_ctrl_ind_->incAddUnitClause(prev_safe_);
600  }
601 
602  precise_ = true;
603 
604  const vector<int> &s = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE);
605  const vector<int> &c = VarManager::instance().getVarsOfType(VarInfo::CTRL);
606  const vector<int> &i = VarManager::instance().getVarsOfType(VarInfo::INPUT);
607  vector<int> si;
608  si.reserve(s.size() + i.size());
609  si.insert(si.end(), s.begin(), s.end());
610  si.insert(si.end(), i.begin(), i.end());
611  size_t it_cnt = 0;
612  vector<int> model_or_core;
613  while(true)
614  {
615  ++it_cnt;
616  if(coordinator_.result_ != UNKNOWN) // should be atomic
617  return;
618  considerNewInfoFromOthers();
619  statistics_.notifyBeforeComputeCandidate();
620  bool sat = solver_i_->incIsSatModelOrCore(vector<int>(), si, model_or_core);
621  statistics_.notifyAfterComputeCandidate();
622  if(!sat)
623  {
624  if(precise_)
625  {
626  coordinator_.result_ = REALIZABLE; // should be atomic
627  return;
628  }
629  if(coordinator_.result_ != UNKNOWN) // should be atomic
630  return;
631  L_DBG("Explorer " << instance_nr_ << " restarts (iteration " << it_cnt << ").");
632 
633  // minor performance optimization:
634  // if a restart is ongoing, we do not trigger another one but wait here until it is
635  // done. This is not a correctness issue: doing multiple restarts triggered by
636  // several threads is also safe (but a waste of resources).
637  bool restart_available = waitUntilOngoingRestartDone();
638  if(restart_available == false)
639  coordinator_.triggerExplorerRestart();
640  continue;
641  }
642 
643  vector<int> state_input = model_or_core;
644  vector<int> state = Utils::extract(state_input, VarInfo::PRES_STATE);
645  vector<int> input = Utils::extract(state_input, VarInfo::INPUT);
646 
647  if(coordinator_.result_ != UNKNOWN) // should be atomic
648  return;
649  statistics_.notifyBeforeCheckCandidate();
650  Utils::randomize(state);
651  sat = solver_ctrl_->incIsSatModelOrCore(state, input, c, model_or_core);
652  if(!sat)
653  {
654  if(use_ind_)
655  {
656  // we now try to minimize the core further using reachability information:
657  vector<int> orig_core(model_or_core);
658  for(size_t lit_cnt = 0; lit_cnt < orig_core.size(); ++lit_cnt)
659  {
660  vector<int> tmp(model_or_core);
661  Utils::remove(tmp, orig_core[lit_cnt]);
662  solver_ctrl_ind_->incPush();
663 
664  // use what we already know immediately:
665  vector<int> prev_core(model_or_core);
666  presentToPrevious(prev_core);
667  vector<int> next_core(model_or_core);
668  Utils::swapPresentToNext(next_core);
669  solver_ctrl_ind_->incAddNegCubeAsClause(model_or_core);
670  solver_ctrl_ind_->incAddNegCubeAsClause(prev_core);
671  solver_ctrl_ind_->incAddNegCubeAsClause(next_core);
672 
673  solver_ctrl_ind_->incAddCube(input);
674  solver_ctrl_ind_->incAddCube(tmp);
675  vector<int> prev_tmp(tmp);
676  presentToPrevious(prev_tmp);
677  prev_tmp.push_back(-current_state_is_initial_);
678  solver_ctrl_ind_->incAddNegCubeAsClause(prev_tmp);
679  sat = solver_ctrl_ind_->incIsSat();
680  solver_ctrl_ind_->incPop();
681  if(!sat)
682  model_or_core = tmp;
683  }
684  }
685 
686  if(Utils::containsInit(model_or_core))
687  {
688  coordinator_.result_ = UNREALIZABLE; // should be atomic
689  return;
690  }
691 
692  vector<int> blocking_clause(model_or_core);
693  for(size_t cnt = 0; cnt < blocking_clause.size(); ++cnt)
694  blocking_clause[cnt] = -blocking_clause[cnt];
695  statistics_.notifyAfterCheckCandidateFound(s.size(), blocking_clause.size());
696  coordinator_.notifyNewWinRegClause(blocking_clause, EXPL);
697  coordinator_.notifyNewCounterexample(state_input, model_or_core);
698  precise_ = false;
699  }
700  else
701  {
702  statistics_.notifyAfterCheckCandidateFailed();
703  vector<int> ctrl_cube = model_or_core;
704  if(coordinator_.result_ != UNKNOWN) // should be atomic
705  return;
706  statistics_.notifyBeforeRefine();
707  sat = solver_i_->incIsSatModelOrCore(state_input, ctrl_cube, vector<int>(), model_or_core);
708  MASSERT(sat == false, "Impossible " << instance_nr_);
709  for(size_t cnt = 0; cnt < model_or_core.size(); ++cnt)
710  model_or_core[cnt] = -model_or_core[cnt];
711  coordinator_.notifyNewUselessInputClause(model_or_core, restart_level_);
712  statistics_.notifyAfterRefine(si.size(), model_or_core.size());
713  }
714  }

920 {

922

924  solver_ctrl_->incAddCNF(A2C.getNextSafeStates());
925  solver_ctrl_->incAddCNF(A2C.getTrans());
926  solver_ctrl_->incAddCNF(A2C.getSafeStates());
927

929  solver_win_->startIncrementalSession(ps_vars, false);
930  solver_win_->incAddCNF(A2C.getSafeStates());
931 
932  if(use_ind_)
933  {
934  solver_ctrl_ind_->startIncrementalSession(vars_to_keep_, true);
935  solver_ctrl_ind_->incAddCNF(A2C.getNextSafeStates());
936  solver_ctrl_ind_->incAddCNF(A2C.getTrans());
937  solver_ctrl_ind_->incAddCNF(A2C.getSafeStates());
938  solver_ctrl_ind_->incAddCNF(prev_trans_or_initial_);
939  solver_ctrl_ind_->incAddUnitClause(prev_safe_);
940  }
941 
942  pair<vector<int>, vector<int> > task;
943  while(true)
944  {
945  if(coordinator_.result_ != UNKNOWN) // should be atomic
946  return;
947  coordinator_.counterexamples_lock_.lock();
948  bool empty = coordinator_.counterexamples_.empty();
949  if(!empty)
950  {
951  task = coordinator_.counterexamples_.front();
952  coordinator_.counterexamples_.pop_front();
953  }
954  coordinator_.counterexamples_lock_.unlock();
955  if(empty)
956  {
957  bored();
958  continue;
959  }
960 
961  vector<int> in = Utils::extract(task.first, VarInfo::INPUT);
962  vector<int> orig_ce_state = Utils::extract(task.first, VarInfo::PRES_STATE);
963  vector<int> first_gen = task.second;
964  size_t orig_gen_size = first_gen.size();
965  generalizeCeFuther(first_gen, in);
966  if(first_gen.size() < orig_gen_size)
967  {
968  if(solver_win_->incIsSat(first_gen))
969  {
970  vector<int> blocking_clause(first_gen);
971  Utils::negateLiterals(blocking_clause);
972  coordinator_.notifyNewWinRegClause(blocking_clause, CE_GEN);
973  }
974  }
975 
976  do_if_bored_.push_back(make_pair(first_gen, in));
977 
978  list<set<int> > must_not_contain_queue;
979  list<vector<int> > gen_ces;
980  gen_ces.push_back(first_gen);
981  for(size_t cnt = 0; cnt < first_gen.size(); ++cnt)
982  {
983  set<int> singleton_set;
984  singleton_set.insert(first_gen[cnt]);
985  must_not_contain_queue.push_back(singleton_set);
986  }
987 
988  while(!must_not_contain_queue.empty())
989  {
990  set<int> must_not_contain = must_not_contain_queue.front();
991  must_not_contain_queue.pop_front();
992 
993  // check if we already have a ce in gen_ces which satisfies our needs:
994  bool found = false;
995  for(list<vector<int> >::const_iterator it = gen_ces.begin(); it != gen_ces.end(); ++it)
996  {
997  if(Utils::intersectionEmpty(*it, must_not_contain))
998  {
999  found = true;
1000  break;
1001  }
1002  }
1003  if(found)
1004  continue;
1005 
1006  // if not: we compute one
1007  vector<int> new_ce(orig_ce_state);
1008  for(set<int>::const_iterator it = must_not_contain.begin(); it != must_not_contain.end(); ++it)
1009  Utils::remove(new_ce, *it);
1010  size_t orig_size = new_ce.size();
1011  if(coordinator_.result_ != UNKNOWN) // should be atomic
1012  return;
1013  statistics_.notifyBeforeCheckCandidate();
1014  bool is_ok = generalizeCounterexample(new_ce, in);
1015  if(is_ok)
1016  {
1017  statistics_.notifyAfterCheckCandidateFound(orig_size, new_ce.size());
1018 
1019  if(Utils::containsInit(new_ce))
1020  {
1021  coordinator_.result_ = UNREALIZABLE; // should be atomic
1022  return;
1023  }
1024 
1025  gen_ces.push_back(new_ce);
1026  vector<int> blocking_clause(new_ce);
1027  Utils::negateLiterals(blocking_clause);
1028  for(size_t cnt = 0; cnt < new_ce.size(); ++cnt)
1029  {
1030  set<int> new_must_not_contain(must_not_contain);
1031  new_must_not_contain.insert(new_ce[cnt]);
1032  must_not_contain_queue.push_back(new_must_not_contain);
1033  }
1034 
1035  // check if blocking_clause is implied by existing clauses:
1036  if(solver_win_->incIsSat(new_ce))
1037  {
1038  do_if_bored_.push_back(make_pair(new_ce, in));
1039  coordinator_.notifyNewWinRegClause(blocking_clause, CE_GEN);
1040  }
1041  }
1042  else
1043  statistics_.notifyAfterCheckCandidateFailed();
1044  }
1045  }

1667 // -------------------------------------------------------------------------------------------
1668 void IFM13Explorer::addBlockedState(const vector<int> &state_cube, size_t level)
1669 {
1670  // We require that the cube does not intersect with R0, i.e., the bad states.
1671  // The intuitive reason is that the bad states do not have to have a successor (in
1672  // R[level-1]) order to be problematic.
1673  // Often there are bad states, which do not have a successor in the bad states.
1674  // Before minimization with unsat cores, the blocked state is actually completely outside
1675  // of the bad states. Now, instead of coming up with a complex generalization procedure
1676  // which ensures that the cube does not intersect with the bad states, we can simply add
1677  // this property later, because the bad states are characterized by one state bit in our
1678  // setting. This saves time.
1679 
1680  vector<int> blocking_clause;
1681  blocking_clause.reserve(state_cube.size() + 1);
1682  for(size_t cnt = 0; cnt < state_cube.size(); ++cnt)
1683  blocking_clause.push_back(-state_cube[cnt]);
1684  if(!Utils::contains(blocking_clause, error_var_))
1685  blocking_clause.push_back(error_var_);
1686  vector<int> next_blocking_clause(blocking_clause);
1687  Utils::swapPresentToNext(next_blocking_clause);
1688  for(size_t l_cnt = 0; l_cnt <= level; ++l_cnt)
1689  {
1690  getR(l_cnt).addClause(blocking_clause);
1691  getGotoNextLowerSolver(l_cnt+1)->incAddClause(next_blocking_clause);
1692  getGenBlockTransSolver(l_cnt+1)->incAddClause(next_blocking_clause);
1693  }
1694 
1695  // BEGIN optimization C
1696  // Performance optimization not mentioned in the IFM13 paper, but exploited in
1697  // Andreas Morgenstern's implementation:
1698  // We try to push the blocking_clause forward as far as possible:
1699  vector<int> neg_clause(state_cube);
1700  if(!Utils::contains(neg_clause, -error_var_))
1701  neg_clause.push_back(-error_var_);
1702  for(size_t l_cnt = level + 1; l_cnt < r_.size(); ++l_cnt)
1703  {
1704  if(!getGotoNextLowerSolver(l_cnt)->incIsSat(neg_clause))
1705  {
1706  getR(l_cnt).addClause(blocking_clause);
1707  getGotoNextLowerSolver(l_cnt + 1)->incAddClause(next_blocking_clause);
1708  getGenBlockTransSolver(l_cnt + 1)->incAddClause(next_blocking_clause);
1709  }
1710  else
1711  break;
1712  }
1713  // END optimization C
1714 
1715 }
