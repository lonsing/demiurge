119 PrevStateInfo::PrevStateInfo(bool use_ind):
120  use_ind_(use_ind),
121  current_state_is_initial_(0),
122  prev_safe_(0)
123 {
124  if(use_ind_)
125  {
126  // build previous-state copy of the transition relation:
127  AIG2CNF& A2C = AIG2CNF::instance();
128  VarManager &VM = VarManager::instance();
129  current_to_previous_map_.resize(VM.getMaxCNFVar()+1, 0);
130  const vector<int> &s = VM.getVarsOfType(VarInfo::PRES_STATE);
131  const vector<int> &s_next = VM.getVarsOfType(VarInfo::NEXT_STATE);
132  const vector<int> &c = VM.getVarsOfType(VarInfo::CTRL);
133  const vector<int> &i = VM.getVarsOfType(VarInfo::INPUT);
134  vector<int> t = VM.getVarsOfType(VarInfo::TMP);
135 
136  for(size_t v_cnt = 0; v_cnt < s.size(); ++v_cnt)
137  current_to_previous_map_[s[v_cnt]] = VM.createFreshPrevVar();
138  for(size_t v_cnt = 0; v_cnt < c.size(); ++v_cnt)
139  current_to_previous_map_[c[v_cnt]] = VM.createFreshTmpVar();
140  for(size_t v_cnt = 0; v_cnt < i.size(); ++v_cnt)
141  current_to_previous_map_[i[v_cnt]] = VM.createFreshTmpVar();
142  for(size_t v_cnt = 0; v_cnt < t.size(); ++v_cnt)
143  current_to_previous_map_[t[v_cnt]] = VM.createFreshTmpVar();
144  for(size_t v_cnt = 0; v_cnt < s_next.size(); ++v_cnt)
145  current_to_previous_map_[s_next[v_cnt]] = s[v_cnt];
146 
147  current_state_is_initial_ = VM.createFreshTmpVar();
148  list<vector<int> > prev_trans_clauses = A2C.getTrans().getClauses();
149  for(CNF::ClauseIter it = prev_trans_clauses.begin(); it != prev_trans_clauses.end(); ++it)
150  {
151  presentToPrevious(*it);
152  it->push_back(current_state_is_initial_);
153  }
154  prev_trans_or_initial_.swapWith(prev_trans_clauses);
155  // if one of the state variables is true, then current_state_is_initial must be false:
156  for(size_t cnt = 0; cnt < s.size(); ++cnt)
157  prev_trans_or_initial_.add2LitClause(-s[cnt], -current_state_is_initial_);
158  prev_safe_ = -current_to_previous_map_[VM.getPresErrorStateVar()];
159 
160  // begin activation variables for a previous state clause:
161  px_unused_.reserve(s.size());
162  px_neg_.reserve(s.size());
163  px_act_.reserve(s.size() + 1);
164  for(size_t cnt = 0; cnt < s.size(); ++cnt)
165  {
166  px_unused_.push_back(VM.createFreshPrevVar());
167  px_neg_.push_back(VM.createFreshPrevVar());
168  px_act_.push_back(VM.createFreshPrevVar());
169  }
170  px_act_.push_back(current_state_is_initial_);
171  prev_trans_or_initial_.addClause(px_act_);
172  for(size_t cnt = 0; cnt < s.size(); ++cnt)
173  {
174  int prev = current_to_previous_map_[s[cnt]];
175  prev_trans_or_initial_.add2LitClause(-px_unused_[cnt], -px_act_[cnt]);
176  prev_trans_or_initial_.add4LitClause(px_unused_[cnt], px_neg_[cnt], prev, -px_act_[cnt]);
177  prev_trans_or_initial_.add4LitClause(px_unused_[cnt], px_neg_[cnt], -prev, px_act_[cnt]);
178  prev_trans_or_initial_.add4LitClause(px_unused_[cnt], -px_neg_[cnt], prev, px_act_[cnt]);
179  prev_trans_or_initial_.add4LitClause(px_unused_[cnt], -px_neg_[cnt], -prev, -px_act_[cnt]);
180  }
181  // end activation variables for a previous state clause
182  prev_vars_ = VM.getVarsOfType(VarInfo::PREV);
183  }
184 }

391 bool ParallelLearner::run()
392 {
393  VarManager::instance().push();
394  L_INF("Starting to compute a winning region using " << nr_of_threads_ << " threads ...");
395  statistics_.notifyWinRegStart();
396  winning_region_.clear();
397  winning_region_.addCNF(AIG2CNF::instance().getSafeStates());
398 
399  // Start the threads:
400  vector<thread> explorer_threads;
401  explorer_threads.reserve(clause_explorers_.size() - 1);
402  for(size_t cnt = 1; cnt < clause_explorers_.size(); ++cnt)
403  explorer_threads.push_back(thread(&ClauseExplorerSAT::exploreClauses,
404  clause_explorers_[cnt]));
405 
406  vector<thread> gen_threads;
407  gen_threads.reserve(ce_generalizers_.size());
408  for(size_t cnt = 0; cnt < ce_generalizers_.size(); ++cnt)
409  gen_threads.push_back(thread(&CounterGenSAT::generalizeCounterexamples,
410  ce_generalizers_[cnt]));
411 
412  vector<thread> minimizer_threads;
413  minimizer_threads.reserve(clause_minimizers_.size());
414  for(size_t cnt = 0; cnt < clause_minimizers_.size(); ++cnt)
415  minimizer_threads.push_back(thread(&ClauseMinimizerQBF::minimizeClauses,
416  clause_minimizers_[cnt]));
417 
418  vector<thread> ifm_threads;
419  ifm_threads.reserve(ifm_explorers_.size());
420  for(size_t cnt = 0; cnt < ifm_explorers_.size(); ++cnt)
421  ifm_threads.push_back(thread(&IFM13Explorer::exploreClauses,
422  ifm_explorers_[cnt]));
423 
424  vector<thread> templ_threads;
425  templ_threads.reserve(templ_explorers_.size());
426  for(size_t cnt = 0; cnt < templ_explorers_.size(); ++cnt)
427  templ_threads.push_back(thread(&TemplExplorer::computeWinningRegion,
428  templ_explorers_[cnt]));
429 
430  //The main thread executes the first explorer:
431  MASSERT(clause_explorers_.size() > 0, "There must be at least one explorer thread");
432  clause_explorers_[0]->exploreClauses();
433 
434  // Wait until the threads are finished:
435  for(size_t cnt = 0; cnt < explorer_threads.size(); ++cnt)
436  explorer_threads[cnt].join();
437  for(size_t cnt = 0; cnt < gen_threads.size(); ++cnt)
438  gen_threads[cnt].join();
439  for(size_t cnt = 0; cnt < minimizer_threads.size(); ++cnt)
440  minimizer_threads[cnt].join();
441  for(size_t cnt = 0; cnt < ifm_threads.size(); ++cnt)
442  ifm_threads[cnt].join();
443  for(size_t cnt = 0; cnt < templ_threads.size(); ++cnt)
444  templ_threads[cnt].join();
445 
446  // Merge statistics:
447  statistics_.notifyWinRegEnd();
448  for(size_t cnt = 0; cnt < clause_explorers_.size(); ++cnt)
449  statistics_.mergeWith(clause_explorers_[cnt]->getStatistics());
450  for(size_t cnt = 0; cnt < ce_generalizers_.size(); ++cnt)
451  statistics_.mergeWith(ce_generalizers_[cnt]->getStatistics());
452 
453  // Extract a circuit:
454  if(result_ == UNREALIZABLE)
455  {
456  L_RES("The specification is unrealizable.");
457  statistics_.logStatistics();
458  return false;
459  }
460  L_RES("The specification is realizable.");
461  Utils::debugCheckWinReg(winning_region_);
462  if(Options::instance().doRealizabilityOnly())
463  {
464  statistics_.logStatistics();
465  return true;
466  }
467 
468  // cleaning up the winning region explorers to save memory:
469  for(size_t cnt = 0; cnt < templ_explorers_.size(); ++cnt)
470  delete templ_explorers_[cnt];
471  templ_explorers_.clear();
472  for(size_t cnt = 0; cnt < clause_explorers_.size(); ++cnt)
473  delete clause_explorers_[cnt];
474  clause_explorers_.clear();
475  for(size_t cnt = 0; cnt < ce_generalizers_.size(); ++cnt)
476  delete ce_generalizers_[cnt];
477  ce_generalizers_.clear();
478  for(size_t cnt = 0; cnt < ifm_explorers_.size(); ++cnt)
479  delete ifm_explorers_[cnt];
480  ifm_explorers_.clear();
481  for(size_t cnt = 0; cnt < clause_minimizers_.size(); ++cnt)
482  delete clause_minimizers_[cnt];
483  clause_minimizers_.clear();
484  unminimized_clauses_.clear();
485  counterexamples_.clear();
486  expander_.cleanup();
487 
488  L_INF("Starting to extract a circuit ...");
489  impl_extractor_->extractCircuit(winning_region_);
490  L_INF("Synthesis done.");
491  statistics_.logStatistics();
492  impl_extractor_->logStatistics();
493  return true;

560 void ParallelLearner::triggerExplorerRestart()
561 {
562  var_man_lock_.lock();
563  winning_region_lock_.lock();
564 
565  bool some_in_mode0 = false;
566  bool some_in_mode1 = false;
567  for(size_t cnt = 0; cnt < clause_explorers_.size(); ++cnt)
568  if(clause_explorers_[cnt]->mode_ == 0)
569  some_in_mode0 = true;
570  else
571  some_in_mode1 = true;
572 
573  Utils::compressStateCNF(winning_region_);
574  CNF win(winning_region_);
575  winning_region_lock_.unlock();
576 
577  CNF leave_win;
578  if(some_in_mode0)
579  {
580  VarManager::instance().resetToLastPush();
581  leave_win = win;
582  leave_win.swapPresentToNext();
583  leave_win.negate();
584  }
585 
586  for(size_t cnt = 0; cnt < clause_explorers_.size(); ++cnt)
587  {
588  if(clause_explorers_[cnt]->mode_ == 0)
589  {
590  SatSolver *next_solver = clause_explorers_[cnt]->getFreshISolver();
591  next_solver->startIncrementalSession(vars_to_keep_i_, false);
592  next_solver->incAddCNF(win);
593  next_solver->incAddCNF(leave_win);
594  next_solver->incAddCNF(AIG2CNF::instance().getTrans());
595  clause_explorers_[cnt]->notifyRestart(next_solver);
596  }
597  }
598  for(size_t cnt = 0; cnt < clause_explorers_.size(); ++cnt)
599  if(clause_explorers_[cnt]->mode_ == 0)
600  clause_explorers_[cnt]->notifyAfterNewInfo();
601 
602  if(some_in_mode1 == false)
603  {
604  var_man_lock_.unlock();
605  return;
606  }
607 
608  // now we compute the mode1 restarts, which is more expensive:
609  // the mode0 explorers can already work in the meantime
610  vector<SatSolver*> solvers;
611  for(size_t cnt = 0; cnt < clause_explorers_.size(); ++cnt)
612  if(clause_explorers_[cnt]->mode_ != 0)
613  solvers.push_back(clause_explorers_[cnt]->getFreshISolver());
614  bool limit_exceeded = expander_.resetSolverIExp(win, solvers, true);
615  if(limit_exceeded)
616  {
617  L_LOG("Memory exceeded: Falling back from mode 1 to 0");
618  for(size_t scnt = 0; scnt < solvers.size(); ++scnt)
619  delete solvers[scnt];
620  if(some_in_mode0 == false)
621  {
622  VarManager::instance().resetToLastPush();
623  leave_win = win;
624  leave_win.swapPresentToNext();
625  leave_win.negate();
626  }
627  for(size_t cnt = 0; cnt < clause_explorers_.size(); ++cnt)
628  {
629  if(clause_explorers_[cnt]->mode_ != 0)
630  {
631  SatSolver *next_solver = clause_explorers_[cnt]->getFreshISolver();
632  next_solver->startIncrementalSession(vars_to_keep_i_, false);
633  next_solver->incAddCNF(win);
634  next_solver->incAddCNF(leave_win);
635  next_solver->incAddCNF(AIG2CNF::instance().getTrans());
636  clause_explorers_[cnt]->notifyRestart(next_solver);
637  clause_explorers_[cnt]->mode_ = 0;
638  clause_explorers_[cnt]->notifyAfterNewInfo();
639  }
640  }
641  }
642  else
643  {
644  size_t next_solver_idx = 0;
645  for(size_t cnt = 0; cnt < clause_explorers_.size(); ++cnt)
646  {
647  if(clause_explorers_[cnt]->mode_ != 0)
648  {
649  clause_explorers_[cnt]->notifyRestart(solvers[next_solver_idx++]);
650  clause_explorers_[cnt]->notifyAfterNewInfo();
651  }
652  }
653  }
654  var_man_lock_.unlock();
655 }

781 // -------------------------------------------------------------------------------------------
782 void ClauseExplorerSAT::exploreClauses()
783 {
784  const AIG2CNF& A2C = AIG2CNF::instance();
785  const vector<int> &s = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE);
786  const vector<int> &c = VarManager::instance().getVarsOfType(VarInfo::CTRL);
787  const vector<int> &i = VarManager::instance().getVarsOfType(VarInfo::INPUT);
788  vector<int> si;
789  si.reserve(s.size() + i.size());
790  si.insert(si.end(), s.begin(), s.end());
791  si.insert(si.end(), i.begin(), i.end());
792  vector<int> sic;
793  sic.reserve(si.size() + c.size());
794  sic.insert(sic.end(), si.begin(), si.end());
795  sic.insert(sic.end(), c.begin(), c.end());
796 
797  solver_ctrl_->startIncrementalSession(vars_to_keep_, false);
798  solver_ctrl_->incAddCNF(A2C.getNextSafeStates());
799  solver_ctrl_->incAddCNF(A2C.getTrans());
800  solver_ctrl_->incAddCNF(A2C.getSafeStates());
801 
802  if(psi_.use_ind_)
803  {
804  exp_.initSolverCExp(solver_ctrl_ind_, psi_.prev_vars_);
805  solver_ctrl_ind_->incAddCNF(A2C.getSafeStates());
806  vector<int> safe;
807  safe.push_back(-VarManager::instance().getPresErrorStateVar());
808  exp_.addExpNxtClauseToC(safe, solver_ctrl_ind_);
809  solver_ctrl_ind_->incAddCNF(psi_.prev_trans_or_initial_);
810  solver_ctrl_ind_->incAddUnitClause(psi_.prev_safe_);
811  }
812 
813  // mode 0 can start without waiting for the initialization of the mode1 explorers
814  solver_i_ = getFreshISolver();
815  solver_i_->startIncrementalSession(sic, false);
816  solver_i_->incAddCNF(A2C.getNextUnsafeStates());
817  solver_i_->incAddCNF(A2C.getTrans());
818  solver_i_->incAddCNF(A2C.getSafeStates());
819 
820  if(mode_ == 1)
821  {
822  // in order to initialize solver_i_, we perform a restart
823  bool first_restart_available = waitUntilOngoingRestartDone();
824  if(coordinator_.result_ != UNKNOWN)
825  return;
826  if(first_restart_available == false)
827  coordinator_.triggerInitialMode1Restart(); //does not block mode 0 explorers
828  }
829 
830  precise_ = true;
831 
832  size_t it_cnt = 0;
833  vector<int> model_or_core;
834  while(true)
835  {
836  ++it_cnt;
837  if(coordinator_.result_ != UNKNOWN)
838  return;
839  considerNewInfoFromOthers();
840  statistics_.notifyBeforeComputeCandidate();
841  if(it_cnt == 1)
842  L_DBG("Explorer " << instance_nr_ << " starts to work.");
843  bool sat = solver_i_->incIsSatModelOrCore(vector<int>(), si, model_or_core);
844  statistics_.notifyAfterComputeCandidate();
845  if(!sat)
846  {
847  if(precise_)
848  {
849  coordinator_.result_ = REALIZABLE;
850  return;
851  }
852  if(coordinator_.result_ != UNKNOWN)
853  return;
854  L_DBG("Explorer " << instance_nr_ << " restarts (iteration " << it_cnt << ").");
855 
856  // minor performance optimization:
857  // if a restart is ongoing, we do not trigger another one but wait here until it is
858  // done. This is not a correctness issue: doing multiple restarts triggered by
859  // several threads is also safe (but a waste of resources).
860  bool restart_available = waitUntilOngoingRestartDone();
861  if(coordinator_.result_ != UNKNOWN)
862  return;
863  if(restart_available == false)
864  coordinator_.triggerExplorerRestart();
865  continue;
866  }
867 
868  if(instance_nr_ != 0)
869  Utils::randomize(model_or_core);
870 
871  vector<int> state_input = model_or_core;
872  vector<int> state = Utils::extract(state_input, VarInfo::PRES_STATE);
873  vector<int> input = Utils::extract(state_input, VarInfo::INPUT);
874 
875  if(coordinator_.result_ != UNKNOWN)
876  return;
877 
878  statistics_.notifyBeforeCheckCandidate();
879  if(psi_.use_ind_ && mode_ == 1) // race condition does not harm here. Just a performance thing.
880  sat = solver_ctrl_ind_->incIsSatModelOrCore(state, input, c, model_or_core);
881  else
882  sat = solver_ctrl_->incIsSatModelOrCore(state, input, c, model_or_core);
883 
884  if(!sat)
885  {
886  // we now try to minimize the core further:
887  bool changed = true;
888  while(changed)
889  {
890  changed = false;
891  vector<int> blocking_clause(model_or_core);
892  Utils::negateLiterals(blocking_clause);
893  if(psi_.use_ind_)
894  {
895  solver_ctrl_ind_->incAddClause(blocking_clause);
896  exp_.addExpNxtClauseToC(blocking_clause, solver_ctrl_ind_);
897  }
898  else
899  {
900  vector<int> next_blocking_clause(blocking_clause);
901  Utils::swapPresentToNext(next_blocking_clause);
902  solver_ctrl_->incAddClause(blocking_clause);
903  solver_ctrl_->incAddClause(next_blocking_clause);
904  }
905 
906  vector<int> orig_core = model_or_core;
907  if(instance_nr_ != 0)
908  Utils::randomize(orig_core);
909  for(size_t lit_cnt = 0; lit_cnt < orig_core.size(); ++lit_cnt)
910  {
911  vector<int> tmp(model_or_core);
912  Utils::remove(tmp, orig_core[lit_cnt]);
913 
914  vector<int> assumptions;
915  assumptions.reserve(input.size() + tmp.size() + s.size());
916  assumptions.insert(assumptions.end(), input.begin(), input.end());
917  assumptions.insert(assumptions.end(), tmp.begin(), tmp.end());
918  if(psi_.use_ind_)
919  {
920  // build the previous state-copy of tmp using the activation variables:
921  for(size_t s_cnt = 0; s_cnt < s.size(); ++s_cnt)
922  {
923  if(Utils::contains(tmp, s[s_cnt]))
924  assumptions.push_back(psi_.px_neg_[s_cnt]);
925  else if(Utils::contains(tmp, -s[s_cnt]))
926  assumptions.push_back(-psi_.px_neg_[s_cnt]);
927  else
928  assumptions.push_back(psi_.px_unused_[s_cnt]);
929  }
930  sat = solver_ctrl_ind_->incIsSat(assumptions);
931  }
932  else
933  sat = solver_ctrl_->incIsSat(assumptions);
934  if(!sat)
935  {
936  model_or_core = tmp;
937  changed = true;
938  }
939  }
940  }
941 
942  if(Utils::containsInit(model_or_core))
943  {
944  coordinator_.result_ = UNREALIZABLE;
945  return;
946  }
947 
948  // We compute the corresponding blocking clause, and update the winning region and the
949  // solvers:
950  vector<int> blocking_clause(model_or_core);
951  Utils::negateLiterals(blocking_clause);
952  statistics_.notifyAfterCheckCandidateFound(s.size(), blocking_clause.size());
953  coordinator_.notifyNewWinRegClause(blocking_clause, EXPL);
954  coordinator_.notifyNewCounterexample(state_input, model_or_core);
955  precise_ = false;
956 
957  }
958  else // sat == true, i.e., this is not a real counterexample
959  {
960  statistics_.notifyAfterCheckCandidateFailed();
961  vector<int> ctrl_cube = model_or_core;
962  if(coordinator_.result_ != UNKNOWN) // should be atomic
963  return;
964  statistics_.notifyBeforeRefine();
965  sat = solver_i_->incIsSatModelOrCore(state_input, ctrl_cube, vector<int>(), model_or_core);
966  MASSERT(sat == false, "Impossible " << instance_nr_);
967  Utils::negateLiterals(model_or_core);
968  coordinator_.notifyNewUselessInputClause(model_or_core, restart_level_);
969  statistics_.notifyAfterRefine(si.size(), model_or_core.size());
970  }
971  }
972 }

1044 // -------------------------------------------------------------------------------------------
1045 void ClauseExplorerSAT::considerNewInfoFromOthers()
1046 {
1047  new_info_lock_.lock();
1048  if(next_solver_i_ != NULL)
1049  {
1050  statistics_.notifyRestart();
1051  delete solver_i_;
1052  solver_i_ = next_solver_i_;
1053  next_solver_i_ = NULL;
1054  restart_level_ = new_restart_level_;
1055  precise_ = true;
1056  }
1057  if(new_win_reg_clauses_for_solver_ctrl_.getNrOfClauses() > 0)
1058  {
1059  if(clauses_added_ > win_.getNrOfClauses() + 100)
1060  {
1061  // reset solver_ctrl_ and solver_ctrl_ind_:
1062  if(reset_c_cnt_ % 1000 == 999)
1063  Utils::compressStateCNF(win_, true);
1064  else if(reset_c_cnt_ % 100 == 99)
1065  Utils::compressStateCNF(win_, false);
1066  CNF next_win(win_);
1067  next_win.swapPresentToNext();
1068  solver_ctrl_->startIncrementalSession(vars_to_keep_, false);
1069  solver_ctrl_->incAddCNF(win_);
1070  solver_ctrl_->incAddCNF(AIG2CNF::instance().getTrans());
1071  solver_ctrl_->incAddCNF(next_win);
1072  if(psi_.use_ind_)
1073  {
1074  CNF prev_win(win_);
1075  psi_.presentToPrevious(prev_win);
1076  exp_.resetSolverCExp(solver_ctrl_ind_);
1077  solver_ctrl_ind_->incAddCNF(win_);
1078  const list<vector<int> > &cl = win_.getClauses();
1079  for(CNF::ClauseConstIter it = cl.begin(); it != cl.end(); ++it)
1080  exp_.addExpNxtClauseToC(*it, solver_ctrl_ind_);
1081  solver_ctrl_ind_->incAddCNF(psi_.prev_trans_or_initial_);
1082  solver_ctrl_ind_->incAddCNF(prev_win);
1083  clauses_added_ = win_.getNrOfClauses();
1084  }
1085  new_win_reg_clauses_for_solver_ctrl_.clear();
1086  reset_c_cnt_++;
1087  }
1088  else
1089  {
1090  CNF next_win_reg_clauses(new_win_reg_clauses_for_solver_ctrl_);
1091  next_win_reg_clauses.swapPresentToNext();
1092  solver_ctrl_->incAddCNF(new_win_reg_clauses_for_solver_ctrl_);
1093  solver_ctrl_->incAddCNF(next_win_reg_clauses);
1094  if(psi_.use_ind_)
1095  {
1096  CNF prev_win_reg_clauses(new_win_reg_clauses_for_solver_ctrl_);
1097  psi_.presentToPrevious(prev_win_reg_clauses);
1098  solver_ctrl_ind_->incAddCNF(new_win_reg_clauses_for_solver_ctrl_);
1099  const list<vector<int> > &cl = new_win_reg_clauses_for_solver_ctrl_.getClauses();
1100  for(CNF::ClauseConstIter it = cl.begin(); it != cl.end(); ++it)
1101  exp_.addExpNxtClauseToC(*it, solver_ctrl_ind_);
1102  solver_ctrl_ind_->incAddCNF(prev_win_reg_clauses);
1103  }
1104  clauses_added_ += new_win_reg_clauses_for_solver_ctrl_.getNrOfClauses();
1105  new_win_reg_clauses_for_solver_ctrl_.clear();
1106  }
1107  }
1108  if(new_win_reg_clauses_for_solver_i_.getNrOfClauses() > 0)
1109  {
1110  solver_i_->incAddCNF(new_win_reg_clauses_for_solver_i_);
1111  precise_ = false;
1112  new_win_reg_clauses_for_solver_i_.clear();
1113  }
1114  if(!precise_ && new_foreign_win_reg_clauses_for_solver_i_.getNrOfClauses() > 0)
1115  {
1116  // We only consider foreign clauses for solver_i_ if we are already imprecise. Otherwise,
1117  // the foreign clauses could prevent the explorers from staying in a precise state long
1118  // enough to proof realizability:
1119  solver_i_->incAddCNF(new_foreign_win_reg_clauses_for_solver_i_);
1120  precise_ = false;
1121  new_foreign_win_reg_clauses_for_solver_i_.clear();
1122  }
1123 
1124  if(mode_ == 0 && restart_level_ == new_useless_input_clauses_level_)
1125  {
1126  solver_i_->incAddCNF(new_useless_input_clauses_);
1127  new_useless_input_clauses_.clear();
1128  }
1129  new_info_lock_.unlock();
1130 }

2087  // of the bad states. Now, instead of coming up with a complex generalization procedure
