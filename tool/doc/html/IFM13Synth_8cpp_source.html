125  bool realizable = computeWinningRegion();
126 
127  if(!realizable)

134  if(Options::instance().doRealizabilityOnly())

136

145 bool IFM13Synth::computeWinningRegion()
146 {
147  const vector<int> &state_vars = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE);
148  vector<int> initial_state_cube(state_vars.size(), 0);
149  for(size_t cnt = 0; cnt < state_vars.size(); ++cnt)
150  initial_state_cube[cnt] = -state_vars[cnt];
151 
152  size_t k = 1;
153  while(true)
154  {
155  L_DBG("------ Iteration k=" << k);
156  if(recBlockCube(initial_state_cube, k) == IS_LOSE)
157  return false;
158 
159  debugCheckInvariants(k);
160  size_t equal = propagateBlockedStates(k);
161  //debugCheckInvariants(k);
162  if(equal != 0)
163  {
164  L_LOG("Found two equal clause sets: R" << (equal-1) << " and R" << equal);
165  L_LOG("Nr of iterations: " << k);
166  //L_DBG("Clauses: " << getR(equal).toString());
167  winning_region_ = getR(equal);
168  winning_region_.negate();
169  neg_winning_region_ = getR(equal);
170  return true;
171  }
172  ++k;
173  }
174 }

176 // -------------------------------------------------------------------------------------------
177 size_t IFM13Synth::propagateBlockedStates(size_t max_level)
178 {
179  for(size_t i = 0; i <= max_level; ++i)
180  getR(i).removeDuplicates();
181 
182  for(size_t i = 1; i <= max_level; ++i)
183  {
184  bool equal = true;
185  const list<vector<int> > &r1clauses = getR(i).getClauses();
186  CNF& r2cnf = getR(i+1);
187  list<vector<int> > r2clauses = r2cnf.getClauses();
188  list<vector<int> >::const_iterator it1 = r1clauses.begin();
189  list<vector<int> >::iterator it2 = r2clauses.begin();
190 
191  for(; it1 != r1clauses.end(); )
192  {
193  if(it2 != r2clauses.end() && *it1 == *it2)
194  {
195  ++it1;
196  ++it2;
197  continue;
198  }
199  vector<int> neg_clause(*it1);
200  for(size_t lit_cnt = 0; lit_cnt < neg_clause.size(); ++lit_cnt)
201  neg_clause[lit_cnt] = -neg_clause[lit_cnt];
202  if(!getGotoNextLowerSolver(i+1)->incIsSat(neg_clause))
203  {
204  // There is no edge from s (the negated clause) to Ri
205  // --> no state in s can be part of Ri+1
206  it2 = r2clauses.insert(it2,*it1);
207  vector<int> propagated(*it1);
208  Utils::swapPresentToNext(propagated);
209  getGotoNextLowerSolver(i+2)->incAddClause(propagated);
210  getGenBlockTransSolver(i+2)->incAddClause(propagated);
211  ++it1;
212  ++it2;
213  }
214  else
215  {
216  // Propagation failed
217  ++it1;
218  equal = false;
219  }
220  }
221  r2cnf.swapWith(r2clauses);
222  if(equal)
223  return i + 1;
224  }
225  return 0;
226 }

228 // -------------------------------------------------------------------------------------------
229 bool IFM13Synth::recBlockCube(const vector<int> &state_cube, size_t level)
230 {
231  list<IFMProofObligation> queue;
232  queue.push_back(IFMProofObligation(state_cube, level));
233 
234  vector<int> model_or_core;
235  while(!queue.empty())
236  {
237  IFMProofObligation proof_obligation = popMin(queue);
238  const vector<int> &s = proof_obligation.getState();
239  size_t s_level = proof_obligation.getLevel();
240 
241  if(isLose(s))
242  continue;
243  if(isBlocked(s, s_level))
244  {
245  // BEGIN optimization A
246  // Performance optimization not mentioned in the IFM'13 paper, but exploited in
247  // Andreas Morgenstern's implementation:
248  // We block the transition of the predecessor to s
249  if(proof_obligation.hasPre())
250  {
251  const vector<int> &si = proof_obligation.getPreStateInCube();
252  const vector<int> &c = proof_obligation.getPreCtrlCube();
253  genAndBlockTrans(si, c, s_level + 1);
254  }
255  // END optimization A
256  continue;
257  }
258  SatSolver *goto_next_lower = getGotoNextLowerSolver(s_level);
259  bool isSat = goto_next_lower->incIsSatModelOrCore(s, sin_, model_or_core);
260  if(isSat)
261  {
262  //L_DBG("Found transition from R" << s_level << " to R" << (s_level - 1));
263  vector<int> succ = Utils::extractNextAsPresent(model_or_core);
264  if(s_level == 1 || isLose(succ))
265  {
266  //L_DBG(" Successor is losing");
267  vector<int> in_cube = Utils::extract(model_or_core, VarInfo::INPUT);
268  vector<int> state_cube = Utils::extract(model_or_core, VarInfo::PRES_STATE);
269  isSat = goto_win_solver_->incIsSatModelOrCore(state_cube, in_cube, sicn_, model_or_core);
270  if(isSat)
271  {
272  //L_DBG(" We can also go to a winning state");
273  succ = Utils::extractNextAsPresent(model_or_core);
274  vector<int> si = Utils::extractPresIn(model_or_core);
275  vector<int> c = Utils::extract(model_or_core, VarInfo::CTRL);
276  if(s_level == 1 || isBlocked(succ, s_level - 1))
277  {
278  // generalize the state-input pair (s,u):
279  // find subset of s,i literals such that c input does not lead to Ri-1
280  // A QBF call could also be a good option here
281  genAndBlockTrans(si, c, s_level);
282  }
283  else
284  queue.push_back(IFMProofObligation(succ, s_level - 1, si, c));
285  queue.push_back(proof_obligation);
286  }
287  else
288  {
289  //L_DBG(" We cannot go to a winning state");
290  if(Utils::containsInit(model_or_core))
291  return IS_LOSE;
292  addLose(model_or_core);
293  }
294  }
295  else
296  {
297  //L_DBG(" Successor is winning");
298  vector<int> si = Utils::extractPresIn(model_or_core);
299  vector<int> c = Utils::extract(model_or_core, VarInfo::CTRL);
300  queue.push_back(IFMProofObligation(succ, s_level - 1, si, c));
301  queue.push_back(proof_obligation);
302  }
303  }
304  else
305  {
306  //L_DBG(" No transition from R" << s_level << " to R" << (s_level - 1));
307  addBlockedState(model_or_core, s_level);
308  // BEGIN optimization A
309  // Performance optimization not mentioned in the IFM13 paper, but exploited in
310  // Andreas Morgenstern's implementation:
311  // We block the transition of the predecessor to s
312  if(proof_obligation.hasPre())
313  {
314  const vector<int> &si = proof_obligation.getPreStateInCube();
315  const vector<int> &c = proof_obligation.getPreCtrlCube();
316  genAndBlockTrans(si, c, s_level + 1);
317  }
318  // END optimization A
319 
320  // BEGIN optimization B
321  // Performance optimization not mentioned in the IFM13 paper, but exploited in
322  // Andreas Morgenstern's implementation:
323  // We aggressively decide s also on later levels:
324  //if(s_level < r_.size() - 1)
325  //{
326  // queue.push_back(IFMProofObligation(s, s_level+1, proof_obligation.getPreStateInCube(),
327  // proof_obligation.getPreCtrlCube()));
328  //}
329  // END optimization B
330  }
331  }
332  return IS_GREATER;
333 }

336 void IFM13Synth::addBlockedTransition(const vector<int> &state_in_cube, size_t level)
337 {
338  vector<int> blocking_clause(state_in_cube);
339  for(size_t cnt = 0; cnt < blocking_clause.size(); ++cnt)
340  blocking_clause[cnt] = -blocking_clause[cnt];
341  for(size_t l_cnt = 1; l_cnt <= level; ++l_cnt)
342  {
343 #ifndef NDEBUG
344  getU(l_cnt).addClauseAndSimplify(blocking_clause);
345 #endif
346  getGotoNextLowerSolver(l_cnt)->incAddClause(blocking_clause);
347  }

350 // -------------------------------------------------------------------------------------------
351 void IFM13Synth::addBlockedState(const vector<int> &state_cube, size_t level)
352 {
353  // We require that the cube does not intersect with R0, i.e., the bad states.
354  // The intuitive reason is that the bad states do not have to have a successor (in
355  // R[level-1]) order to be problematic.
356  // Often there are bad states, which do not have a successor in the bad states.
357  // Before minimization with unsat cores, the blocked state is actually completely outside
358  // of the bad states. Now, instead of coming up with a complex generalization procedure
359  // which ensures that the cube does not intersect with the bad states, we can simply add
360  // this property later, because the bad states are characterized by one state bit in our
361  // setting. This saves time.
362 
363  vector<int> blocking_clause;
364  blocking_clause.reserve(state_cube.size() + 1);
365  for(size_t cnt = 0; cnt < state_cube.size(); ++cnt)
366  blocking_clause.push_back(-state_cube[cnt]);
367  blocking_clause.push_back(VarManager::instance().getPresErrorStateVar());
368  vector<int> next_blocking_clause(blocking_clause);
369  Utils::swapPresentToNext(next_blocking_clause);
370  for(size_t l_cnt = 0; l_cnt <= level; ++l_cnt)
371  {
372  getR(l_cnt).addClause(blocking_clause);
373  getGotoNextLowerSolver(l_cnt+1)->incAddClause(next_blocking_clause);
374  getGenBlockTransSolver(l_cnt+1)->incAddClause(next_blocking_clause);
375  }
376 
377  // BEGIN optimization C
378  // Performance optimization not mentioned in the IFM13 paper, but exploited in
379  // Andreas Morgenstern's implementation:
380  // We try to push the blocking_clause forward as far as possible:
381  vector<int> neg_clause(state_cube);
382  neg_clause.push_back(-VarManager::instance().getPresErrorStateVar());
383  for(size_t l_cnt = level + 1; l_cnt < r_.size(); ++l_cnt)
384  {
385  if(!getGotoNextLowerSolver(l_cnt)->incIsSat(neg_clause))
386  {
387  getR(l_cnt).addClause(blocking_clause);
388  getGotoNextLowerSolver(l_cnt + 1)->incAddClause(next_blocking_clause);
389  getGenBlockTransSolver(l_cnt + 1)->incAddClause(next_blocking_clause);
390  }
391  else
392  break;
393  }
394  // END optimization C
395 
396 }

475 IFMProofObligation IFM13Synth::popMin(list<IFMProofObligation> &queue)
476 {
477  MASSERT(!queue.empty(), "cannot pop from empty queue");
478  list<IFMProofObligation>::iterator min = queue.begin();
479  list<IFMProofObligation>::iterator check = queue.begin();
480  ++check;
481  for(;check != queue.end(); ++check)
482  {
483  if(check->getLevel() < min->getLevel())
484  min = check;
485  }
486  IFMProofObligation res = *min;
487  queue.erase(min);
488  return res;
489 }

501 // -------------------------------------------------------------------------------------------
502 void IFM13Synth::debugCheckInvariants(size_t k)
503 {
504 #ifndef NDEBUG
505  SatSolver *sat_solver = Options::instance().getSATSolver();
506 
507  // check if W implies P:
508  CNF check(AIG2CNF::instance().getUnsafeStates());
509  check.addCNF(win_);
510  bool v = sat_solver->isSat(check);
511  MASSERT(!v, "W does not imply P");
512 
513  // check if R0 is still !P:
514  check.clear();
515  check.addCNF(getR(0));
516  check.negate();
517  check.addCNF(AIG2CNF::instance().getUnsafeStates());
518  v = sat_solver->isSat(check);
519  MASSERT(!v, "R0 is unequal to !P");
520  check.clear();
521  check.addCNF(AIG2CNF::instance().getUnsafeStates());
522  check.negate();
523  check.addCNF(getR(0));
524  v = sat_solver->isSat(check);
525  MASSERT(!v, "R0 is unequal to !P");
526 
527  // check if R_i implies Ri+1:
528  for(size_t i = 0; i < k; ++i)
529  {
530  CNF check(getR(i+1));
531  check.negate();
532  check.addCNF(getR(i));
533  bool v = sat_solver->isSat(check);
534  MASSERT(!v, "R" << i << " does not imply R" << i+1);
535  }
536 
537  // check if all R_i imply not s_0:
538  for(size_t i = 0; i <= k; ++i)
539  {
540  CNF check(getR(i));
541  check.addCNF(AIG2CNF::instance().getInitial());
542  bool v = sat_solver->isSat(check);
543  MASSERT(!v, "R" << i << " does not imply !s0" << i+1);
544  }
545 
546  // check if U_i implies Ui+1:
547  for(size_t i = 1; i < k; ++i)
548  {
549  CNF check(getU(i+1));
550  check.negate();
551  check.addCNF(getU(i));
552  bool v = sat_solver->isSat(check);
553  MASSERT(!v, "U" << i << " does not imply U" << i+1);
554  }
555 
556 
557  // check if Force^e_1(R_i) => Ri+1
558  // i.e., if all states from which then environment can enforce R_i are contained in Ri+1:
559  // \exists x,i: \forall c: \exists x',tmp: !Ri+1 & T & ri'
560  DepQBFApi qbf_solver;
561  vector<pair<VarInfo::VarKind, QBFSolver::Quant> > env_quant;
562  env_quant.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
563  env_quant.push_back(make_pair(VarInfo::INPUT, QBFSolver::E));
564  env_quant.push_back(make_pair(VarInfo::CTRL, QBFSolver::A));
565  env_quant.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
566  env_quant.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
567 
568  for(size_t i = 0; i < k; ++i)
569  {
570  CNF check(getR(i));
571  check.swapPresentToNext();
572  check.addCNF(AIG2CNF::instance().getTrans());
573  CNF neg_r_next(getR(i+1));
574  neg_r_next.negate();
575  check.addCNF(neg_r_next);
576  bool v = qbf_solver.isSat(env_quant, check);
577  MASSERT(!v, "R" << i+1 << " misses a state from which the ENV can enforce R" << i);
578  }
579 
580  // U_i+1 must not exclude a state-input pair for which the environment can enforce
581  // to go to R_i
582  for(size_t i = 0; i < k; ++i)
583  {
