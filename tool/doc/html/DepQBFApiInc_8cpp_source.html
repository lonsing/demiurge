57 bool DepQBFApiInc::isSat(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
58  const CNF& cnf)
59 {
60  if(use_bloqqer_)
61  {
62  initBloqqer(quantifier_prefix, cnf);
63  bloqqer_preprocess();
64  int res = bloqqer_solve();
65  //debugCheckBloqqerVerdict(res, quantifier_prefix, cnf);
66  bloqqer_release();
67  if(res == 10)
68  return true;
69  else if(res == 20)
70  return false;
71  MASSERT(false, "Strange return code from Bloqqer/DepQBF.");
72  }
73  else
74  {
75  QDPLL *solver = qdpll_create();
76  initDepQBF(solver, quantifier_prefix, cnf);
77  QDPLLResult res = qdpll_sat(solver);
78  qdpll_delete(solver);
79  if(res == QDPLL_RESULT_SAT)
80  return true;
81  if(res == QDPLL_RESULT_UNSAT)
82  return false;
83  MASSERT(false, "Strange return code from DepQBF.");
84  }
85  return false;
86 }

121 bool DepQBFApiInc::isSatModel(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
122  const CNF& cnf,
123  vector<int> &model)
124 {
125  if(use_bloqqer_)
126  {
127  initBloqqer(quantifier_prefix, cnf);
128  bloqqer_preprocess();
129  int res = bloqqer_solve();
130  if(res == 20)
131  {
132  debugCheckBloqqerVerdict(res, quantifier_prefix, cnf);
133  bloqqer_release();
134  return false;
135  }
136  MASSERT(res == 10, "Strange return code from Bloqqer/DepQBF.");
137  VarManager &VM = VarManager::instance();
138  size_t model_size = 0;
139  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
140  {
141  model_size += VM.getVarsOfType(quantifier_prefix[q_cnt].first).size();
142  if(q_cnt + 1 < quantifier_prefix.size())
143  if(quantifier_prefix[q_cnt].second != quantifier_prefix[q_cnt+1].second)
144  break;
145  }
146  model.clear();
147  model.reserve(model_size);
148  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
149  {
150  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[q_cnt].first);
151  for(size_t v_cnt = 0; v_cnt < vars.size(); ++v_cnt)
152  {
153  VarValue val = bloqqer_getvalue(vars[v_cnt]);
154  MASSERT(val != UNKN, "Bloqqer return an UNKN as value");
155  if(val == NEG)
156  model.push_back(-vars[v_cnt]);
157  else if(val == POS)
158  model.push_back(vars[v_cnt]);
159  }
160  if(q_cnt + 1 < quantifier_prefix.size())
161  if(quantifier_prefix[q_cnt].second != quantifier_prefix[q_cnt+1].second)
162  break;
163  }
164  //debugCheckBloqqerModel(model, quantifier_prefix, cnf);
165  bloqqer_release();
166  return true;
167  }
168  else
169  {
170  QDPLL *solver = qdpll_create();
171  initDepQBF(solver, quantifier_prefix, cnf);
172  QDPLLResult res = qdpll_sat(solver);
173  if(res == QDPLL_RESULT_UNSAT)
174  {
175  qdpll_delete(solver);
176  return false;
177  }
178  MASSERT(res != QDPLL_RESULT_UNKNOWN, "Strange return code from DepQBF.");
179  VarManager &VM = VarManager::instance();
180  size_t model_size = 0;
181  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
182  {
183  model_size += VM.getVarsOfType(quantifier_prefix[q_cnt].first).size();
184  if(q_cnt + 1 < quantifier_prefix.size())
185  if(quantifier_prefix[q_cnt].second != quantifier_prefix[q_cnt+1].second)
186  break;
187  }
188  model.clear();
189  model.reserve(model_size);
190  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
191  {
192  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[q_cnt].first);
193  for(size_t v_cnt = 0; v_cnt < vars.size(); ++v_cnt)
194  {
195  QDPLLAssignment val = qdpll_get_value(solver, vars[v_cnt]);
196  if(val == QDPLL_ASSIGNMENT_FALSE)
197  model.push_back(-vars[v_cnt]);
198  else if(val == QDPLL_ASSIGNMENT_TRUE)
199  model.push_back(vars[v_cnt]);
200  }
201  if(q_cnt + 1 < quantifier_prefix.size())
202  if(quantifier_prefix[q_cnt].second != quantifier_prefix[q_cnt+1].second)
203  break;
204  }
205  qdpll_delete(solver);
206  return true;

327  for(size_t v_cnt = 0; v_cnt < vars.size(); ++v_cnt)
328  inc_model_vars_.push_back(vars[v_cnt]);

484  model_or_core.push_back(inc_model_vars_[var_cnt]);

486  qdpll_reset(inc_solver_);
487  return true;
488  }
489  else if(res == QDPLL_RESULT_UNSAT)
490  {
491  if(assumptions.size() > 0)
492  {
493  model_or_core.clear();
494  model_or_core.reserve(assumptions.size());
495  extractCore(inc_solver_, model_or_core);
496  }
497  else
498  qdpll_reset(inc_solver_);
499  return false;
500  }
501  else
502  {
503  MASSERT(false, "Strange return code from DepQBF.");
504  }
505 #else
506  MASSERT(false, "This feature is only available with an experimental version of DepQBF. ");
507 #endif

574 // -------------------------------------------------------------------------------------------
575 void DepQBFApiInc::initDepQBF(QDPLL *solver,
576  const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
577  const CNF& cnf)
578 {
579  VarManager &VM = VarManager::instance();
580  qdpll_adjust_vars(solver, VM.getMaxCNFVar());
581  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
582  {
583  if(quantifier_prefix[q_cnt].second == E)
584  qdpll_new_scope(solver, QDPLL_QTYPE_EXISTS);
585  else
586  qdpll_new_scope(solver, QDPLL_QTYPE_FORALL);
587  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[q_cnt].first);
588  for(size_t v_cnt = 0; v_cnt < vars.size(); ++v_cnt)
589  qdpll_add(solver, vars[v_cnt]);
590  qdpll_add(solver, 0);
591  }
592  const list<vector<int> > &clauses = cnf.getClauses();
593  for(CNF::ClauseConstIter it = clauses.begin(); it != clauses.end(); ++it)
594  {

603  const vector<pair<vector<int>, Quant> > &quantifier_prefix,
604  const CNF& cnf)
605 {
606  int max_cnf_var = 0;
607  for(size_t level = 0; level < quantifier_prefix.size(); ++level)
608  {
609  for(size_t var_cnt = 0; var_cnt < quantifier_prefix[level].first.size(); ++var_cnt)

611  if(quantifier_prefix[level].first[var_cnt] > max_cnf_var)
612  max_cnf_var = quantifier_prefix[level].first[var_cnt];
613  }
614  }
615 
616  qdpll_adjust_vars(solver, max_cnf_var);
617  for(size_t q_cnt = 0; q_cnt < quantifier_prefix.size(); ++q_cnt)
618  {
619  if(quantifier_prefix[q_cnt].second == E)
620  qdpll_new_scope(solver, QDPLL_QTYPE_EXISTS);
621  else
622  qdpll_new_scope(solver, QDPLL_QTYPE_FORALL);
623  const vector<int> &vars = quantifier_prefix[q_cnt].first;
624  for(size_t v_cnt = 0; v_cnt < vars.size(); ++v_cnt)
625  qdpll_add(solver, vars[v_cnt]);
626  qdpll_add(solver, 0);
627  }
628  const list<vector<int> > &clauses = cnf.getClauses();
629  for(CNF::ClauseConstIter it = clauses.begin(); it != clauses.end(); ++it)
630  {
631  for(size_t lit_cnt = 0; lit_cnt < it->size(); ++lit_cnt)

633  qdpll_add(solver, 0);

638 void DepQBFApiInc::extractCore(QDPLL *solver, vector<int> &core)
639 {
640 #ifdef DEPQBF_CORE_AVAILABLE
641  LitID *raw_core = qdpll_get_relevant_assumptions(solver);
642  LitID *next_core_element = raw_core;
643  while(*next_core_element != 0)
644  {
645  core.push_back(*next_core_element);
646  ++next_core_element;
647  }
648  free(raw_core);
649  raw_core = NULL;
650  qdpll_reset(solver);
651  if(min_cores_)
652  {
653  vector<int> orig_core(core);
654  Utils::randomize(orig_core);
655  for(size_t cnt = 0; cnt < orig_core.size(); ++cnt)
656  {
657  vector<int> tmp(core);
658  bool found = Utils::remove(tmp, orig_core[cnt]);
659  if(found)
660  {
661  for(size_t ass_cnt = 0; ass_cnt < tmp.size(); ++ass_cnt)
662  qdpll_assume(solver, tmp[ass_cnt]);
663  QDPLLResult res = qdpll_sat(solver);
664  if(res == QDPLL_RESULT_UNSAT)
665  {
666  core.clear();
667  core.reserve(tmp.size());
668  raw_core = qdpll_get_relevant_assumptions(solver);
669  next_core_element = raw_core;
670  while(*next_core_element != 0)
671  {
672  core.push_back(*next_core_element);
673  ++next_core_element;
674  }
675  }
676  qdpll_reset(solver);
677  }
678  }
679  }
680 #else
681  MASSERT(false, "This feature is only available with an experimental version of DepQBF. ");
682 #endif
683 }
