48 LearningImplExtractor::LearningImplExtractor() :
49  qbf_solver_(Options::instance().getQBFSolver()),
50  standalone_circuit_(aiger_init())
51 {
52  // compute some lists that are used often:
53  const vector<int> &in = VarManager::instance().getVarsOfType(VarInfo::INPUT);
54  const vector<int> &pres = VarManager::instance().getVarsOfType(VarInfo::PRES_STATE);
55  const vector<int> &ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL);
56  ip_.reserve(in.size() + pres.size());
57  ip_.insert(ip_.end(), in.begin(), in.end());
58  ip_.insert(ip_.end(), pres.begin(), pres.end());
59  ipc_.reserve(ip_.size() + ctrl.size());
60  ipc_.insert(ipc_.end(), ip_.begin(), ip_.end());
61  ipc_.insert(ipc_.end(), ctrl.begin(), ctrl.end());
62 
63  // next, we would like to compute a map from CNF variables to the corresponding AIGER
64  // literals in the standalone_circuit_. We want to use a vector instead of a map for
65  // performance reasons. Hence, we first of all need to find out the highest CNF var
66  // we may want to want to transform into the corresponding AIGER variable:
67  int max_cnf_var_of_interest = 0;
68  for(size_t cnt = 0; cnt < ipc_.size(); ++cnt)
69  if(ipc_[cnt] > max_cnf_var_of_interest)
70  max_cnf_var_of_interest = ipc_[cnt];
71  const map<int, set<VarInfo> > &trans_deps = AIG2CNF::instance().getTmpDepsTrans();
72  for(AIG2CNF::DepConstIter it = trans_deps.begin(); it != trans_deps.end(); ++it)
73  if(it->first > max_cnf_var_of_interest)
74  max_cnf_var_of_interest = it->first;
75 
76  // Now we fill the map from CNF variables to AIGER variables. While doing this, we also
77  // initialize the standalone_circuit_:
78  next_free_aig_lit_ = 2;
79  cnf_var_to_standalone_aig_var_.resize(max_cnf_var_of_interest + 1, 0);
80  for(size_t cnt = 0; cnt < ip_.size(); ++cnt)
81  {
82  if(ip_[cnt] != VarManager::instance().getPresErrorStateVar())
83  {
84  cnf_var_to_standalone_aig_var_[ip_[cnt]] = next_free_aig_lit_;
85  ostringstream name;
86  VarInfo info = VarManager::instance().getInfo(ip_[cnt]);
87  name << "Original AIG literal: " << info.getLitInAIG();
88  if(info.getName() != "")
89  name << " (" << info.getName() << ")";
90  aiger_add_input(standalone_circuit_, next_free_aig_lit_, name.str().c_str());
91  next_free_aig_lit_ += 2;
92  }
93  }
94 
95  for(size_t cnt = 0; cnt < ctrl.size(); ++cnt)
96  {
97  cnf_var_to_standalone_aig_var_[ctrl[cnt]] = next_free_aig_lit_;
98  ostringstream name;
99  VarInfo info = VarManager::instance().getInfo(ctrl[cnt]);
100  name << "Original AIG literal: " << info.getLitInAIG();
101  if(info.getName() != "")
102  name << " (" << info.getName() << ")";
103  aiger_add_output(standalone_circuit_, next_free_aig_lit_, name.str().c_str());
104  next_free_aig_lit_ += 2;
105  }
106 
107 
108  for(AIG2CNF::DepConstIter it = trans_deps.begin(); it != trans_deps.end(); ++it)
109  {
110  if(AIG2CNF::instance().isTrueInTrans() && it->first == 1) // CNF var 1 is constant TRUE
111  cnf_var_to_standalone_aig_var_[1] = 1;
112  else
113  {
114  cnf_var_to_standalone_aig_var_[it->first] = next_free_aig_lit_;
115  next_free_aig_lit_ += 2;
116  }
117  }
118 }

202 void LearningImplExtractor::runLearningQBF(const CNF &win_region, const CNF &neg_win_region)
203 {
204  const vector<int> &ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL);
205  const vector<int> &next = VarManager::instance().getVarsOfType(VarInfo::NEXT_STATE);
206  const vector<int> &tmp = VarManager::instance().getVarsOfType(VarInfo::TMP);
207 
208  CNF neg_rel(neg_win_region);
209  neg_rel.swapPresentToNext();
210  neg_rel.addCNF(win_region);
211  neg_rel.addCNF(AIG2CNF::instance().getTrans());
212  // neg_rel will have a clause that sets the error state variable to false (it must be
213  // part of win_region. However, we also replace it by false so that this variable is gone
214  // from the CNF. This makes one state variable less to consider:
215  neg_rel.setVarValue(VarManager::instance().getPresErrorStateVar(), false);
216 
217  vector<int> ctrl_univ(ctrl);
218  vector<int> ctrl_exist;
219  ctrl_exist.reserve(ctrl.size());
220  for(size_t ctrl_cnt = 0; ctrl_cnt < ctrl.size(); ++ctrl_cnt)
221  {
222  statistics.notifyBeforeCtrlSignal();
223  int current_ctrl = ctrl[ctrl_cnt];
224  Utils::remove(ctrl_univ, current_ctrl);
225  ctrl_exist.push_back(current_ctrl);
226 
227  // Build the quantifier prefix:
228  vector<pair<vector<int>, QBFSolver::Quant> > quant;
229  quant.push_back(make_pair(ip_, QBFSolver::E));
230  quant.push_back(make_pair(ctrl_univ, QBFSolver::A));
231  quant.push_back(make_pair(ctrl_exist, QBFSolver::E));
232  quant.push_back(make_pair(next, QBFSolver::E));
233  quant.push_back(make_pair(tmp, QBFSolver::E));
234 
235  // Build the CNF for computing false-positives (ctrl-signal is 1 but must be 0):
236  CNF check(neg_rel);
237  check.setVarValue(current_ctrl, true);
238 
239  // Build the CNF for generalizing false-positives (ctrl-signal can be 0)
240  CNF gen(neg_rel);
241  gen.setVarValue(current_ctrl, false);
242 
243  // do the learning loop:
244  CNF solution;
245  while(true)
246  {
247  // compute a false-positives (ctrl-signal is 1 but must not be):
248  vector<int> false_pos;
249  statistics.notifyBeforeClauseComp();
250  bool false_pos_exists = qbf_solver_->isSatModel(quant, check, false_pos);
251  statistics.notifyAfterClauseComp();
252  if(!false_pos_exists)
253  break;
254 
255  // generalize the false-positives:
256  // (compute a larger set of situations (including the false-positive) for which
257  // setting the control signal to 0 is allowed:
258  statistics.notifyBeforeClauseMin();
259  vector<int> gen_false_pos(false_pos);
260  for(size_t lit_cnt = 0; lit_cnt < false_pos.size(); ++lit_cnt)
261  {
262  vector<int> tmp(gen_false_pos);
263  Utils::remove(tmp, false_pos[lit_cnt]);
264  CNF current_gen(gen);
265  current_gen.addCube(tmp);
266  if(!qbf_solver_->isSat(quant,current_gen))
267  gen_false_pos = tmp;
268  }
269  statistics.notifyAfterClauseMin(false_pos.size(), gen_false_pos.size());
270  check.addNegCubeAsClause(gen_false_pos);
271  solution.addNegCubeAsClause(gen_false_pos);
272  }
273 
274  // dump solution in AIGER format:
275  addToStandAloneAiger(current_ctrl, solution);
276 
277  // re-substitution:
278  neg_rel.addCNF(makeEq(current_ctrl, solution));
279  statistics.notifyAfterCtrlSignal();
280  }
281 }

367 void LearningImplExtractor::runLearningExp(const CNF &win_region, const CNF &neg_win_region)
368 {
369  const vector<int> &ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL);
370  UnivExpander exp;
371 
372  CNF win(win_region);
373  CNF nxt_win;
374  Utils::compressNextStateCNF(win, nxt_win, false);
375  // win_region will have a clause that sets the error state variable to false. However, we
376  // also replace it by false so that this variable is gone from the CNF. This makes one state
377  // variable less to consider, and ensures that the final result cannot depend on the
378  // artificially introduced fake-error-state-variable.
379  win.setVarValue(VarManager::instance().getPresErrorStateVar(), false);
380 
381  vector<int> ctrl_univ(ctrl);
382  vector<int> ctrl_exist;
383  ctrl_exist.reserve(ctrl.size());
384  CNF existing_solutions;
385  SatSolver *solver = Options::instance().getSATSolverExtr(false, true);
386  for(size_t ctrl_cnt = 0; ctrl_cnt < ctrl.size(); ++ctrl_cnt)
387  {
388  statistics.notifyBeforeCtrlSignal();
389  int current_ctrl = ctrl[ctrl_cnt];
390  Utils::remove(ctrl_univ, current_ctrl);
391  ctrl_exist.push_back(current_ctrl);
392  exp.extrExp(nxt_win, solver, ctrl_univ);
393  solver->incAddCNF(existing_solutions);
394  solver->incAddCNF(win);
395  vector<int> check_ass;
396  check_ass.push_back(current_ctrl);
397  vector<int> gen_ass;
398  gen_ass.push_back(-current_ctrl);
399  vector<int> none;
400 
401  // do the learning loop:
402  CNF solution;
403  while(true)
404  {
405  // compute a false-positives (ctrl-signal is 1 but must not be):
406  vector<int> false_pos;
407  statistics.notifyBeforeClauseComp();
408  bool false_pos_exists = solver->incIsSatModelOrCore(none, check_ass, ip_, false_pos);
409  statistics.notifyAfterClauseComp();
410  if(!false_pos_exists)
411  break;
412 
413  // generalize the false-positives:
414  // (compute a larger set of situations (including the false-positive) for which
415  // setting the control signal to 0 is allowed:
416  vector<int> gen_false_pos;
417  statistics.notifyBeforeClauseMin();
418  bool sat = solver->incIsSatModelOrCore(false_pos, gen_ass, none, gen_false_pos);
419  statistics.notifyAfterClauseMin(false_pos.size(), gen_false_pos.size());
420  MASSERT(!sat, "Impossible");
421  solution.addNegCubeAsClause(gen_false_pos);
422  gen_false_pos.push_back(check_ass[0]);
423  solver->incAddNegCubeAsClause(gen_false_pos);
424  }
425 
426  // dump solution in AIGER format:
427  addToStandAloneAiger(current_ctrl, solution);
428 
429  // re-substitution:
430  existing_solutions.addCNF(makeEq(current_ctrl, solution));
431  statistics.notifyAfterCtrlSignal();
432  }
433 }

436 void LearningImplExtractor::runLearningJiangSAT(const CNF &win_region,
437  const CNF &neg_win_region,
438  bool second_run,
439  bool min_cores)
440 {
441  const vector<int> &ctrl = VarManager::instance().getVarsOfType(VarInfo::CTRL);
442  // setting the minimal core flag to false is faster but leads to larger circuits:
443  SatSolver *check = Options::instance().getSATSolverExtr(false, false);
444  SatSolver *gen = Options::instance().getSATSolverExtr(false, min_cores);
445  CNF trans = AIG2CNF::instance().getTrans();
446  vector<int> none;
447 
448  CNF next_win(win_region);
449  next_win.swapPresentToNext();
450  CNF leave_win(neg_win_region);
451  leave_win.swapPresentToNext();
452  leave_win.addCNF(win_region);
453 
454  vector<CNF> c_eq_impl;
455  c_eq_impl.reserve(ctrl.size());
456  vector<CNF> impl;
457  impl.reserve(ctrl.size());
458 
459  vector<int> dep_vars(ipc_);
460  for(size_t ctrl_cnt = 0; ctrl_cnt < ctrl.size(); ++ctrl_cnt)
461  {
462  statistics.notifyBeforeCtrlSignal();
463  int current_ctrl = ctrl[ctrl_cnt];
464  Utils::remove(dep_vars, current_ctrl);
465 
466  // we need to rename all variables except for the dep_vars:
467  int nr_of_vars = VarManager::instance().getMaxCNFVar() + 1;
468  int next_free_var = nr_of_vars;
469  vector<int> rename_map(nr_of_vars, 0);
470  for(int cnt = 1; cnt < nr_of_vars; ++cnt)
471  rename_map[cnt] = next_free_var++;
472  for(size_t cnt = 0; cnt < dep_vars.size(); ++cnt)
473  rename_map[dep_vars[cnt]] = dep_vars[cnt];
474 
475  CNF canBe1(trans);
476  canBe1.setVarValue(current_ctrl, true);
477  canBe1.addCNF(next_win);
478  canBe1.renameVars(rename_map);
479  CNF mustBe1(trans);
480  mustBe1.setVarValue(current_ctrl, false);
481  mustBe1.addCNF(leave_win);
482  mustBe1.addCNF(canBe1);
483  // just to be sure that the solution does not depend on our artificial state variable
484  // (this could otherwise happen if we do not use minimal cores):
485  mustBe1.setVarValue(VarManager::instance().getPresErrorStateVar(), false);
486 
487  CNF canBe0(trans);
488  canBe0.setVarValue(current_ctrl, false);
489  canBe0.addCNF(next_win);
490  canBe0.renameVars(rename_map);
491  CNF mustBe0(trans);
492  mustBe0.setVarValue(current_ctrl, true);
493  mustBe0.addCNF(leave_win);
494  mustBe0.addCNF(canBe0);
495  // just to be sure that the solution does not depend on our artificial state variable
496  // (this could otherwise happen if we do not use minimal cores):
497  mustBe1.setVarValue(VarManager::instance().getPresErrorStateVar(), false);
498 
499  // now we can do the actual learning loop:
500  check->startIncrementalSession(dep_vars, false);
501  check->incAddCNF(mustBe0);
502  gen->startIncrementalSession(dep_vars, false);
503  gen->incAddCNF(mustBe1);
504  CNF solution;
505  while(true)
506  {
507  vector<int> false_pos;
508  // compute a false-positives (ctrl-signal is 1 but must not be):
509  statistics.notifyBeforeClauseComp();
510  bool false_pos_exists = check->incIsSatModelOrCore(none, dep_vars, false_pos);
511  statistics.notifyAfterClauseComp();
512  if(!false_pos_exists)
513  break;
514  // generalize the false-positives:
515  // (compute a larger set of situations (including the false-positive) for which
516  // setting the control signal to 0 is allowed:
517  vector<int> gen_false_pos;
518  statistics.notifyBeforeClauseMin();
519  //Utils::randomize(false_pos);
