104 void ExtQBFSolver::dumpQBF(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
105  const CNF& cnf,
106  const string &filename)
107 {
108  VarManager& VM = VarManager::instance();
109  ofstream oss(filename.c_str());
110  oss << "p cnf " << VM.getMaxCNFVar() << " " << cnf.getNrOfClauses() << endl;
111  for(size_t cnt = 0; cnt < quantifier_prefix.size(); ++cnt)
112  {
113  if(quantifier_prefix[cnt].second == E)
114  oss << "e ";
115  else
116  oss << "a ";
117  // The following code may look strange, but QBFCert does not like two
118  // subsequent 'a' lists, or 'e' lists, so we merge them:
119  while(1)
120  {
121  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[cnt].first);
122  for(size_t var_cnt = 0; var_cnt < vars.size(); ++var_cnt)
123  oss << vars[var_cnt] << " ";
124  if(cnt == quantifier_prefix.size() - 1)
125  break;
126  if(quantifier_prefix[cnt].second != quantifier_prefix[cnt+1].second)
127  break;
128  ++cnt;
129  }
130  oss << "0" << endl;
131  }
132  oss << cnf.toString();
133  oss.close();
134 }

138  const CNF& cnf,
139  const string &filename)

141  int max_cnf_var = 0;
142  for(size_t level = 0; level < quantifier_prefix.size(); ++level)
143  {
144  for(size_t var_cnt = 0; var_cnt < quantifier_prefix[level].first.size(); ++var_cnt)
145  {
146  if(quantifier_prefix[level].first[var_cnt] > max_cnf_var)
147  max_cnf_var = quantifier_prefix[level].first[var_cnt];
148  }
149  }
150 
151  ofstream oss(filename.c_str());
152  oss << "p cnf " << max_cnf_var << " " << cnf.getNrOfClauses() << endl;
153  for(size_t cnt = 0; cnt < quantifier_prefix.size(); ++cnt)
154  {
155  if(quantifier_prefix[cnt].second == E)
156  oss << "e ";
157  else
158  oss << "a ";
159  // The following code may look strange, but QBFCert does not like two
160  // subsequent 'a' lists, or 'e' lists, so we merge them:

164  for(size_t var_cnt = 0; var_cnt < vars.size(); ++var_cnt)

187 // -------------------------------------------------------------------------------------------
188 bool ExtQBFSolver::parseModel(int ret, vector<int> &model) const
189 {
190  MASSERT(ret == 20 || ret == 10, "Solver terminated with strange exit code.");
191 
192  string answer;
193  bool success = FileUtils::readFile(out_file_name_, answer);
194  MASSERT(success, "Could not read result file.");
195  if(answer.find("s cnf 0") == 0)
196  return false;
197  MASSERT(answer.find("s cnf 1") == 0, "Strange response from Solver.");
198  size_t start = answer.find("V ", 0);
199  while(start != string::npos)
200  {
201  start = start + 2;
202  size_t end = answer.find(" 0", start);
203  MASSERT(end != string::npos, "Strange response from Solver.")
204  istringstream iss(answer.substr(start, end-start));
205  int value;
206  iss >> value;
207  model.push_back(value);
208  start = answer.find("V ", end);
209  }
210  return true;
211 }
