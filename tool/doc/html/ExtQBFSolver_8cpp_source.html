105 void ExtQBFSolver::dumpQBF(const vector<pair<VarInfo::VarKind, Quant> > &quantifier_prefix,
106  const CNF& cnf,
107  const string &filename)
108 {
109  VarManager& VM = VarManager::instance();
110  ofstream oss(filename.c_str());
111  oss << "p cnf " << VM.getMaxCNFVar() << " " << cnf.getNrOfClauses() << endl;
112  for(size_t cnt = 0; cnt < quantifier_prefix.size(); ++cnt)
113  {
114  if(quantifier_prefix[cnt].second == E)
115  oss << "e ";
116  else
117  oss << "a ";
118  // The following code may look strange, but QBFCert does not like two
119  // subsequent 'a' lists, or 'e' lists, so we merge them:
120  while(1)
121  {
122  const vector<int> &vars = VM.getVarsOfType(quantifier_prefix[cnt].first);
123  for(size_t var_cnt = 0; var_cnt < vars.size(); ++var_cnt)
124  oss << vars[var_cnt] << " ";
125  if(cnt == quantifier_prefix.size() - 1)
126  break;
127  if(quantifier_prefix[cnt].second != quantifier_prefix[cnt+1].second)
128  break;
129  ++cnt;
130  }
131  oss << "0" << endl;
132  }
133  oss << cnf.toString();
134  oss.close();
135 }

148  max_cnf_var = quantifier_prefix[level].first[var_cnt];

150  }

152  ofstream oss(filename.c_str());
153  oss << "p cnf " << max_cnf_var << " " << cnf.getNrOfClauses() << endl;
154  for(size_t cnt = 0; cnt < quantifier_prefix.size(); ++cnt)
155  {
156  if(quantifier_prefix[cnt].second == E)
157  oss << "e ";
158  else

190 bool ExtQBFSolver::parseModel(int ret, const vector<int> &get, vector<int> &model) const
191 {
192  if(ret != 10 && ret != 20)
193  throw DemiurgeException("Timeout or crash");
194 
195  string answer;
196  bool success = FileUtils::readFile(out_file_name_, answer);
197  MASSERT(success, "Could not read result file.");
198  if(answer.find("s cnf 0") == 0)
199  return false;
200  MASSERT(answer.find("s cnf 1") == 0, "Strange response from Solver.");
201  size_t start = answer.find("V ", 0);
202  while(start != string::npos)
203  {
204  start = start + 2;
205  size_t end = answer.find(" 0", start);
206  MASSERT(end != string::npos, "Strange response from Solver.")
207  istringstream iss(answer.substr(start, end-start));
208  int value;
209  iss >> value;
210  model.push_back(value);
211  start = answer.find("V ", end);
212  }
213  return true;
214 }
