40 LearnSynthQBFInd::LearnSynthQBFInd(CNFImplExtractor *impl_extractor) :
41  BackEnd(),
42  qbf_solver_(Options::instance().getQBFSolver()),
43  current_state_is_initial_(0),
44  do_reach_check_(false),
45  impl_extractor_(impl_extractor)
46 {
47 
48  if(Options::instance().getBackEndMode() == 3 || Options::instance().getBackEndMode() == 6)
49  do_reach_check_ = true;
50 
51  // build the quantifier prefix for checking for counterexamples:
52  // check_quant_ = exists x,i: forall c: exists x',tmp:
53  if(do_reach_check_)
54  {
55  // if do_reach_check_ = true, the quantifier prefix for checking for counterexamples is:
56  // check_quant_ = exists x*,i*,c*,x,i: forall c: exists x',tmp:
57  // x*,i*,c* are variables of type VarInfo::PREV
58  check_quant_.push_back(make_pair(VarInfo::PREV, QBFSolver::E));
59  }
60  check_quant_.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
61  check_quant_.push_back(make_pair(VarInfo::INPUT, QBFSolver::E));
62  check_quant_.push_back(make_pair(VarInfo::CTRL, QBFSolver::A));
63  check_quant_.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
64  check_quant_.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
65 
66  // build the quantifier prefix for generalizing cubes:
67  // gen_quant_ = exists x*,i*,c*: exists x: forall i: exists c,x',tmp:
68  gen_quant_.push_back(make_pair(VarInfo::PREV, QBFSolver::E));
69  gen_quant_.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
70  gen_quant_.push_back(make_pair(VarInfo::INPUT, QBFSolver::A));
71  gen_quant_.push_back(make_pair(VarInfo::CTRL, QBFSolver::E));
72  gen_quant_.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
73  gen_quant_.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
74 
75  // build previous-state copy of the transition relation. We want to have it enabled
76  // if the current state is initial. The result is stored in prev_trans_or_initial_.
77  AIG2CNF& A2C = AIG2CNF::instance();
78  VarManager &VM = VarManager::instance();
79  current_to_previous_map_.resize(VM.getMaxCNFVar()+1, 0);
80  const vector<int> &s = VM.getVarsOfType(VarInfo::PRES_STATE);
81  const vector<int> &s_next = VM.getVarsOfType(VarInfo::NEXT_STATE);
82  const vector<int> &c = VM.getVarsOfType(VarInfo::CTRL);
83  const vector<int> &i = VM.getVarsOfType(VarInfo::INPUT);
84  vector<int> t = VM.getVarsOfType(VarInfo::TMP);
85 
86  for(size_t v_cnt = 0; v_cnt < s.size(); ++v_cnt)
87  current_to_previous_map_[s[v_cnt]] = VM.createFreshPrevVar();
88  for(size_t v_cnt = 0; v_cnt < c.size(); ++v_cnt)
89  current_to_previous_map_[c[v_cnt]] = VM.createFreshPrevVar();
90  for(size_t v_cnt = 0; v_cnt < i.size(); ++v_cnt)
91  current_to_previous_map_[i[v_cnt]] = VM.createFreshPrevVar();
92  for(size_t v_cnt = 0; v_cnt < t.size(); ++v_cnt)
93  current_to_previous_map_[t[v_cnt]] = VM.createFreshTmpVar();
94  for(size_t v_cnt = 0; v_cnt < s_next.size(); ++v_cnt)
95  current_to_previous_map_[s_next[v_cnt]] = s[v_cnt];
96 
97  current_state_is_initial_ = VM.createFreshTmpVar();
98  list<vector<int> > prev_trans_clauses = A2C.getTrans().getClauses();
99  for(CNF::ClauseIter it = prev_trans_clauses.begin(); it != prev_trans_clauses.end(); ++it)
100  {
101  presentToPrevious(*it);
102  it->push_back(current_state_is_initial_);
103  }
104  prev_trans_or_initial_.swapWith(prev_trans_clauses);
105  // if one of the state variables is true, then current_state_is_initial must be false:
106  for(size_t cnt = 0; cnt < s.size(); ++cnt)
107  prev_trans_or_initial_.add2LitClause(-s[cnt], -current_state_is_initial_);
108 
109  // check_reach_ contains
110  // - the previous transition relation (enabled only if the current state is not initial)
111  // - constraints that say that the previous state is different from the current state
112  // check_reach_ can also be set to TRUE if we do not want to include reachability in the
113  // computation of bad states
114  if(do_reach_check_)
115  {
116  check_reach_.addCNF(prev_trans_or_initial_);
117  vector<int> one_is_diff;
118  one_is_diff.reserve(s.size() + 1);
119  one_is_diff.push_back(current_state_is_initial_);
120  for(size_t cnt = 0; cnt < s.size(); ++cnt)
121  {
122  int diff = VM.createFreshTmpVar();
123  one_is_diff.push_back(diff);
124  int curr = s[cnt];
125  int prev = presentToPrevious(curr);
126  // actually, we only need one direction of the implication:
127  check_reach_.add3LitClause(curr, prev, -diff);
128  check_reach_.add3LitClause(-curr, -prev, -diff);
129  //check_reach_.add3LitClause(-curr, prev, diff);
130  //check_reach_.add3LitClause(curr, -prev, diff);
131  }
132  check_reach_.addClause(one_is_diff);
133  }
134 }

147 bool LearnSynthQBFInd::run()
148 {
149  VarManager::instance().push();
150  L_INF("Starting to compute a winning region ...");
151  statistics_.notifyWinRegStart();
152  bool realizable = computeWinningRegion();
153  statistics_.notifyWinRegEnd();
154  if(!realizable)
155  {
156  L_RES("The specification is unrealizable.");
157  statistics_.logStatistics();
158  return false;
159  }
160  L_RES("The specification is realizable.");
161  if(Options::instance().doRealizabilityOnly())
162  {
163  statistics_.logStatistics();
164  return true;
165  }
166 
167  MASSERT(!do_reach_check_, "Not yet implemented.");
168 
169  L_INF("Starting to extract a circuit ...");
170  impl_extractor_->extractCircuit(winning_region_);
171  L_INF("Synthesis done.");
172  statistics_.logStatistics();
173  impl_extractor_->logStatistics();
174  return true;
175 }

197 bool LearnSynthQBFInd::computeWinningRegionOne()
198 {
199  AIG2CNF& A2C = AIG2CNF::instance();
200  VarManager &VM = VarManager::instance();
201  L_DBG("Nr. of state variables: " << VM.getVarsOfType(VarInfo::PRES_STATE).size());
202  L_DBG("Nr. of uncontrollable inputs: " << VM.getVarsOfType(VarInfo::INPUT).size());
203  L_DBG("Nr. of controllable inputs: " << VM.getVarsOfType(VarInfo::CTRL).size());
204 
205  winning_region_.clear();
206  winning_region_ = A2C.getSafeStates();
207  winning_region_large_.clear();
208  winning_region_large_ = A2C.getSafeStates();
209 
210  // build check_cnf = P(x) AND T(x,i,c,x') AND NOT P(x')
211  recomputeCheckCNF();
212 
213  // build generalize_clause_cnf = P(x*) AND (I(x) OR T(x*,i*,c*,x*)) AND P(x) AND T(x,i,c,x') AND P(x')
214  recomputeGenCNF();
215 
216  vector<int> counterexample;
217  vector<int> blocking_clause;
218  blocking_clause.reserve(VM.getVarsOfType(VarInfo::PRES_STATE).size());
219  bool ce_found = computeCounterexampleQBF(counterexample);
220 
221  size_t count = 0;
222  while(ce_found)
223  {
224  count++;
225  L_DBG("Found bad state (nr " << count << "), excluding it from the winning region.");
226  bool realizable = computeBlockingClause(counterexample, blocking_clause);
227  if(!realizable)
228  return false;
229 
230  // update the CNFs:
231  Utils::debugPrint(blocking_clause, "Blocking clause: ");
232  winning_region_.addClauseAndSimplify(blocking_clause);
233  winning_region_large_.addClauseAndSimplify(blocking_clause);
234  vector<int> prev_blocking_clause(blocking_clause);
235  presentToPrevious(prev_blocking_clause);
236  generalize_clause_cnf_.addClauseAndSimplify(prev_blocking_clause);
237  generalize_clause_cnf_.addClauseAndSimplify(blocking_clause);
238  Utils::swapPresentToNext(blocking_clause);
239  generalize_clause_cnf_.addClauseAndSimplify(blocking_clause);
240  //if(count % 100 == 0)
241  //{
242  // reduceExistingClauses();
243  // recomputeGenCNF(false);
244  //}
245 
246  Utils::compressStateCNF(winning_region_);
247  recomputeCheckCNF(false);
248 
249  if(count % 100 == 0)
250  {
251  L_DBG("Statistics at count=" << count);
252  L_DBG("Winning region clauses: " << winning_region_.getNrOfClauses());
253  L_DBG("Winning region lits: " << winning_region_.getNrOfLits());
254  L_DBG("Winning region full clauses: " << winning_region_large_.getNrOfClauses());
255  L_DBG("Winning region full lits: " << winning_region_large_.getNrOfLits());
256  L_DBG("Check CNF clauses: " << check_cnf_.getNrOfClauses());
257  L_DBG("Check CNF lits: " << check_cnf_.getNrOfLits());
258  L_DBG("Generalize CNF clauses: " << generalize_clause_cnf_.getNrOfClauses());
259  L_DBG("Generalize CNF lits: " << generalize_clause_cnf_.getNrOfLits());
260  if(Logger::instance().isEnabled(Logger::DBG))
261  statistics_.logStatistics();
262  }
263  ce_found = computeCounterexampleQBF(counterexample);
264  }
265  return true;
266 }

274  L_DBG("Nr. of uncontrollable inputs: " << VM.getVarsOfType(VarInfo::INPUT).size());

276

279  winning_region_large_.clear();
280  winning_region_large_ = A2C.getSafeStates();

283  recomputeCheckCNF();
284 
285  // build generalize_clause_cnf = P(x*) AND (I(x) OR T(x*,i*,c*,x*)) AND P(x) AND T(x,i,c,x') AND P(x')

323  L_DBG("Check CNF lits: " << check_cnf_.getNrOfLits());
324  L_DBG("Generalize CNF clauses: " << generalize_clause_cnf_.getNrOfClauses());
325  L_DBG("Generalize CNF lits: " << generalize_clause_cnf_.getNrOfLits());
326  if(Logger::instance().isEnabled(Logger::DBG))
327  statistics_.logStatistics();
328  }
329  ce_found = computeCounterexampleQBF(counterexample);
330  }
331  return true;
332 }

363 // -------------------------------------------------------------------------------------------
364 bool LearnSynthQBFInd::computeAllBlockingClauses(vector<int> &ce,
365  vector<vector<int> > &blocking_clauses)
366 {
367  statistics_.notifyBeforeCubeMin();
368  list<set<int> > must_not_contain_queue;
369  list<vector<int> > gen_ces;
370  vector<int> orig_ce(ce);
371  vector<int> first_ce(orig_ce);
372  generalizeCounterexample(first_ce, false);
373  statistics_.notifyCubeMin(orig_ce.size(), first_ce.size());
374  if(Utils::containsInit(first_ce))
375  return false;
376  blocking_clauses.clear();
377  vector<int> blocking_clause(first_ce);
378  Utils::negateLiterals(blocking_clause);
379  blocking_clauses.push_back(blocking_clause);
380 
381  gen_ces.push_back(first_ce);
382  for(size_t cnt = 0; cnt < first_ce.size(); ++cnt)
383  {
384  set<int> singleton_set;
385  singleton_set.insert(first_ce[cnt]);
386  must_not_contain_queue.push_back(singleton_set);
387  }
388 
389  // we add our blocking_clause to generalize_clause_cnf_ to benefit from it immediately:
390  vector<int> prev_blocking_clause(blocking_clause);
391  presentToPrevious(prev_blocking_clause);
392  generalize_clause_cnf_.addClause(prev_blocking_clause);
393  generalize_clause_cnf_.addClause(blocking_clause);
394  Utils::swapPresentToNext(blocking_clause);
395  generalize_clause_cnf_.addClause(blocking_clause);
396  while(!must_not_contain_queue.empty())
397  {
398  set<int> must_not_contain = must_not_contain_queue.front();
399  must_not_contain_queue.pop_front();
400 
401  // check if we already have a ce in gen_ces which satisfies our needs:
402  bool found = false;
403  for(list<vector<int> >::const_iterator it = gen_ces.begin(); it != gen_ces.end(); ++it)
404  {
405  if(Utils::intersectionEmpty(*it, must_not_contain))
406  {
407  found = true;
408  break;
409  }
410  }
411  if(found)
412  continue;
413 
414  // if not: we compute one
415  vector<int> new_ce(orig_ce);
416  set<int>::const_iterator it = must_not_contain.begin();
417  for( ; it != must_not_contain.end(); ++it)
418  Utils::remove(new_ce, *it);
419  size_t orig_size = new_ce.size();
420  bool is_ok = generalizeCounterexample(new_ce);
421  statistics_.notifyCubeMin(orig_size, new_ce.size());
422  if(is_ok)
423  {
424  if(Utils::containsInit(new_ce))
425  return false;
426  gen_ces.push_back(new_ce);
427  blocking_clause = new_ce;
428  Utils::negateLiterals(blocking_clause);
429  blocking_clauses.push_back(blocking_clause);
430  for(size_t cnt = 0; cnt < new_ce.size(); ++cnt)
431  {
432  set<int> new_must_not_contain(must_not_contain);
433  new_must_not_contain.insert(new_ce[cnt]);
434  must_not_contain_queue.push_back(new_must_not_contain);
435  }
436  // we add our blocking_clause to generalize_clause_cnf_ to benefit from it immediately:
437  vector<int> prev_blocking_clause(blocking_clause);
438  presentToPrevious(prev_blocking_clause);
439  generalize_clause_cnf_.addClause(prev_blocking_clause);
440  generalize_clause_cnf_.addClause(blocking_clause);
441  Utils::swapPresentToNext(blocking_clause);
442  generalize_clause_cnf_.addClause(blocking_clause);
443  }
444  }
445 
446  statistics_.notifyAfterCubeMin();
447 
448  return true;
449 }

452 void LearnSynthQBFInd::reduceExistingClauses()
453 {
454  statistics_.notifyBeforeCubeMin();
455 
456  size_t or_cl = winning_region_.getNrOfClauses();
457  size_t or_lits = winning_region_.getNrOfLits();
458  const list<vector<int> > orig_clauses = winning_region_.getClauses();
459  for(CNF::ClauseConstIter it = orig_clauses.begin(); it != orig_clauses.end(); ++it)
460  {
461  vector<int> clause(*it);
462  for(size_t cnt = 0; cnt < it->size(); ++cnt)
463  {
464  vector<int> tmp(clause);
465  Utils::remove(tmp, (*it)[cnt]);
466  CNF tmp_cnf(generalize_clause_cnf_);
467  tmp_cnf.addNegClauseAsCube(tmp);
468  vector<int> prev_tmp;
469  for(size_t i = 0; i < tmp.size(); ++i)
470  prev_tmp.push_back(presentToPrevious(tmp[i]));
471  prev_tmp.push_back(-current_state_is_initial_);
472  tmp_cnf.addClause(prev_tmp);
473  if(!qbf_solver_->isSat(gen_quant_, tmp_cnf))
474  clause = tmp;
475  }
476  if(it->size() > clause.size())
477  {
478  winning_region_.addClauseAndSimplify(clause);
479  winning_region_large_.addClauseAndSimplify(clause);
480  }
481  }
482  size_t new_cl = winning_region_.getNrOfClauses();
483  size_t new_lits = winning_region_.getNrOfLits();;
484  L_DBG("reduceExistingClauses(): " << or_cl << " --> " << new_cl << " clauses.");
485  L_DBG("reduceExistingClauses(): " << or_lits << " --> " << new_lits << " literals.");
486  statistics_.notifyAfterCubeMin(0, 0);
487 }

549 // -------------------------------------------------------------------------------------------
550 void LearnSynthQBFInd::restrictToStates(vector<int> &vec) const
551 {
552  for(size_t cnt = 0; cnt < vec.size(); ++cnt)
553  {
554  int var = (vec[cnt] < 0) ? -vec[cnt] : vec[cnt];
555  // We can always drop the error-bit of the state space (because we can be sure that all
556  // states with error=1 have already been removed from the winning region). This saves
557  // one iteration during the generalization.
558  if(var == VarManager::instance().getPresErrorStateVar() ||
559  VarManager::instance().getInfo(var).getKind() != VarInfo::PRES_STATE)
560  {
561  vec[cnt] = vec[vec.size() - 1];
562  vec.pop_back();
563  cnt--;
564  }
565  }
566 }

569 bool LearnSynthQBFInd::generalizeCounterexample(vector<int> &ce, bool check_sat) const
570 {
571  if(check_sat)
572  {
573  CNF tmp_cnf1(generalize_clause_cnf_);
574  tmp_cnf1.addCube(ce);
575  if(qbf_solver_->isSat(gen_quant_, tmp_cnf1))
576  return false;
577  }
578  vector<int> gen_ce(ce);
579  for(size_t cnt = 0; cnt < ce.size(); ++cnt)
580  {
581  vector<int> tmp = gen_ce;
582  Utils::remove(tmp, ce[cnt]);
583  CNF tmp_cnf(generalize_clause_cnf_);
584  tmp_cnf.addCube(tmp);
585  tmp_cnf.addNegCubeAsClause(gen_ce);
586  vector<int> next_gen_ce(gen_ce);
587  Utils::swapPresentToNext(next_gen_ce);
588  tmp_cnf.addNegCubeAsClause(next_gen_ce);
589  vector<int> prev_tmp(tmp);
590  presentToPrevious(prev_tmp);
591  prev_tmp.push_back(-current_state_is_initial_);
592  tmp_cnf.addNegCubeAsClause(prev_tmp);
593  if(!qbf_solver_->isSat(gen_quant_, tmp_cnf))
594  gen_ce = tmp;
595  }
596  ce = gen_ce;
597  Utils::sort(ce);
598  return true;
599 }

633 void LearnSynthQBFInd::debugCheckWinRegReach() const
634 {
635 #ifndef NDEBUG
636  L_DBG("Checking winning region for correctness ...");
637  LingelingApi sat_solver;
638  CNF check(winning_region_);
639  check.addCNF(AIG2CNF::instance().getInitial());
640  MASSERT(sat_solver.isSat(check), "Initial state is not winning.");
641 
642  // All states of the winning region must be safe
643  check.clear();
644  check.addCNF(winning_region_);
645  check.addCNF(AIG2CNF::instance().getUnsafeStates());
646  MASSERT(!sat_solver.isSat(check), "Winning region is not safe.");
647 
648  // for all potentially reachable state in the winning region, we can enforce to stay in the
649  // winning region:
650  // exists x*,i*,c*: exists x,i: forall c: exists x':
651  // (I(x) | (W(x*) & T(x*,i*,c*,x))) & W(x) & T(x,i,c,x') & !W(x')
652  // must be unsat
653  check.clear();
654  check.addCNF(winning_region_);
655  check.swapPresentToNext();
656  check.negate();
657  check.addCNF(winning_region_);
658  check.addCNF(AIG2CNF::instance().getTrans());
659  check.addCNF(prev_trans_or_initial_);
660  CNF prev_win(winning_region_);
661  presentToPrevious(prev_win);
662  check.addCNF(prev_win);
663 
664  vector<pair<VarInfo::VarKind, QBFSolver::Quant> > check_quant;
665  check_quant.push_back(make_pair(VarInfo::PREV, QBFSolver::E));
666  check_quant.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
667  check_quant.push_back(make_pair(VarInfo::INPUT, QBFSolver::E));
668  check_quant.push_back(make_pair(VarInfo::CTRL, QBFSolver::A));
669  check_quant.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
670  check_quant.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
671  MASSERT(!qbf_solver_->isSat(check_quant, check), "Winning region is not inductive.");
672 #endif
673 }
