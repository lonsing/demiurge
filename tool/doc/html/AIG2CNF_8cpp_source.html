50 void AIG2CNF::initFromAig(aiger *aig)
51 {
52  clear();
53  VarManager &VM = VarManager::instance();
54 
55  // Step 1:
56  // Let's find out which AIG literals could potentially be referenced:
57  vector<bool> refs(2*(aig->maxvar+1), 0);
58  refs[aig->outputs[0].lit] = true;
59  refs[aiger_not(aig->outputs[0].lit)] = true;
60  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
61  {
62  refs[aig->latches[cnt].next] = true;
63  refs[aiger_not(aig->latches[cnt].next)] = true;
64  refs[aig->latches[cnt].lit] = true;
65  refs[aiger_not(aig->latches[cnt].lit)] = true;
66  }
67 
68  bool changed = true;
69  while(changed)
70  {
71  changed = false;
72  for(unsigned cnt = aig->num_ands; 0 < cnt--;)
73  {
74  unsigned lit = aig->ands[cnt].lhs;
75  if (refs[lit])
76  {
77  if(!refs[aig->ands[cnt].rhs0] || !refs[aig->ands[cnt].rhs1])
78  changed = true;
79  refs[aig->ands[cnt].rhs0] = true;
80  refs[aig->ands[cnt].rhs1] = true;
81  }
82  if (refs[aiger_not (lit)])
83  {
84  if(!refs[aiger_not(aig->ands[cnt].rhs0)] || !refs[aiger_not(aig->ands[cnt].rhs1)])
85  changed = true;
86  refs[aiger_not(aig->ands[cnt].rhs0)] = true;
87  refs[aiger_not(aig->ands[cnt].rhs1)] = true;
88  }
89  }
90  }
91 
92 
93  // Step 2:
94  // initialize the VarManager:
95  VM.initFromAig(aig, refs);
96 
97  // Step 3:
98  // now we create the CNF for the transition relation:
99  if (refs[0] || refs[1])
100  {
101  trans_.add1LitClause(VM.aigLitToCnfLit(1));
102  trans_tmp_deps_[VM.aigLitToCnfLit(1)] = set<VarInfo>();
103  }
104  // Step 3a:
105  // clauses defining the outputs of the AND gates:
106  for (unsigned i = 0; i < aig->num_ands; i++)
107  {
108  unsigned out_aig_lit = aig->ands[i].lhs;
109  int out_cnf_lit = VarManager::instance().aigLitToCnfLit(out_aig_lit);
110  int rhs1_cnf_lit = VarManager::instance().aigLitToCnfLit(aig->ands[i].rhs1);
111  int rhs0_cnf_lit = VarManager::instance().aigLitToCnfLit(aig->ands[i].rhs0);
112  if (refs[out_aig_lit])
113  {
114  // (lhs --> rhs1) AND (lhs --> rhs0)
115  trans_.add2LitClause(-out_cnf_lit, rhs1_cnf_lit);
116  trans_.add2LitClause(-out_cnf_lit, rhs0_cnf_lit);
117  }
118  if (refs[out_aig_lit+1])
119  {
120  // (!lhs --> (!rhs1 OR !rhs0)
121  trans_.add3LitClause(out_cnf_lit, -rhs1_cnf_lit, -rhs0_cnf_lit);
122  }
123  set<VarInfo> deps;
124  deps.insert(VM.getInfo((rhs1_cnf_lit < 0) ? -rhs1_cnf_lit : rhs1_cnf_lit));
125  deps.insert(VM.getInfo((rhs0_cnf_lit < 0) ? -rhs0_cnf_lit : rhs0_cnf_lit));
126  trans_tmp_deps_[out_cnf_lit] = deps;
127 
128  }
129  trans_eq_t_ = trans_;
130 
131  // Step 3b:
132  // For trans_, we add clauses saying that the inputs to the latches form the
133  // next states:
134  const vector<int> &next_state_vars = VM.getVarsOfType(VarInfo::NEXT_STATE);
135  // state variable 0 contains the error bit, so we set
136  // next_state_vars[0] <-> error_lit:
137  int error_lit = VM.aigLitToCnfLit(aig->outputs[0].lit);
138  trans_.add2LitClause(next_state_vars[0], -error_lit);
139  trans_.add2LitClause(-next_state_vars[0], error_lit);
140 
141  // we also create equality constraints for the other state bits:
142  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
143  {
144  int next_state_lit = VM.aigLitToCnfLit(aig->latches[cnt].next);
145  int x_prime_lit = next_state_vars[cnt+1];
146  trans_.add2LitClause(next_state_lit, -x_prime_lit);
147  trans_.add2LitClause(-next_state_lit, x_prime_lit);
148  }
149 
150  // Step 3c
151  // For trans_eq_t_, we add clauses saying that the inputs to the latches form the
152  // next states if and only if t_ is true
153  t_ = VM.createFreshTmpVar("t");
154  vector<int> all_equal_clause(aig->num_latches + 2, 0);
155  all_equal_clause[0] = t_;
156  // state variable 0 contains the error bit, so we set
157  // err_eq <-> (next_state_vars[0] <-> error_lit):
158  int err_eq = VM.createFreshTmpVar("x_eq_x'");
159  trans_eq_t_.add3LitClause(error_lit, next_state_vars[0], err_eq);
160  trans_eq_t_.add3LitClause(error_lit, -next_state_vars[0], -err_eq);
161  trans_eq_t_.add3LitClause(-error_lit, next_state_vars[0], -err_eq);
162  trans_eq_t_.add3LitClause(-error_lit, -next_state_vars[0], err_eq);
163  all_equal_clause[1] = -err_eq;
164  // t is false if err_eq is false
165  trans_eq_t_.add2LitClause(err_eq, -t_);
166 
167  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
168  {
169  int next_state_lit = VM.aigLitToCnfLit(aig->latches[cnt].next);
170  int x_prime_lit = next_state_vars[cnt+1];
171  int equal = VM.createFreshTmpVar("x_eq_x'");
172  all_equal_clause[cnt+2] = -equal;
173  // equal is true iff next_state_lit is equal to x_prime_lit
174  trans_eq_t_.add3LitClause(next_state_lit, x_prime_lit, equal);
175  trans_eq_t_.add3LitClause(next_state_lit, -x_prime_lit, -equal);
176  trans_eq_t_.add3LitClause(-next_state_lit, x_prime_lit, -equal);
177  trans_eq_t_.add3LitClause(-next_state_lit, -x_prime_lit, equal);
178  // t is false if one of the 'equal'-literals is false
179  trans_eq_t_.add2LitClause(equal, -t_);
180  }
181  // t is true all the 'equal'-literals are true
182  trans_eq_t_.addClause(all_equal_clause);
183 
184 
185  // Step 4:
186  // Initialize the CNFs characterizing the good/bad/initial states:
187  safe_.add1LitClause(-VM.getPresErrorStateVar());
188  unsafe_.add1LitClause(VM.getPresErrorStateVar());
189  next_safe_.add1LitClause(-VM.getNextErrorStateVar());
190  next_unsafe_.add1LitClause(VM.getNextErrorStateVar());
191  const vector<int> &state_vars = VM.getVarsOfType(VarInfo::PRES_STATE);
192  for(size_t cnt = 0; cnt < state_vars.size(); ++cnt)
193  initial_.add1LitClause(-state_vars[cnt]);
194 }

265 const map<int, set<VarInfo> >& AIG2CNF::getTmpDepsTrans()
266 {
267  // we compute the transitive dependencies only lazily:
268  typedef set<VarInfo>::const_iterator SetConstIter;
269  while(trans_tmp_deps_trans_.size() != trans_tmp_deps_.size())
270  {
271  for(DepConstIter it = trans_tmp_deps_.begin(); it != trans_tmp_deps_.end(); ++it)
272  {
273  if(trans_tmp_deps_trans_.count(it->first)) // already done
274  continue;
275  bool abort = false;
276  set<VarInfo> trans_dep;
277  for(SetConstIter it2 = it->second.begin(); it2 != it->second.end(); ++it2)
278  {
279  if(it2->getKind() == VarInfo::TMP)
280  {
281  DepConstIter d = trans_tmp_deps_trans_.find(it2->getLitInCNF());
282  if(d == trans_tmp_deps_trans_.end())
283  {
284  abort = true;
285  break;
286  }
287  else
288  {
289  trans_dep.insert(d->second.begin(), d->second.end());
290  }
291  }
292  else
293  trans_dep.insert(*it2);
294  }
295  if(!abort)
296  trans_tmp_deps_trans_[it->first] = trans_dep;
297  }
298  }
299  return trans_tmp_deps_trans_;
300 }
