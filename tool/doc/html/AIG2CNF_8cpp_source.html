51 void AIG2CNF::initFromAig(aiger *aig)
52 {
53  clear();
54  VarManager &VM = VarManager::instance();
55 
56  // Step 1:
57  // Let's find out which AIG literals could potentially be referenced:
58  vector<bool> refs(2*(aig->maxvar+1), 0);
59  refs[aig->outputs[0].lit] = true;
60  refs[aiger_not(aig->outputs[0].lit)] = true;
61  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
62  {
63  refs[aig->latches[cnt].next] = true;
64  refs[aiger_not(aig->latches[cnt].next)] = true;
65  refs[aig->latches[cnt].lit] = true;
66  refs[aiger_not(aig->latches[cnt].lit)] = true;
67  }
68  for(unsigned cnt = 0; cnt < aig->num_inputs; ++cnt)
69  {
70  string aig_name;
71  if(aig->inputs[cnt].name != NULL)
72  aig_name = aig->inputs[cnt].name;
73  string aig_name_lower = StringUtils::toLowerCase(aig_name);
74  if(aig_name_lower.find("controllable_") == 0)
75  {
76  refs[aig->inputs[cnt].lit] = true;
77  refs[aiger_not(aig->inputs[cnt].lit)] = true;
78  }
79  }
80 
81  bool changed = true;
82  while(changed)
83  {
84  changed = false;
85  for(unsigned cnt = aig->num_ands; 0 < cnt--;)
86  {
87  unsigned lit = aig->ands[cnt].lhs;
88  if (refs[lit])
89  {
90  if(!refs[aig->ands[cnt].rhs0] || !refs[aig->ands[cnt].rhs1])
91  changed = true;
92  refs[aig->ands[cnt].rhs0] = true;
93  refs[aig->ands[cnt].rhs1] = true;
94  }
95  if (refs[aiger_not (lit)])
96  {
97  if(!refs[aiger_not(aig->ands[cnt].rhs0)] || !refs[aiger_not(aig->ands[cnt].rhs1)])
98  changed = true;
99  refs[aiger_not(aig->ands[cnt].rhs0)] = true;
100  refs[aiger_not(aig->ands[cnt].rhs1)] = true;
101  }
102  }
103  }
104 
105 
106  // Step 2:
107  // initialize the VarManager:
108  VM.initFromAig(aig, refs);
109 
110  // Step 3:
111  // now we create the CNF for the transition relation:
112  if (refs[0] || refs[1])
113  {
114  true_in_trans_ = true;
115  trans_.add1LitClause(VM.aigLitToCnfLit(1));
116  trans_tmp_deps_[VM.aigLitToCnfLit(1)] = set<VarInfo>();
117  }
118  // Step 3a:
119  // clauses defining the outputs of the AND gates:
120  for (unsigned i = 0; i < aig->num_ands; i++)
121  {
122  unsigned out_aig_lit = aig->ands[i].lhs;
123  if (refs[out_aig_lit] || refs[out_aig_lit+1])
124  {
125  int out_cnf_lit = VarManager::instance().aigLitToCnfLit(out_aig_lit);
126  int rhs1_cnf_lit = VarManager::instance().aigLitToCnfLit(aig->ands[i].rhs1);
127  int rhs0_cnf_lit = VarManager::instance().aigLitToCnfLit(aig->ands[i].rhs0);
128 
129  // (lhs --> rhs1) AND (lhs --> rhs0)
130  trans_.add2LitClause(-out_cnf_lit, rhs1_cnf_lit);
131  trans_.add2LitClause(-out_cnf_lit, rhs0_cnf_lit);
132  // (!lhs --> (!rhs1 OR !rhs0)
133  trans_.add3LitClause(out_cnf_lit, -rhs1_cnf_lit, -rhs0_cnf_lit);
134 
135  set<VarInfo> deps;
136  deps.insert(VM.getInfo((rhs1_cnf_lit < 0) ? -rhs1_cnf_lit : rhs1_cnf_lit));
137  deps.insert(VM.getInfo((rhs0_cnf_lit < 0) ? -rhs0_cnf_lit : rhs0_cnf_lit));
138  trans_tmp_deps_[out_cnf_lit] = deps;
139  }
140  }
141  trans_eq_t_ = trans_;
142 
143  // Step 3b:
144  // For trans_, we add clauses saying that the inputs to the latches form the
145  // next states:
146  const vector<int> &next_state_vars = VM.getVarsOfType(VarInfo::NEXT_STATE);
147  // state variable 0 contains the error bit, so we set
148  // next_state_vars[0] <-> error_lit:
149  int error_lit = VM.aigLitToCnfLit(aig->outputs[0].lit);
150  trans_.add2LitClause(-error_lit, next_state_vars[0]);
151  trans_.add2LitClause(error_lit, -next_state_vars[0]);
152 
153 
154  // we also create equality constraints for the other state bits:
155  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
156  {
157  int next_state_lit = VM.aigLitToCnfLit(aig->latches[cnt].next);
158  int x_prime_lit = next_state_vars[cnt+1];
159  trans_.add2LitClause(-next_state_lit, x_prime_lit);
160  trans_.add2LitClause(next_state_lit, -x_prime_lit);
161  }
162 
163  // Step 3c
164  // For trans_eq_t_, we add clauses saying that the inputs to the latches form the
165  // next states if and only if t_ is true
166  t_ = VM.createFreshTmpVar("t");
167  vector<int> all_equal_clause(aig->num_latches + 2, 0);
168  all_equal_clause[0] = t_;
169  // state variable 0 contains the error bit, so we set
170  // err_eq <-> (next_state_vars[0] <-> error_lit):
171  int err_eq = VM.createFreshTmpVar("x_eq_x'");
172  trans_eq_t_.add3LitClause(error_lit, next_state_vars[0], err_eq);
173  trans_eq_t_.add3LitClause(error_lit, -next_state_vars[0], -err_eq);
174  trans_eq_t_.add3LitClause(-error_lit, next_state_vars[0], -err_eq);
175  trans_eq_t_.add3LitClause(-error_lit, -next_state_vars[0], err_eq);
176  all_equal_clause[1] = -err_eq;
177  // t is false if err_eq is false
178  trans_eq_t_.add2LitClause(err_eq, -t_);
179 
180  for(unsigned cnt = 0; cnt < aig->num_latches; ++cnt)
181  {
182  int next_state_lit = VM.aigLitToCnfLit(aig->latches[cnt].next);
183  int x_prime_lit = next_state_vars[cnt+1];
184  int equal = VM.createFreshTmpVar("x_eq_x'");
185  all_equal_clause[cnt+2] = -equal;
186  // equal is true iff next_state_lit is equal to x_prime_lit
187  trans_eq_t_.add3LitClause(next_state_lit, x_prime_lit, equal);
188  trans_eq_t_.add3LitClause(next_state_lit, -x_prime_lit, -equal);
189  trans_eq_t_.add3LitClause(-next_state_lit, x_prime_lit, -equal);
190  trans_eq_t_.add3LitClause(-next_state_lit, -x_prime_lit, equal);
191  // t is false if one of the 'equal'-literals is false
192  trans_eq_t_.add2LitClause(equal, -t_);
193  }
194  // t is true all the 'equal'-literals are true
195  trans_eq_t_.addClause(all_equal_clause);
196 
197 
198  // Step 4:
199  // Initialize the CNFs characterizing the good/bad/initial states:
200  safe_.add1LitClause(-VM.getPresErrorStateVar());
201  unsafe_.add1LitClause(VM.getPresErrorStateVar());
202  next_safe_.add1LitClause(-VM.getNextErrorStateVar());
203  next_unsafe_.add1LitClause(VM.getNextErrorStateVar());
204  const vector<int> &state_vars = VM.getVarsOfType(VarInfo::PRES_STATE);
205  for(size_t cnt = 0; cnt < state_vars.size(); ++cnt)
206  initial_.add1LitClause(-state_vars[cnt]);
207 }

278 const map<int, set<VarInfo> >& AIG2CNF::getTmpDepsTrans()
279 {
280  // we compute the transitive dependencies only lazily:
281  typedef set<VarInfo>::const_iterator SetConstIter;
282  while(trans_tmp_deps_trans_.size() != trans_tmp_deps_.size())
283  {
284  for(DepConstIter it = trans_tmp_deps_.begin(); it != trans_tmp_deps_.end(); ++it)
285  {
286  if(trans_tmp_deps_trans_.count(it->first)) // already done
287  continue;
288  bool abort = false;
289  set<VarInfo> trans_dep;
290  for(SetConstIter it2 = it->second.begin(); it2 != it->second.end(); ++it2)
291  {
292  if(it2->getKind() == VarInfo::TMP)
293  {
294  DepConstIter d = trans_tmp_deps_trans_.find(it2->getLitInCNF());
295  if(d == trans_tmp_deps_trans_.end())
296  {
297  abort = true;
298  break;
299  }
300  else
301  {
302  trans_dep.insert(d->second.begin(), d->second.end());
303  }
304  }
305  else
306  trans_dep.insert(*it2);
307  }
308  if(!abort)
309  trans_tmp_deps_trans_[it->first] = trans_dep;
310  }
311  }
312  return trans_tmp_deps_trans_;
313 }
