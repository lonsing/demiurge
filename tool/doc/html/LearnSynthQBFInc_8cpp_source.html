70 {
71  VarManager::instance().push();
72  //L_DBG(VarManager::instance().toString());
73  //L_DBG("TRANS:");
74  //L_DBG(AIG2CNF::instance().getTrans().toString());
75  L_INF("Starting to compute a winning region ...");
76  statistics_.notifyWinRegStart();
77  bool realizable = computeWinningRegion();
78  statistics_.notifyWinRegEnd();
79  if(!realizable)
80  {
81  L_RES("The specification is unrealizable.");
82  statistics_.logStatistics();
83  return false;
84  }
85  L_RES("The specification is realizable.");
86  Utils::debugCheckWinReg(winning_region_);
87  if(Options::instance().doRealizabilityOnly())
88  {
89  statistics_.logStatistics();
90  return true;
91  }
92 
93  L_INF("Starting to extract a circuit ...");
94  statistics_.notifyRelDetStart();
95  QBFCertImplExtractor extractor;
96  extractor.extractCircuit(winning_region_);
97  statistics_.notifyRelDetEnd();
98  L_INF("Synthesis done.");
99  statistics_.logStatistics();

113 {
114  AIG2CNF& A2C = AIG2CNF::instance();
115  VarManager &VM = VarManager::instance();
116  L_DBG("Nr. of state variables: " << VM.getVarsOfType(VarInfo::PRES_STATE).size());
117  L_DBG("Nr. of uncontrollable inputs: " << VM.getVarsOfType(VarInfo::INPUT).size());
118  L_DBG("Nr. of controllable inputs: " << VM.getVarsOfType(VarInfo::CTRL).size());
119 
120  winning_region_.clear();
121  winning_region_ = A2C.getSafeStates();
122  winning_region_large_.clear();
123  winning_region_large_ = A2C.getSafeStates();
124 
125  bool precise = true;
126  // build check_cnf = P(x) AND T(x,i,c,x') AND NOT P(x')
127  recomputeCheckCNF();
128  solver_check_.startIncrementalSession(check_quant_);
129  solver_check_.incAddCNF(check_cnf_);
130  // build generalize_clause_cnf = P(x) AND T(x,i,c,x') AND P(x')
131  recomputeGenCNF();
132  solver_gen_.startIncrementalSession(gen_quant_);
133  solver_gen_.incAddCNF(generalize_clause_cnf_);
134 
135  vector<int> counterexample;
136  vector<int> blocking_clause;
137  blocking_clause.reserve(VM.getVarsOfType(VarInfo::PRES_STATE).size());
138 
139  size_t count = 0;
140  while(true)
141  {
142  count++;
143  bool ce_found = computeCounterexampleQBF(counterexample);
144  if(!ce_found)
145  {
146  if(precise)
147  return true;
148  Utils::compressStateCNF(winning_region_);
149  recomputeCheckCNF();
150  solver_check_.startIncrementalSession(check_quant_);
151  solver_check_.incAddCNF(check_cnf_);
152  recomputeGenCNF();
153  solver_gen_.startIncrementalSession(gen_quant_);
154  solver_gen_.incAddCNF(generalize_clause_cnf_);
155  precise = true;
156  continue;
157  }
158 
159  L_DBG("Found bad state (nr " << count << "), excluding it from the winning region.");
160  bool realizable = computeBlockingClause(counterexample, blocking_clause);
161  if(!realizable)
162  return false;
163 
164  // update the CNFs:
165  Utils::debugPrint(blocking_clause, "Blocking clause: ");
166  winning_region_.addClauseAndSimplify(blocking_clause);
167  winning_region_large_.addClauseAndSimplify(blocking_clause);
168  solver_check_.incAddClause(blocking_clause);
169  //solver_gen_.incAddClause(blocking_clause);
170  Utils::compressStateCNF(winning_region_);
171  recomputeGenCNF();
172  solver_gen_.startIncrementalSession(gen_quant_);
173  solver_gen_.incAddCNF(generalize_clause_cnf_);
174  precise = false;
175 
176  if(count % 100 == 0)
177  {
178  L_DBG("Statistics at count=" << count);
179  L_DBG("Winning region clauses: " << winning_region_.getNrOfClauses());
180  L_DBG("Winning region lits: " << winning_region_.getNrOfLits());
181  L_DBG("Winning region full clauses: " << winning_region_large_.getNrOfClauses());
182  L_DBG("Winning region full lits: " << winning_region_large_.getNrOfLits());
183  if(Logger::instance().isEnabled(Logger::DBG))
184  statistics_.logStatistics();
185  }
186  }
187  //reduceExistingClauses();
188  return true;
189 }

289 bool LearnSynthQBFInc::computeBlockingClause(vector<int> &ce, vector<int> &blocking_clause)
290 {
291  statistics_.notifyBeforeCubeMin();
292  Utils::randomize(ce);
293  if(Utils::containsInit(ce))
294  return false;
295  size_t orig_size = ce.size();
296  generalizeCounterexample(ce);
297  if(Utils::containsInit(ce))
298  return false;
299  // BEGIN minimize further
300  // It pays of to add the blocking clause to generalize_clause_cnf_ and do a second
301  // iteration:
302  blocking_clause = ce;
303  Utils::negateLiterals(blocking_clause);
304  generalize_clause_cnf_.addClause(blocking_clause);
305  Utils::swapPresentToNext(blocking_clause);
306  generalize_clause_cnf_.addClause(blocking_clause);
307  solver_gen_.startIncrementalSession(gen_quant_);
308  solver_gen_.incAddCNF(generalize_clause_cnf_);
309  generalizeCounterexample(ce);
310  if(Utils::containsInit(ce))
311  return false;
312  // END minimize further
313  blocking_clause = ce;
314  Utils::negateLiterals(blocking_clause);
315  L_DBG("computeBlockingClause(): " << orig_size << " --> " << blocking_clause.size());
316  statistics_.notifyAfterCubeMin(orig_size, blocking_clause.size());
317  return true;
318 }

321 bool LearnSynthQBFInc::computeAllBlockingClauses(vector<int> &ce,
322  vector<vector<int> > &blocking_clauses)
323 {
324  statistics_.notifyBeforeCubeMin();
325  list<set<int> > must_not_contain_queue;
326  list<vector<int> > gen_ces;
327  vector<int> orig_ce(ce);
328  vector<int> first_ce(orig_ce);
329  generalizeCounterexample(first_ce);
330  statistics_.notifyCubeMin(orig_ce.size(), first_ce.size());
331  if(Utils::containsInit(first_ce))
332  return false;
333  blocking_clauses.clear();
334  vector<int> blocking_clause(first_ce);
335  Utils::negateLiterals(blocking_clause);
336  blocking_clauses.push_back(blocking_clause);
337 
338  gen_ces.push_back(first_ce);
339  for(size_t cnt = 0; cnt < first_ce.size(); ++cnt)
340  {
341  set<int> singleton_set;
342  singleton_set.insert(first_ce[cnt]);
343  must_not_contain_queue.push_back(singleton_set);
344  }
345 
346  while(!must_not_contain_queue.empty())
347  {
348  set<int> must_not_contain = must_not_contain_queue.front();
349  must_not_contain_queue.pop_front();
350 
351  // check if we already have a ce in gen_ces which satisfies our needs:
352  bool found = false;
353  for(list<vector<int> >::const_iterator it = gen_ces.begin(); it != gen_ces.end(); ++it)
354  {
355  if(Utils::intersectionEmpty(*it, must_not_contain))
356  {
357  found = true;
358  break;
359  }
360  }
361  if(found)
362  continue;
363 
364  // if not: we compute one
365  vector<int> new_ce(orig_ce);
366  set<int>::const_iterator it = must_not_contain.begin();
367  for(; it != must_not_contain.end(); ++it)
368  Utils::remove(new_ce, *it);
369  size_t orig_size = new_ce.size();
370  bool is_ok = generalizeCounterexample(new_ce);
371  statistics_.notifyCubeMin(orig_size, new_ce.size());
372  if(is_ok)
373  {
374  if(Utils::containsInit(new_ce))
375  return false;
376  gen_ces.push_back(new_ce);
377  blocking_clause = new_ce;
378  Utils::negateLiterals(blocking_clause);
379  blocking_clauses.push_back(blocking_clause);
380  for(size_t cnt = 0; cnt < new_ce.size(); ++cnt)
381  {
382  set<int> new_must_not_contain(must_not_contain);
383  new_must_not_contain.insert(new_ce[cnt]);
384  must_not_contain_queue.push_back(new_must_not_contain);
385  }
386  }
387  }
388 
389  // BEGIN minimize further
390  // This is all we could do with the current incremental session of solver_gen_.
391  // However, in order to reduce the size of the clauses further, we can add all we have
392  // learned meanwhile to solver_gen_ and see if we can reduce the clauses
393  // further:
394  for(size_t cl_cnt = 0; cl_cnt < blocking_clauses.size(); ++cl_cnt)
395  {
396  vector<int> new_blocking_clause(blocking_clauses[cl_cnt]);
397  generalize_clause_cnf_.addClause(new_blocking_clause);
398  Utils::swapPresentToNext(new_blocking_clause);
399  generalize_clause_cnf_.addClause(new_blocking_clause);
400  }
401  solver_gen_.startIncrementalSession(gen_quant_);
402  solver_gen_.incAddCNF(generalize_clause_cnf_);
403  for(size_t cl_cnt = 0; cl_cnt < blocking_clauses.size(); ++cl_cnt)
404  {
405  vector<int> min_ce(blocking_clauses[cl_cnt]);
406  Utils::negateLiterals(min_ce);
407  size_t size_before_min = min_ce.size();
408  generalizeCounterexample(min_ce);
409  if(min_ce.size() < size_before_min)
410  {
411  Utils::negateLiterals(min_ce);
412  blocking_clauses[cl_cnt] = min_ce;
413  }
414  }
415  // END minimize further
416 
417 
418  statistics_.notifyAfterCubeMin();
419 
420  return true;
421 }

424 void LearnSynthQBFInc::reduceExistingClauses()
425 {
426  statistics_.notifyBeforeCubeMin();
427 
428  size_t or_cl = winning_region_.getNrOfClauses();
429  size_t or_lits = winning_region_.getNrOfLits();
430  const list<vector<int> > orig_clauses = winning_region_.getClauses();
431  for(CNF::ClauseConstIter it = orig_clauses.begin(); it != orig_clauses.end(); ++it)
432  {
433  vector<int> counterexample(*it);
434  Utils::negateLiterals(counterexample);
435  size_t size_before = counterexample.size();
436  generalizeCounterexample(counterexample);
437  if(counterexample.size() < size_before)
438  {
439  vector<int> clause(counterexample);
440  Utils::negateLiterals(clause);
441  winning_region_.addClauseAndSimplify(clause);
442  winning_region_large_.addClauseAndSimplify(clause);
443  generalize_clause_cnf_.addClause(clause);
444  Utils::swapPresentToNext(clause);
445  generalize_clause_cnf_.addClause(clause);
446  solver_gen_.startIncrementalSession(gen_quant_);
447  solver_gen_.incAddCNF(generalize_clause_cnf_);
448  }
449  }
450  size_t new_cl = winning_region_.getNrOfClauses();
451  size_t new_lits = winning_region_.getNrOfLits();;
452  L_DBG("reduceExistingClauses(): " << or_cl << " --> " << new_cl << " clauses.");
453  L_DBG("reduceExistingClauses(): " << or_lits << " --> " << new_lits << " literals.");
454  statistics_.notifyAfterCubeMin(0, 0);
455 }

502 void LearnSynthQBFInc::restrictToStates(vector<int> &vec) const
503 {
504  for(size_t cnt = 0; cnt < vec.size(); ++cnt)
505  {
506  int var = (vec[cnt] < 0) ? -vec[cnt] : vec[cnt];
507  // We can always drop the error-bit of the state space (because we can be sure that all
508  // states with error=1 have already been removed from the winning region). This saves
509  // one iteration during the generalization.
510  if(var == VarManager::instance().getPresErrorStateVar() ||
511  VarManager::instance().getInfo(var).getKind() != VarInfo::PRES_STATE)
512  {
513  vec[cnt] = vec[vec.size() - 1];
514  vec.pop_back();
515  cnt--;
516  }
517  }
