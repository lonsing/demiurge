40 LearnSynthQBFInc::LearnSynthQBFInc(CNFImplExtractor *impl_extractor) :
41  BackEnd(),
42  impl_extractor_(impl_extractor)
43 {
44  // build the quantifier prefix for checking for counterexamples:
45  // check_quant_ = exists x,i: forall c: exists x',tmp:
46  // (we misuse the type TEMPL_PARAMS for activation variables we may use)
47  check_quant_.push_back(make_pair(VarInfo::TEMPL_PARAMS, QBFSolver::E));
48  check_quant_.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
49  check_quant_.push_back(make_pair(VarInfo::INPUT, QBFSolver::E));
50  check_quant_.push_back(make_pair(VarInfo::CTRL, QBFSolver::A));
51  check_quant_.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
52  check_quant_.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
53 
54  // build the quantifier prefix for generalizing cubes:
55  // gen_quant_ = exists x: forall i: exists c,x',tmp:
56  gen_quant_.push_back(make_pair(VarInfo::PRES_STATE, QBFSolver::E));
57  gen_quant_.push_back(make_pair(VarInfo::INPUT, QBFSolver::A));
58  gen_quant_.push_back(make_pair(VarInfo::CTRL, QBFSolver::E));
59  gen_quant_.push_back(make_pair(VarInfo::NEXT_STATE, QBFSolver::E));
60  gen_quant_.push_back(make_pair(VarInfo::TMP, QBFSolver::E));
61 
62 
63 }

73 bool LearnSynthQBFInc::run()
74 {
75  VarManager::instance().push();
76  //L_DBG(VarManager::instance().toString());
77  //L_DBG("TRANS:");
78  //L_DBG(AIG2CNF::instance().getTrans().toString());
79  L_INF("Starting to compute a winning region ...");
80  statistics_.notifyWinRegStart();
81  bool realizable = computeWinningRegion();
82  statistics_.notifyWinRegEnd();
83  if(!realizable)
84  {
85  L_RES("The specification is unrealizable.");
86  statistics_.logStatistics();
87  return false;
88  }
89  L_RES("The specification is realizable.");
90  Utils::debugCheckWinReg(winning_region_);
91  if(Options::instance().doRealizabilityOnly())
92  {
93  statistics_.logStatistics();
94  return true;
95  }
96 
97  L_INF("Starting to extract a circuit ...");
98  impl_extractor_->extractCircuit(winning_region_);
99  L_INF("Synthesis done.");
100  statistics_.logStatistics();
101  impl_extractor_->logStatistics();
102  return true;
103 }

122 bool LearnSynthQBFInc::computeWinningRegionOne()
123 {
124  AIG2CNF& A2C = AIG2CNF::instance();
125  VarManager &VM = VarManager::instance();
126  L_DBG("Nr. of state variables: " << VM.getVarsOfType(VarInfo::PRES_STATE).size());
127  L_DBG("Nr. of uncontrollable inputs: " << VM.getVarsOfType(VarInfo::INPUT).size());
128  L_DBG("Nr. of controllable inputs: " << VM.getVarsOfType(VarInfo::CTRL).size());
129 
130  winning_region_.clear();
131  winning_region_ = A2C.getSafeStates();
132  winning_region_large_.clear();
133  winning_region_large_ = A2C.getSafeStates();
134 
135  bool precise = true;
136  // build check_cnf = P(x) AND T(x,i,c,x') AND NOT P(x')
137  recomputeCheckCNF();
138  solver_check_.startIncrementalSession(check_quant_);
139  solver_check_.incAddCNF(check_cnf_);
140  solver_check_.doMinCores(false);
141  // build generalize_clause_cnf = P(x) AND T(x,i,c,x') AND P(x')
142  recomputeGenCNF();
143  solver_gen_.startIncrementalSession(gen_quant_);
144  solver_gen_.incAddCNF(generalize_clause_cnf_);
145  solver_gen_.doMinCores(false);
146 
147  vector<int> counterexample;
148  vector<int> blocking_clause;
149  blocking_clause.reserve(VM.getVarsOfType(VarInfo::PRES_STATE).size());
150 
151  size_t count = 0;
152  while(true)
153  {
154  count++;
155  bool ce_found = computeCounterexampleQBF(counterexample);
156  if(!ce_found)
157  {
158  if(precise)
159  return true;
160  Utils::compressStateCNF(winning_region_);
161  recomputeCheckCNF();
162  solver_check_.startIncrementalSession(check_quant_);
163  solver_check_.incAddCNF(check_cnf_);
164  precise = true;
165  continue;
166  }
167 
168  L_DBG("Found bad state (nr " << count << "), excluding it from the winning region.");
169  bool realizable = computeBlockingClause(counterexample, blocking_clause);
170  if(!realizable)
171  return false;
172 
173  // update the CNFs:
174  Utils::debugPrint(blocking_clause, "Blocking clause: ");
175  winning_region_.addClauseAndSimplify(blocking_clause);
176  winning_region_large_.addClauseAndSimplify(blocking_clause);
177  solver_check_.incAddClause(blocking_clause);
178  solver_gen_.incAddClause(blocking_clause);
179  Utils::swapPresentToNext(blocking_clause);
180  solver_gen_.incAddClause(blocking_clause);
181 
182  precise = false;
183 
184  if(count % 100 == 0)
185  {
186  L_DBG("Statistics at count=" << count);
187  L_DBG("Winning region clauses: " << winning_region_.getNrOfClauses());
188  L_DBG("Winning region lits: " << winning_region_.getNrOfLits());
189  L_DBG("Winning region full clauses: " << winning_region_large_.getNrOfClauses());
190  L_DBG("Winning region full lits: " << winning_region_large_.getNrOfLits());
191  if(Logger::instance().isEnabled(Logger::DBG))
192  statistics_.logStatistics();
193  }
194  }
195  return true;
196 }

273 bool LearnSynthQBFInc::computeWinningRegionOnePool()
274 {
275  AIG2CNF& A2C = AIG2CNF::instance();
276  VarManager &VM = VarManager::instance();
277  L_DBG("Nr. of state variables: " << VM.getVarsOfType(VarInfo::PRES_STATE).size());
278  L_DBG("Nr. of uncontrollable inputs: " << VM.getVarsOfType(VarInfo::INPUT).size());
279  L_DBG("Nr. of controllable inputs: " << VM.getVarsOfType(VarInfo::CTRL).size());
280 
281  winning_region_.clear();
282  winning_region_ = A2C.getSafeStates();
283  winning_region_large_.clear();
284  winning_region_large_ = A2C.getSafeStates();
285 
286  // build check_cnf = P(x) AND T(x,i,c,x') AND NOT P(x')
287  int pool_size = 20;
288  negcl_var_pool_.clear();
289  negcl_var_pool_.reserve(pool_size);
290  negcl_var_pool_act_.clear();
291  negcl_var_pool_act_.reserve(pool_size);
292  for(int cnt = 0; cnt < pool_size; ++cnt)
293  {
294  negcl_var_pool_.push_back(VM.createFreshTmpVar());
295  negcl_var_pool_act_.push_back(VM.createFreshTemplParam());
296  }
297  solver_check_.startIncrementalSession(check_quant_);
298  solver_check_.incAddCNF(AIG2CNF::instance().getTrans());
299  solver_check_.incAddUnitClause(-VM.getPresErrorStateVar());
300  for(size_t cnt = 0; cnt < negcl_var_pool_act_.size(); ++cnt)
301  solver_check_.incAdd2LitClause(-negcl_var_pool_act_[cnt], -negcl_var_pool_[cnt]);
302  vector<int> next_unsafe;
303  next_unsafe.reserve(negcl_var_pool_.size() + 1);
304  next_unsafe.push_back(VM.getNextErrorStateVar());
305  next_unsafe.insert(next_unsafe.end(), negcl_var_pool_.begin(), negcl_var_pool_.end());
306  solver_check_.incAddClause(next_unsafe);
307 
308 
309  // build generalize_clause_cnf = P(x) AND T(x,i,c,x') AND P(x')
310  solver_gen_.startIncrementalSession(gen_quant_);
311  solver_gen_.incAddCNF(AIG2CNF::instance().getTrans());
312  solver_gen_.incAddUnitClause(-VM.getPresErrorStateVar());
313  solver_gen_.incAddUnitClause(-VM.getNextErrorStateVar());
314  solver_gen_.doMinCores(false);
315 
316  vector<int> counterexample;
317  vector<int> blocking_clause;
318  blocking_clause.reserve(VM.getVarsOfType(VarInfo::PRES_STATE).size());
319 
320  size_t count = 0;
321  while(true)
322  {
323  count++;
324 
325  bool ce_found = computeCounterexampleQBFPool(counterexample);
326 
327  if(!ce_found)
328  return true;
329 
330  L_DBG("Found bad state (nr " << count << "), excluding it from the winning region.");
331  bool realizable = computeBlockingClause(counterexample, blocking_clause);
332  if(!realizable)
333  return false;
334 
335  // update the CNFs:
336  Utils::debugPrint(blocking_clause, "Blocking clause: ");
337  winning_region_.addClauseAndSimplify(blocking_clause);
338  winning_region_large_.addClauseAndSimplify(blocking_clause);
339  vector<int> next_blocking_clause(blocking_clause);
340  Utils::swapPresentToNext(next_blocking_clause);
341  solver_gen_.incAddClause(blocking_clause);
342  solver_gen_.incAddClause(next_blocking_clause);
343  if(negcl_var_pool_.empty())
344  {
345  VM.resetToLastPush();
346  Utils::compressStateCNF(winning_region_);
347  negcl_var_pool_.clear();
348  negcl_var_pool_.reserve(pool_size);
349  negcl_var_pool_act_.clear();
350  negcl_var_pool_act_.reserve(pool_size);
351  for(int cnt = 0; cnt < pool_size; ++cnt)
352  {
353  negcl_var_pool_.push_back(VM.createFreshTmpVar());
354  negcl_var_pool_act_.push_back(VM.createFreshTemplParam());
355  }
356  vector<int> next_unsafe;
357  next_unsafe.reserve(negcl_var_pool_.size() + winning_region_.getNrOfClauses());
358  next_unsafe.insert(next_unsafe.end(), negcl_var_pool_.begin(), negcl_var_pool_.end());
359 
360  CNF check_cnf(AIG2CNF::instance().getTrans());
361  check_cnf.addCNF(winning_region_);
362  CNF next_win_reg(winning_region_);
363  next_win_reg.swapPresentToNext();
364  const list<vector<int> >& cl = next_win_reg.getClauses();
365  for(CNF::ClauseConstIter it = cl.begin(); it != cl.end(); ++it)
366  {
367  if(it->size() == 1)
368  next_unsafe.push_back(-(*it)[0]);
369  else
370  {
371  int clause_false_lit = VarManager::instance().createFreshTmpVar();
372  next_unsafe.push_back(clause_false_lit);
373  for(size_t lit_cnt = 0; lit_cnt < it->size(); ++lit_cnt)
374  check_cnf.add2LitClause(-clause_false_lit, -((*it)[lit_cnt]));
375  }
376  }
377  for(size_t cnt = 0; cnt < negcl_var_pool_act_.size(); ++cnt)
378  check_cnf.add2LitClause(-negcl_var_pool_act_[cnt], -negcl_var_pool_[cnt]);
379  check_cnf.addClause(next_unsafe);
380  solver_check_.startIncrementalSession(check_quant_);
381  solver_check_.incAddCNF(check_cnf);
382  }
383  else
384  {
385  solver_check_.incAddClause(blocking_clause);
386  int not_cl = negcl_var_pool_.back();
387  negcl_var_pool_.pop_back();
388  negcl_var_pool_act_.pop_back();
389  for(size_t lit_cnt = 0; lit_cnt < next_blocking_clause.size(); ++lit_cnt)
390  solver_check_.incAdd2LitClause(-not_cl, -next_blocking_clause[lit_cnt]);
391  }
392 
393  if(count % 100 == 0)
394  {
395  L_DBG("Statistics at count=" << count);
396  L_DBG("Winning region clauses: " << winning_region_.getNrOfClauses());
397  L_DBG("Winning region lits: " << winning_region_.getNrOfLits());
398  L_DBG("Winning region full clauses: " << winning_region_large_.getNrOfClauses());
399  L_DBG("Winning region full lits: " << winning_region_large_.getNrOfLits());
400  if(Logger::instance().isEnabled(Logger::DBG))
401  statistics_.logStatistics();
402  }
403  }
404  return true;
405 }

565 bool LearnSynthQBFInc::computeWinningRegionAllPool()
566 {
567  AIG2CNF& A2C = AIG2CNF::instance();
568  VarManager &VM = VarManager::instance();
569  L_DBG("Nr. of state variables: " << VM.getVarsOfType(VarInfo::PRES_STATE).size());
570  L_DBG("Nr. of uncontrollable inputs: " << VM.getVarsOfType(VarInfo::INPUT).size());
571  L_DBG("Nr. of controllable inputs: " << VM.getVarsOfType(VarInfo::CTRL).size());
572 
573  winning_region_.clear();
574  winning_region_ = A2C.getSafeStates();
575  winning_region_large_.clear();
576  winning_region_large_ = A2C.getSafeStates();
577 
578  // build check_cnf = P(x) AND T(x,i,c,x') AND NOT P(x')
579  int pool_size = 50;
580  negcl_var_pool_.clear();
581  negcl_var_pool_.reserve(pool_size);
582  negcl_var_pool_act_.clear();
583  negcl_var_pool_act_.reserve(pool_size);
584  for(int cnt = 0; cnt < pool_size; ++cnt)
585  {
586  negcl_var_pool_.push_back(VM.createFreshTmpVar());
587  negcl_var_pool_act_.push_back(VM.createFreshTemplParam());
588  }
589  solver_check_.startIncrementalSession(check_quant_);
590  solver_check_.incAddCNF(AIG2CNF::instance().getTrans());
591  solver_check_.incAddUnitClause(-VM.getPresErrorStateVar());
592  for(size_t cnt = 0; cnt < negcl_var_pool_act_.size(); ++cnt)
593  solver_check_.incAdd2LitClause(-negcl_var_pool_act_[cnt], -negcl_var_pool_[cnt]);
594  vector<int> next_unsafe;
595  next_unsafe.reserve(negcl_var_pool_.size() + 1);
596  next_unsafe.push_back(VM.getNextErrorStateVar());
597  next_unsafe.insert(next_unsafe.end(), negcl_var_pool_.begin(), negcl_var_pool_.end());
598  solver_check_.incAddClause(next_unsafe);
599 
600  // build generalize_clause_cnf = P(x) AND T(x,i,c,x') AND P(x')
601  solver_gen_.startIncrementalSession(gen_quant_);
602  solver_gen_.incAddCNF(AIG2CNF::instance().getTrans());
603  solver_gen_.incAddUnitClause(-VM.getPresErrorStateVar());
604  solver_gen_.incAddUnitClause(-VM.getNextErrorStateVar());
605  solver_gen_.doMinCores(true);
606 
607  vector<int> counterexample;
608  vector<vector<int> > all_blocking_clauses;
609  all_blocking_clauses.reserve(1000);
610 
611  size_t count = 0;
612  while(true)
613  {
614  count++;
615  bool ce_found = computeCounterexampleQBFPool(counterexample);
616  if(!ce_found)
617  return true;
618 
619  L_DBG("Found bad state (nr " << count << "), excluding it from the winning region.");
620  bool realizable = computeAllBlockingClauses(counterexample, all_blocking_clauses);
621  if(!realizable)
622  return false;
623 
624  // update the CNFs and the solver_gen_:
625  for(size_t clause_count = 0; clause_count < all_blocking_clauses.size(); ++clause_count)
626  {
627  vector<int> &blocking_clause = all_blocking_clauses[clause_count];
628  Utils::debugPrint(blocking_clause, "Blocking clause: ");
629  winning_region_.addClauseAndSimplify(blocking_clause);
630  winning_region_large_.addClauseAndSimplify(blocking_clause);
631  vector<int> next_blocking_clause(blocking_clause);
632  Utils::swapPresentToNext(next_blocking_clause);
633  solver_gen_.incAddClause(blocking_clause);
634  solver_gen_.incAddClause(next_blocking_clause);
635  }
636 
637  // update the solver_check_:
638  if(negcl_var_pool_.size() < all_blocking_clauses.size())
639  {
640  VM.resetToLastPush();
641  reduceExistingClauses();
642  Utils::compressStateCNF(winning_region_);
643  negcl_var_pool_.clear();
644  negcl_var_pool_.reserve(pool_size);
645  negcl_var_pool_act_.clear();
646  negcl_var_pool_act_.reserve(pool_size);
647  for(int cnt = 0; cnt < pool_size; ++cnt)
648  {
649  negcl_var_pool_.push_back(VM.createFreshTmpVar());
650  negcl_var_pool_act_.push_back(VM.createFreshTemplParam());
651  }
652  vector<int> next_unsafe;
653  next_unsafe.reserve(negcl_var_pool_.size() + winning_region_.getNrOfClauses());
654  next_unsafe.insert(next_unsafe.end(), negcl_var_pool_.begin(), negcl_var_pool_.end());
655 
656  CNF check_cnf(AIG2CNF::instance().getTrans());
657  check_cnf.addCNF(winning_region_);
658  CNF next_win_reg(winning_region_);
659  next_win_reg.swapPresentToNext();
660  const list<vector<int> >& cl = next_win_reg.getClauses();
661  for(CNF::ClauseConstIter it = cl.begin(); it != cl.end(); ++it)
662  {
663  if(it->size() == 1)
664  next_unsafe.push_back(-(*it)[0]);
665  else
666  {
667  int clause_false_lit = VarManager::instance().createFreshTmpVar();
668  next_unsafe.push_back(clause_false_lit);
669  for(size_t lit_cnt = 0; lit_cnt < it->size(); ++lit_cnt)
670  check_cnf.add2LitClause(-clause_false_lit, -((*it)[lit_cnt]));
671  }
672  }
673  for(size_t cnt = 0; cnt < negcl_var_pool_act_.size(); ++cnt)
674  check_cnf.add2LitClause(-negcl_var_pool_act_[cnt], -negcl_var_pool_[cnt]);
675  check_cnf.addClause(next_unsafe);
676  solver_check_.startIncrementalSession(check_quant_);
677  solver_check_.incAddCNF(check_cnf);
678  }
679  else
680  {
681  for(size_t clause_count = 0; clause_count < all_blocking_clauses.size(); ++clause_count)
682  {
683  vector<int> &blocking_clause = all_blocking_clauses[clause_count];
684  vector<int> next_blocking_clause(blocking_clause);
685  Utils::swapPresentToNext(next_blocking_clause);
686  solver_check_.incAddClause(blocking_clause);
687  int not_cl = negcl_var_pool_.back();
688  negcl_var_pool_.pop_back();
689  negcl_var_pool_act_.pop_back();
690  for(size_t lit_cnt = 0; lit_cnt < next_blocking_clause.size(); ++lit_cnt)
691  solver_check_.incAdd2LitClause(-not_cl, -next_blocking_clause[lit_cnt]);
692  }
693  }
694 
695  if(count % 100 == 0)
696  {
697  L_DBG("Statistics at count=" << count);
698  L_DBG("Winning region clauses: " << winning_region_.getNrOfClauses());
699  L_DBG("Winning region lits: " << winning_region_.getNrOfLits());
700  L_DBG("Winning region full clauses: " << winning_region_large_.getNrOfClauses());
701  L_DBG("Winning region full lits: " << winning_region_large_.getNrOfLits());
702  if(Logger::instance().isEnabled(Logger::DBG))
703  statistics_.logStatistics();
704  }
705  }
706  return true;

733 bool LearnSynthQBFInc::computeBlockingClause(vector<int> &ce, vector<int> &blocking_clause)
734 {
735  statistics_.notifyBeforeCubeMin();
736  Utils::randomize(ce);
737  if(Utils::containsInit(ce))
738  return false;
739  size_t orig_size = ce.size();
740  bool worked = generalizeCounterexample(ce);
741  MASSERT(worked, "Impossible.");

743  return false;

745  // It pays of to add the blocking clause to generalize_clause_cnf_ and do a second
746  // iteration:
747  blocking_clause = ce;

750  Utils::swapPresentToNext(blocking_clause);

752  // TODO: maybe no core but explicit loop instead of the next line:
753  generalizeCounterexampleFurther(ce);
754  if(Utils::containsInit(ce))
755  return false;
756  // END minimize further
757  blocking_clause = ce;
758  Utils::negateLiterals(blocking_clause);
759  L_DBG("computeBlockingClause(): " << orig_size << " --> " << blocking_clause.size());
760  statistics_.notifyAfterCubeMin(orig_size, blocking_clause.size());

764 // -------------------------------------------------------------------------------------------
765 bool LearnSynthQBFInc::computeAllBlockingClauses(vector<int> &ce,
766  vector<vector<int> > &blocking_clauses)
767 {
768  statistics_.notifyBeforeCubeMin();
769  list<set<int> > must_not_contain_queue;
770  list<vector<int> > gen_ces;
771  vector<int> orig_ce(ce);
772  vector<int> first_ce(orig_ce);
773  bool worked = generalizeCounterexample(first_ce);
774  MASSERT(worked, "Impossible");
775  statistics_.notifyCubeMin(orig_ce.size(), first_ce.size());
776  if(Utils::containsInit(first_ce))
777  return false;
778  blocking_clauses.clear();
779  vector<int> blocking_clause(first_ce);
780  Utils::negateLiterals(blocking_clause);
781  blocking_clauses.push_back(blocking_clause);
782 
783  gen_ces.push_back(first_ce);
784  for(size_t cnt = 0; cnt < first_ce.size(); ++cnt)
785  {
786  set<int> singleton_set;
787  singleton_set.insert(first_ce[cnt]);
788  must_not_contain_queue.push_back(singleton_set);
789  }
790 
791  while(!must_not_contain_queue.empty())
792  {
793  set<int> must_not_contain = must_not_contain_queue.front();
794  must_not_contain_queue.pop_front();
795 
796  // check if we already have a ce in gen_ces which satisfies our needs:
797  bool found = false;
798  for(list<vector<int> >::const_iterator it = gen_ces.begin(); it != gen_ces.end(); ++it)
799  {
800  if(Utils::intersectionEmpty(*it, must_not_contain))
801  {
802  found = true;
803  break;
804  }
805  }
806  if(found)
807  continue;
808 
809  // if not: we compute one
810  vector<int> new_ce(orig_ce);
811  set<int>::const_iterator it = must_not_contain.begin();
812  for(; it != must_not_contain.end(); ++it)
813  Utils::remove(new_ce, *it);
814  size_t orig_size = new_ce.size();
815  bool is_ok = generalizeCounterexample(new_ce);
816  statistics_.notifyCubeMin(orig_size, new_ce.size());
817  if(is_ok)
818  {
819  if(Utils::containsInit(new_ce))
820  return false;
821  gen_ces.push_back(new_ce);
822  blocking_clause = new_ce;
823  Utils::negateLiterals(blocking_clause);
824  blocking_clauses.push_back(blocking_clause);
825  for(size_t cnt = 0; cnt < new_ce.size(); ++cnt)
826  {
827  set<int> new_must_not_contain(must_not_contain);
828  new_must_not_contain.insert(new_ce[cnt]);
829  must_not_contain_queue.push_back(new_must_not_contain);
830  }
831  }
832  }
833 
834  // BEGIN minimize further
835  // In order to reduce the size of the clauses further, we can add all we have
836  // learned meanwhile to solver_gen_ and see if we can reduce the clauses
837  // further:
838  for(size_t cl_cnt = 0; cl_cnt < blocking_clauses.size(); ++cl_cnt)
839  {
840  vector<int> new_blocking_clause(blocking_clauses[cl_cnt]);
841  solver_gen_.incAddClause(new_blocking_clause);
842  Utils::swapPresentToNext(new_blocking_clause);
843  solver_gen_.incAddClause(new_blocking_clause);
844  }
845  for(size_t cl_cnt = 0; cl_cnt < blocking_clauses.size(); ++cl_cnt)
846  {
847  vector<int> min_ce(blocking_clauses[cl_cnt]);
848  Utils::negateLiterals(min_ce);
849  size_t size_before_min = min_ce.size();
850  // TODO: maybe no core but explicit loop instead of the next line:
851  generalizeCounterexampleFurther(min_ce);
852  if(min_ce.size() < size_before_min)
853  {
854  Utils::negateLiterals(min_ce);
855  blocking_clauses[cl_cnt] = min_ce;
856  }
857  }
858  // END minimize further
859  statistics_.notifyAfterCubeMin();
860 
861  return true;
862 }
