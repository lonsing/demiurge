61 inline Lit c2m(Solver &s, int lit)
62 {
63  int var = lit < 0 ? -lit : lit;
64  while(var >= s.nVars())
65  s.newVar();
66  int minisat_lit = var + var;
67  if(lit < 0)
68  minisat_lit += 1;
69  Lit p;
70  p.x = minisat_lit;
71  return p;
72 }

108 void randModel(Solver &solver, const vec<Lit> &assumptions, vector<int> &model)
109 {
110  vec<Lit> ass_mod(assumptions.size() + model.size());
111  for(int ass_cnt = 0; ass_cnt < assumptions.size(); ++ass_cnt)
112  ass_mod[ass_cnt] = assumptions[ass_cnt];
113 
114  for(size_t cnt = 0; cnt < model.size(); ++cnt)
115  {
116  if(rand() % 2 == 0)
117  {
118  vector<int> mutated_model(model);
119  mutated_model[cnt] = -mutated_model[cnt];
120  for(size_t mod_cnt = 0; mod_cnt < mutated_model.size(); ++mod_cnt)
121  ass_mod[mod_cnt + assumptions.size()] = c2m(solver, mutated_model[mod_cnt]);
122  if(solver.solve(ass_mod))
123  model = mutated_model;
124  }
125  }
126 }

129 bool MiniSatApi::isSat(const CNF &cnf)
130 {
131  Solver solver;
132  const list<vector<int> > &clauses = cnf.getClauses();
133  for(CNF::ClauseConstIter it = clauses.begin(); it != clauses.end(); ++it)
134  {
135  vec<Lit> clause(it->size());
136  for(size_t lit_cnt = 0; lit_cnt < it->size(); ++lit_cnt)
137  clause[lit_cnt] = c2m(solver, (*it)[lit_cnt]);
138  solver.addClause(clause);
139  }
140  return solver.solve();

144 bool MiniSatApi::isSatModelOrCore(const CNF &cnf,
145  const vector<int> &assumptions,
146  const vector<int> &vars_of_interest,
147  vector<int> &model_or_core)
148 {
149 
150  Solver solver;
151  const list<vector<int> > &clauses = cnf.getClauses();
152  for(CNF::ClauseConstIter it = clauses.begin(); it != clauses.end(); ++it)
153  {
154  vec<Lit> clause(it->size());
155  for(size_t lit_cnt = 0; lit_cnt < it->size(); ++lit_cnt)
156  clause[lit_cnt] = c2m(solver, (*it)[lit_cnt]);
157  solver.addClause(clause);
158  }
159 
160  vec<Lit> ass(assumptions.size());
161  for(size_t ass_cnt = 0; ass_cnt < assumptions.size(); ++ass_cnt)
162  ass[ass_cnt] = c2m(solver, assumptions[ass_cnt]);
163 
164  bool sat = solver.solve(ass);
165  if(sat)
166  {
167  model_or_core.clear();
168  model_or_core.reserve(vars_of_interest.size());
169  for(size_t var_cnt = 0; var_cnt < vars_of_interest.size(); ++var_cnt)
170  {
171  if(solver.modelValue(c2m(solver, vars_of_interest[var_cnt])) == l_True)
172  model_or_core.push_back(vars_of_interest[var_cnt]);
173  else
174  model_or_core.push_back(-vars_of_interest[var_cnt]);
175  }
176  if(rand_models_)
177  randModel(solver, ass, model_or_core);
178  return true;
179  }
180  else
181  {
182  model_or_core.clear();
183  model_or_core.reserve(solver.conflict.size());
184  for(int ass_cnt = 0; ass_cnt < solver.conflict.size(); ++ass_cnt)
185  model_or_core.push_back(m2c(solver.conflict[ass_cnt]));
186  if(min_cores_)
187  {
188  vector<int> orig_core(model_or_core);
189  for(size_t lit_cnt = 0; lit_cnt < orig_core.size(); ++lit_cnt)
190  {
191  vector<int> tmp(model_or_core);
192  Utils::remove(tmp, orig_core[lit_cnt]);
193  // TODO: make incremental:
194  CNF min(cnf);
195  for(size_t tmp_cnt = 0; tmp_cnt < tmp.size(); ++tmp_cnt)
196  min.add1LitClause(tmp[tmp_cnt]);
197  if(!isSat(min))
198  model_or_core = tmp;
199  }
200  }
201  return false;
202  }
203 }

412  for(size_t ass_cnt = 0; ass_cnt < more_assumptions.size(); ++ass_cnt)

414  for(size_t ass_cnt = 0; ass_cnt < core_assumptions.size(); ++ass_cnt)
415  ass[ass_cnt + more_assumptions.size()] = c2m(*solver, core_assumptions[ass_cnt]);
416  bool sat = solver->solve(ass);
417  if(sat)
418  {
419  model_or_core.clear();
420  model_or_core.reserve(vars_of_interest.size());
421  for(size_t var_cnt = 0; var_cnt < vars_of_interest.size(); ++var_cnt)

429  randModel(*solver, ass, model_or_core);
430  return true;
431  }
432  else
433  {

435  for(size_t cnt = 0; cnt < core_assumptions.size(); ++cnt)

454  vec<Lit> smaller_ass(tmp.size() + more_assumptions.size());
455  for(size_t ass_cnt = 0; ass_cnt < more_assumptions.size(); ++ass_cnt)
456  smaller_ass[ass_cnt] = c2m(*solver, more_assumptions[ass_cnt]);
